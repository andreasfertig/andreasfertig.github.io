96 OutputFormatHelper& CodeGenerator::LambdaScopeHandler::GetBuffer(OutputFormatHelper& outputFormatHelper) const
97 {
98  // Find the most outer element to place the lambda class definition. For example, if we have this:
99  // Test( [&]() {} );
100  // The lambda's class definition needs to be placed _before_ the CallExpr to Test.
101  auto* element = [&]() -> LambdaHelper* {
102  for(auto& l : mStack) {
103  switch(l.callerType()) {
104  case LambdaCallerType::CallExpr:
105  case LambdaCallerType::VarDecl:
106  case LambdaCallerType::ReturnStmt:
107  case LambdaCallerType::OperatorCallExpr:
108  case LambdaCallerType::MemberCallExpr:
109  case LambdaCallerType::BinaryOperator: return &l;
110  default: break;
111  }
112  }
113 
114  return nullptr;
115  }();
116 
117  if(element) {
118  return element->buffer();
119  }
120 
121  return outputFormatHelper;
122 }

134 void CodeGenerator::InsertArg(const CXXForRangeStmt* rangeForStmt)
135 {
136  mOutputFormatHelper.OpenScope();
137 
138  InsertArg(rangeForStmt->getRangeStmt());
139  InsertArg(rangeForStmt->getBeginStmt());
140  InsertArg(rangeForStmt->getEndStmt());
141 
142  // add blank line after the declarations
143  mOutputFormatHelper.AppendNewLine();
144 
145  mOutputFormatHelper.Append("for( ; ");
146 
147  InsertArg(rangeForStmt->getCond());
148 
149  mOutputFormatHelper.Append("; ");
150 
151  InsertArg(rangeForStmt->getInc());
152 
153  mOutputFormatHelper.AppendNewLine(" )");
154  // open for loop scope
155  mOutputFormatHelper.OpenScope();
156 
157  InsertArg(rangeForStmt->getLoopVariable());
158 
159  const auto* body = rangeForStmt->getBody();
160  const bool isBodyBraced = isa<CompoundStmt>(body);
161 
162  /* we already opened a scope. Skip the initial one */
163  if(!isBodyBraced) {
164  InsertArg(body);
165  } else {
166  HandleCompoundStmt(dyn_cast_or_null<CompoundStmt>(body));
167  }
168 
169  if(!isBodyBraced && !isa<NullStmt>(body)) {
170  mOutputFormatHelper.AppendNewLine(';');
171  }
172 
173  // close range-for scope in for
174  mOutputFormatHelper.CloseScope(OutputFormatHelper::NoNewLineBefore::Yes);
175 
176  // close outer range-for scope
177  mOutputFormatHelper.CloseScope();
178  mOutputFormatHelper.AppendNewLine();
179 }

317 void CodeGenerator::InsertArg(const MemberExpr* stmt)
318 {
319  const auto* base = stmt->getBase();
320  bool skipBase{false};
321  if(const auto* implicitCast = dyn_cast_or_null<ImplicitCastExpr>(base)) {
322  if(CastKind::CK_UncheckedDerivedToBase == implicitCast->getCastKind()) {
323  const auto* castDest = implicitCast->IgnoreImpCasts();
324 
325  // if this calls a protected function we cannot cast it to the base, this would not compile
326  if(isa<CXXThisExpr>(castDest)) {
327  skipBase = true;
328  }
329  }
330  }
331 
332  if(skipBase) {
333  mOutputFormatHelper.Append("/* ");
334  }
335 
336  InsertArg(base);
337 
338  const std::string op{stmt->isArrow() ? "->" : "."};
339  const auto* meDecl = stmt->getMemberDecl();
340  bool skipTemplateArgs{false};
341  const auto name = [&]() -> std::string {
342  // Handle a special case where we have a lambda static invoke operator. In that case use the appropriate using
343  // retType as return type
344  if(const auto* m = dyn_cast_or_null<CXXMethodDecl>(meDecl)) {
345  if(const auto* rd = m->getParent(); rd && rd->isLambda()) {
346  skipTemplateArgs = true;
347 
348  return StrCat("operator ", GetLambdaName(*rd), "::retType");
349  }
350  }
351 
352  return stmt->getMemberNameInfo().getName().getAsString();
353  }();
354 
355  mOutputFormatHelper.Append(op);
356 
357  if(skipBase) {
358  mOutputFormatHelper.Append(" */ ");
359  }
360 
361  mOutputFormatHelper.Append(name);
362 
363  if(!skipTemplateArgs) {
364  if(const auto cxxMethod = dyn_cast_or_null<CXXMethodDecl>(meDecl)) {
365  if(const auto* tmplArgs = cxxMethod->getAsFunction()->getTemplateSpecializationArgs()) {
366  OutputFormatHelper ofm{};
367 
368  ofm.Append('<');
369 
370  bool haveArg{false};
371  bool first{true};
372  for(const auto& arg : tmplArgs->asArray()) {
373 
374  if(arg.getKind() == TemplateArgument::Integral) {
375  if(first) {
376  first = false;
377  } else {
378  ofm.Append(", ");
379  }
380 
381  ofm.Append(arg.getAsIntegral());
382  haveArg = true;
383  } else {
384 
385  break;
386  }
387  }
388 
389  if(haveArg) {
390  mOutputFormatHelper.Append(ofm.GetString(), ">");
391  }
392  }
393  }
394  }
395 }

504 {
505  const auto baseVarName{[&]() {
506  if(const auto* declName = FindDeclRef(decompositionDeclStmt->getInit())) {
507  std::string name = GetPlainName(*declName);
508 
509  const std::string operatorName{"operator"};
510  if(name.find(operatorName) != std::string::npos) {
511  return operatorName;
512  }
513 
514  return name;
515  }
516 
517  // We approached an unnamed decl. This happens for example like this: auto& [x, y] = Point{};
518  return std::string{""};
519  }()};
520 
521  const std::string tmpVarName{
522  BuildInternalVarName(baseVarName, GetBeginLoc(decompositionDeclStmt), GetSM(*decompositionDeclStmt))};
523 
524  mOutputFormatHelper.Append(GetTypeNameAsParameter(decompositionDeclStmt->getType(), tmpVarName), " = ");
525 
526  InsertArg(decompositionDeclStmt->getInit());
527 
528  mOutputFormatHelper.AppendNewLine(';');
529 
530  for(const auto* bindingDecl : decompositionDeclStmt->bindings()) {
531  if(const auto* binding = bindingDecl->getBinding()) {
532 
533  DPrint("sb name: %s\n", GetName(binding->getType()));
534 
535  const auto* holdingVarOrMemberExpr = [&]() -> const Expr* {
536  if(const auto* holdingVar = bindingDecl->getHoldingVar()) {
537  return holdingVar->getAnyInitializer();
538  }
539 
540  return dyn_cast_or_null<MemberExpr>(binding);
541  }();
542 
543  const std::string refOrRefRef = [&]() -> std::string {
544  const bool isRefToObject{IsReference(*decompositionDeclStmt)};
545  const bool isArrayBinding{isa<ArraySubscriptExpr>(binding) && isRefToObject};
546  const bool isNotTemporary{holdingVarOrMemberExpr && !isa<ExprWithCleanups>(holdingVarOrMemberExpr)};
547  if(isArrayBinding || isNotTemporary) {
548  return "&";
549  }
550 
551  return "";
552  }();
553 
554  mOutputFormatHelper.Append(GetName(bindingDecl->getType()), refOrRefRef, " ", GetName(*bindingDecl), " = ");
555 
556  // tuple decomposition
557  if(holdingVarOrMemberExpr) {
558  DPrint("4444\n");
559 
560  StructuredBindingsCodeGenerator structuredBindingsCodeGenerator{mOutputFormatHelper, tmpVarName};
561  CodeGenerator& codeGenerator = structuredBindingsCodeGenerator;
562  codeGenerator.InsertArg(holdingVarOrMemberExpr);
563 
564  // array decomposition
565  } else if(const auto* arraySubscription = dyn_cast_or_null<ArraySubscriptExpr>(binding)) {
566  mOutputFormatHelper.Append(tmpVarName);
567 
568  InsertArg(arraySubscription);
569 
570  } else {
571  TODO(bindingDecl, mOutputFormatHelper);
572  }

575  }

644  } else {

664  mOutputFormatHelper.Append(GetTypeNameAsParameter(stmt->getType(), varName));
665  }

1142  const bool isCXXMethod{callee && isa<CXXMethodDecl>(callee->getDecl())};
1143

1145  const auto* param1 = dyn_cast_or_null<DeclRefExpr>(stmt->getArg(0)->IgnoreImpCasts());
1146  const auto* param2 = dyn_cast_or_null<DeclRefExpr>(stmt->getArg(1)->IgnoreImpCasts());
1147 
1148  if(callee && param1 && param2) {
1149 
1150  const std::string replace = [&]() {
1151  if(isa<CXXMethodDecl>(callee->getDecl())) {
1152  const std::string tmpl = [&] {

1154  OutputFormatHelper outputFormatHelper;
1155  CodeGenerator codeGenerator{outputFormatHelper};
1156 
1157  codeGenerator.InsertTemplateArgs(tvd->getTemplateArgs().asArray());
1158 
1159  return outputFormatHelper.GetString();
1160  }
1161 
1162  return std::string{};
1163  }();
1164 
1165  return StrCat(GetName(*param1), tmpl, ".", GetName(*callee), "(", GetName(*param2), ")");
1166  } else {
1167  return StrCat(GetName(*callee), "(", GetName(*param1), ", ", GetName(*param2), ")");
1168  }
1169  }();
1170 
1171  mOutputFormatHelper.Append(replace);
1172 
1173  return;
1174  }
1175  }
1176

1179

1181  // skip arg0
1182  ++cb;
1183

1185  ++cb;
1186 
1187  // operators in a namespace but outside a class so operator goes first
1188  if(!isCXXMethod) {
1189  if(callee) {
1190  mOutputFormatHelper.Append(GetName(*callee), "(");
1191  } else {
1192  InsertArg(stmt->getCallee()->IgnoreImpCasts());
1193  mOutputFormatHelper.Append("(");
1194  }
1195  }
1196

1199  InsertArgWithParensIfNeeded(fallbackArg0);
1200 
1201  } else {
1202  InsertArgWithParensIfNeeded(arg1);
1203  }
1204

1206  if(isCXXMethod) {
1207  const OverloadedOperatorKind opKind = stmt->getOperator();
1208 
1209  mOutputFormatHelper.Append(".operator", getOperatorSpelling(opKind), "(");
1210  }
1211

1213  const auto childRange = llvm::make_range(cb, stmt->child_end());
1214
