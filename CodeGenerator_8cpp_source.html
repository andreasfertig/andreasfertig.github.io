160 OutputFormatHelper& CodeGenerator::LambdaScopeHandler::GetBuffer(OutputFormatHelper& outputFormatHelper) const
161 {
162  // Find the most outer element to place the lambda class definition. For example, if we have this:
163  // Test( [&]() {} );
164  // The lambda's class definition needs to be placed _before_ the CallExpr to Test.
165  auto* element = [&]() -> LambdaHelper* {
166  for(auto& l : mStack) {
167  switch(l.callerType()) {
168  case LambdaCallerType::CallExpr:
169  case LambdaCallerType::VarDecl:
170  case LambdaCallerType::ReturnStmt:
171  case LambdaCallerType::OperatorCallExpr:
172  case LambdaCallerType::MemberCallExpr:
173  case LambdaCallerType::BinaryOperator: return &l;
174  default: break;
175  }
176  }
177 
178  return nullptr;
179  }();
180 
181  if(element) {
182  return element->buffer();
183  }
184 
185  return outputFormatHelper;
186 }

219 void CodeGenerator::InsertArg(const CXXForRangeStmt* rangeForStmt)
220 {
221  auto& langOpts{GetLangOpts(*rangeForStmt->getLoopVariable())};
222  const bool onlyCpp11{not langOpts.CPlusPlus14};
223 
224  auto* rwStmt = const_cast<CXXForRangeStmt*>(rangeForStmt);
225 
226  std::vector<Stmt*> outerScopeStmts{};
227  std::vector<Stmt*> bodyStmts{};
228 
229  // C++20 init-statement
230  AddStmt(outerScopeStmts, rangeForStmt->getInit());
231 
232  // range statement
233  AddStmt(outerScopeStmts, rangeForStmt->getRangeStmt());
234 
235  if(not onlyCpp11) {
236  AddStmt(outerScopeStmts, rangeForStmt->getBeginStmt());
237  AddStmt(outerScopeStmts, rangeForStmt->getEndStmt());
238  }
239 
240  // add the loop variable to the body
241  AddStmt(bodyStmts, rangeForStmt->getLoopVarStmt());
242 
243  // add the body itself, without the CompoundStmt
244  AddBodyStmts(bodyStmts, rwStmt->getBody());
245 
246  const auto& ctx = rangeForStmt->getLoopVariable()->getASTContext();
247  Decl* decls[2]{rwStmt->getBeginStmt()->getSingleDecl(), rwStmt->getEndStmt()->getSingleDecl()};
248  auto dgRef = DeclGroupRef::Create(const_cast<ASTContext&>(ctx), decls, 2);
249 
250  auto* declStmt = [&]() -> DeclStmt* {
251  if(onlyCpp11) {
252  return new(ctx) DeclStmt(dgRef, rangeForStmt->getBeginLoc(), rangeForStmt->getEndLoc());
253  }
254 
255  return nullptr;
256  }();
257 
258  ArrayRef<Stmt*> innerScopeStmtsRef{bodyStmts};
259  auto* innerScope =
260  CompoundStmt::Create(ctx, innerScopeStmtsRef, rangeForStmt->getBeginLoc(), rangeForStmt->getEndLoc());
261 
262  auto* forStmt = new(ctx) ForStmt(ctx,
263  declStmt,
264  rwStmt->getCond(),
265  rwStmt->getLoopVariable(),
266  rwStmt->getInc(),
267  innerScope,
268  rangeForStmt->getBeginLoc(),
269  rangeForStmt->getEndLoc(),
270  rangeForStmt->getEndLoc());
271 
272  AddStmt(outerScopeStmts, forStmt);
273 
274  ArrayRef<Stmt*> outerScopeStmtsRef{outerScopeStmts};
275  auto* outerScope =
276  CompoundStmt::Create(ctx, outerScopeStmtsRef, rangeForStmt->getBeginLoc(), rangeForStmt->getEndLoc());
277 
278  InsertArg(outerScope);
279 
280  mOutputFormatHelper.AppendNewLine();
281 }

424 void CodeGenerator::InsertArg(const MemberExpr* stmt)
425 {
426  const auto* base = stmt->getBase();
427  bool skipBase{false};
428  if(const auto* implicitCast = dyn_cast_or_null<ImplicitCastExpr>(base)) {
429  if(CastKind::CK_UncheckedDerivedToBase == implicitCast->getCastKind()) {
430  const auto* castDest = implicitCast->IgnoreImpCasts();
431 
432  // if this calls a protected function we cannot cast it to the base, this would not compile
433  if(isa<CXXThisExpr>(castDest)) {
434  skipBase = true;
435  }
436  }
437  }
438 
439  if(skipBase) {
440  mOutputFormatHelper.Append("/* ");
441  }
442 
443  InsertArg(base);
444 
445  const std::string op{stmt->isArrow() ? "->" : "."};
446  const auto* meDecl = stmt->getMemberDecl();
447  bool skipTemplateArgs{false};
448  const auto name = [&]() -> std::string {
449  // Handle a special case where we have a lambda static invoke operator. In that case use the appropriate
450  // using retType as return type
451  if(const auto* m = dyn_cast_or_null<CXXMethodDecl>(meDecl)) {
452  if(const auto* rd = m->getParent(); rd && rd->isLambda()) {
453  skipTemplateArgs = true;
454 
455  return StrCat("operator ", GetLambdaName(*rd), "::", BuildRetTypeName(*rd));
456  }
457  }
458 
459  return stmt->getMemberNameInfo().getName().getAsString();
460  }();
461 
462  mOutputFormatHelper.Append(op);
463 
464  if(skipBase) {
465  mOutputFormatHelper.Append(" */ ");
466  }
467 
468  mOutputFormatHelper.Append(name);
469 
470  if(!skipTemplateArgs) {
471  if(const auto cxxMethod = dyn_cast_or_null<CXXMethodDecl>(meDecl)) {
472  if(const auto* tmplArgs = cxxMethod->getAsFunction()->getTemplateSpecializationArgs()) {
473  OutputFormatHelper ofm{};
474 
475  ofm.Append('<');
476 
477  bool haveArg{false};
478  OnceFalse needsComma{};
479  for(const auto& arg : tmplArgs->asArray()) {
480 
481  if(arg.getKind() == TemplateArgument::Integral) {
482  ofm.AppendComma(needsComma);
483 
484  ofm.Append(arg.getAsIntegral());
485  haveArg = true;
486  } else {
487 
488  break;
489  }
490  }
491 
492  if(haveArg) {
493  mOutputFormatHelper.Append(ofm.GetString(), ">");
494  }
495  }
496  }
497  }
498 }

604 void CodeGenerator::InsertArg(const VarDecl* stmt)
605 {
606  LAMBDA_SCOPE_HELPER(VarDecl);
607 
608  if(InsertComma()) {
609  mOutputFormatHelper.Append(',');
610  }
611 
612  if(isa<VarTemplateSpecializationDecl>(stmt)) {
613  mOutputFormatHelper.AppendNewLine("template<>");
614  }
615 
616  if(IsTrivialStaticClassVarDecl(*stmt)) {
617  HandleLocalStaticNonTrivialClass(stmt);
618 
619  } else {
620  if(InsertVarDecl()) {
621  mOutputFormatHelper.Append(GetQualifiers(*stmt));
622 
623  if(const auto type = stmt->getType(); type->isFunctionPointerType()) {
624  const auto lineNo = GetSM(*stmt).getSpellingLineNumber(stmt->getSourceRange().getBegin());
625  const std::string funcPtrName{StrCat("FuncPtr_", lineNo, " ")};
626 
627  mOutputFormatHelper.AppendNewLine("using ", funcPtrName, "= ", GetName(type), ";");
628  mOutputFormatHelper.Append(funcPtrName, GetName(*stmt));
629  } else {
630  const auto varName = [&]() {
631  std::string name{GetName(*stmt)};
632 
633  if(const auto* tvd = dyn_cast_or_null<VarTemplateSpecializationDecl>(stmt)) {
634  OutputFormatHelper outputFormatHelper{};
635  CodeGenerator codeGenerator{outputFormatHelper};
636 
637  codeGenerator.InsertTemplateArgs(tvd->getTemplateArgs().asArray());
638 
639  name += outputFormatHelper.GetString();
640  }
641 
642  return name;
643  }();
644 
645  // TODO: to keep the special handling for lambdas, do this only for template specializations
646  if(stmt->getType()->getAs<TemplateSpecializationType>()) {
647  mOutputFormatHelper.Append(GetNameAsWritten(stmt->getType()), " ", varName);
648  } else {
649  mOutputFormatHelper.Append(GetTypeNameAsParameter(stmt->getType(), varName));
650  }
651  }
652  } else {
653  const std::string pointer = [&]() {
654  if(stmt->getType()->isAnyPointerType()) {
655  return " *";
656  }
657  return " ";
658  }();
659 
660  mOutputFormatHelper.Append(pointer, GetName(*stmt));
661  }
662 
663  if(stmt->hasInit()) {
664  mOutputFormatHelper.Append(" = ");
665 
666  InsertArg(stmt->getInit());
667  };
668 
669  if(stmt->isNRVOVariable()) {
670  mOutputFormatHelper.Append(" /* NRVO variable */");
671  }
672 
673  if(InsertSemi()) {
674  mOutputFormatHelper.AppendSemiNewLine();
675  }
676 
677  // Insert the bindings of a DecompositionDecl if this VarDecl is a DecompositionDecl.
678  if(const auto* decompDecl = dyn_cast_or_null<DecompositionDecl>(stmt)) {
679  StructuredBindingsCodeGenerator codeGenerator{mOutputFormatHelper, GetName(*stmt)};
680 
681  codeGenerator.InsertDecompositionBindings(*decompDecl);
682  }
683  }
684 }

688 {
689  if(cxxMethodDecl && cxxMethodDecl->isLambdaStaticInvoker()) {
690  mOutputFormatHelper.AppendNewLine();

693  const auto* callOp = lambda->getLambdaCallOperator();

695  const TemplateArgumentList* tal = cxxMethodDecl->getTemplateSpecializationArgs();
696  FunctionTemplateDecl* callOpTemplate = callOp->getDescribedFunctionTemplate();
697  void* insertPos = nullptr;
698  FunctionDecl* correspondingCallOpSpecialization =
699  callOpTemplate->findSpecialization(tal->asArray(), insertPos);
700  callOp = cast<CXXMethodDecl>(correspondingCallOpSpecialization);
701  }
702 
703  InsertArg(callOp->getBody());
704  mOutputFormatHelper.AppendNewLine();
705 
706  return true;
707  }
708 
709  return false;
710 }
