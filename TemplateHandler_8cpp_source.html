28 /// \brief Inserts the instantiation point of a template.
29 //
30 // This reveals at which place the template is first used.
31 static void
32 InsertInstantiationPoint(OutputFormatHelper& outputFormatHelper, const SourceManager& sm, const SourceLocation& instLoc)
33 {
34  const auto lineNo = sm.getSpellingLineNumber(instLoc);
35  const auto& fileId = sm.getFileID(instLoc);
36  const auto* file = sm.getFileEntryForID(fileId);
37  if(file) {
38  const auto fileWithDirName = file->getName();
39  const auto fileName = llvm::sys::path::filename(fileWithDirName);
40 
41  outputFormatHelper.AppendNewLine("/* First instantiated from: ", fileName, ":", lineNo, " */");
42  }
43 }

51 /// \brief Insert the instantiated template with the resulting code.
52 template<typename T>
53 static OutputFormatHelper InsertInstantiatedTemplate(const T& decl, const MatchFinder::MatchResult& result)
54 {
55  OutputFormatHelper outputFormatHelper{};
56  outputFormatHelper.AppendNewLine();
57  outputFormatHelper.AppendNewLine();
58 
59  const auto& sm = GetSM(result);
60 
61  if constexpr(not is_same_v<VarTemplateDecl, T>) { // NOLINT
62  InsertInstantiationPoint(outputFormatHelper, sm, decl.getPointOfInstantiation());
63  }
64 
65  outputFormatHelper.AppendNewLine("#ifdef INSIGHTS_USE_TEMPLATE");
66  CodeGenerator codeGenerator{outputFormatHelper};
67 
68  if constexpr(is_same_v<VarTemplateDecl, T>) {
69  for(const auto& spec : decl.specializations()) {
70  InsertInstantiationPoint(outputFormatHelper, sm, spec->getPointOfInstantiation());
71  codeGenerator.InsertArg(spec);
72  }
73  } else {
74  codeGenerator.InsertArg(&decl);
75  }
76 
77  outputFormatHelper.AppendNewLine("#endif");
78 
79  return outputFormatHelper;
80 }

83 TemplateHandler::TemplateHandler(Rewriter& rewrite, MatchFinder& matcher)
84 : InsightsBase(rewrite)
85 {
86  matcher.addMatcher(
87  functionDecl(allOf(unless(isExpansionInSystemHeader()),
88  unless(isMacroOrInvalidLocation()),
89  hasParent(functionTemplateDecl(unless(hasParent(classTemplateSpecializationDecl())),
90  unless(hasParent(cxxRecordDecl(isLambda()))))),
91  isTemplateInstantiationPlain()))
92  .bind("func"),
93  this);
94 
95  // match typical use where a class template is defined and it is used later.
96  matcher.addMatcher(classTemplateSpecializationDecl(unless(isExpansionInSystemHeader()),
97  hasParent(classTemplateDecl().bind("decl")))
98  .bind("class"),
99  this);
100 
101  // special case, where a class template is defined and somewhere else we request an explicit instantiation
102  matcher.addMatcher(classTemplateSpecializationDecl(unless(anyOf(isExpansionInSystemHeader(),
103  hasParent(classTemplateDecl()),
104  isExplicitTemplateSpecialization())))
105  .bind("class"),
106  this);
107 
108  matcher.addMatcher(
109  varTemplateDecl(unless(isExpansionInSystemHeader()), unless(hasParent(classTemplateDecl()))).bind("vd"), this);
110 }

113 void TemplateHandler::run(const MatchFinder::MatchResult& result)
114 {
115  if(const auto* functionDecl = result.Nodes.getNodeAs<FunctionDecl>("func")) {
116  if(not functionDecl->getBody()) {
117  return;
118  }
119 
120  OutputFormatHelper outputFormatHelper = InsertInstantiatedTemplate(*functionDecl, result);
121  const auto endOfCond = FindLocationAfterSemi(GetEndLoc(*functionDecl), result);
122 
123  InsertIndentedText(endOfCond.getLocWithOffset(1), outputFormatHelper);
124 
125  } else if(const auto* clsTmplSpecDecl = result.Nodes.getNodeAs<ClassTemplateSpecializationDecl>("class")) {
126  // skip classes/struct's without a definition
127  if(not clsTmplSpecDecl->hasDefinition()) {
128  return;
129  }
130 
131  OutputFormatHelper outputFormatHelper = InsertInstantiatedTemplate(*clsTmplSpecDecl, result);
132 
133  if(const auto* clsTmplDecl = result.Nodes.getNodeAs<ClassTemplateDecl>("decl")) {
134  const auto endOfCond = FindLocationAfterSemi(GetEndLoc(clsTmplDecl), result);
135  InsertIndentedText(endOfCond, outputFormatHelper);
136 
137  } else { // explicit specialization, we have to remove the specialization
138  mRewrite.ReplaceText(clsTmplSpecDecl->getSourceRange(), outputFormatHelper.GetString());
139  }
140 
141  } else if(const auto* vd = result.Nodes.getNodeAs<VarTemplateDecl>("vd")) {
142  OutputFormatHelper outputFormatHelper = InsertInstantiatedTemplate(*vd, result);
143 
144  const auto endOfCond = FindLocationAfterSemi(GetEndLoc(vd), result);
145  InsertIndentedText(endOfCond, outputFormatHelper);
146  }
147 }
