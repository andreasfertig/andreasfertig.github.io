37 {

56 TemplateHandler::TemplateHandler(Rewriter& rewrite, MatchFinder& matcher)
57 : InsightsBase(rewrite)
58 {
59  matcher.addMatcher(
60  functionDecl(allOf(unless(isExpansionInSystemHeader()),
61  unless(isMacroOrInvalidLocation()),
62  hasParent(functionTemplateDecl(unless(hasParent(classTemplateSpecializationDecl())),
63  unless(hasAncestor(cxxRecordDecl())))),
64  isTemplateInstantiationPlain()))
65  .bind("func"),
66  this);
67 
68  // match typical use where a class template is defined and it is used later.
69  matcher.addMatcher(
70  classTemplateSpecializationDecl(unless(anyOf(isExpansionInSystemHeader(), hasAncestor(cxxRecordDecl()))),
71  hasParent(classTemplateDecl().bind("decl")))
72  .bind("class"),
73  this);
74 
75  // special case, where a class template is defined and somewhere else we request an explicit instantiation
76  matcher.addMatcher(classTemplateSpecializationDecl(unless(anyOf(isExpansionInSystemHeader(),
77  hasParent(classTemplateDecl()),
78  isExplicitTemplateSpecialization())))
79  .bind("class"),
80  this);
81 
82  matcher.addMatcher(
83  varTemplateDecl(unless(isExpansionInSystemHeader()), unless(hasParent(classTemplateDecl()))).bind("vd"), this);
84 }

87 void TemplateHandler::run(const MatchFinder::MatchResult& result)
88 {
89  if(const auto* functionDecl = result.Nodes.getNodeAs<FunctionDecl>("func")) {
90  if(not functionDecl->getBody() && not isa<CXXDeductionGuideDecl>(functionDecl)) {
91  return;
92  }
93 
94  OutputFormatHelper outputFormatHelper = InsertInstantiatedTemplate(*functionDecl);
95  const auto endOfCond = FindLocationAfterSemi(GetEndLoc(*functionDecl), result);
96 
97  InsertIndentedText(endOfCond.getLocWithOffset(1), outputFormatHelper);
98 
99  } else if(const auto* clsTmplSpecDecl = result.Nodes.getNodeAs<ClassTemplateSpecializationDecl>("class")) {
100  // skip classes/struct's without a definition
101  if(not clsTmplSpecDecl->hasDefinition()) {
102  return;
103  }
104 
105  OutputFormatHelper outputFormatHelper = InsertInstantiatedTemplate(*clsTmplSpecDecl);
106 
107  if(const auto* clsTmplDecl = result.Nodes.getNodeAs<ClassTemplateDecl>("decl")) {
108  const auto endOfCond = FindLocationAfterSemi(GetEndLoc(clsTmplDecl), result);
109  InsertIndentedText(endOfCond, outputFormatHelper);
110 
111  } else { // explicit specialization, we have to remove the specialization
112  mRewrite.ReplaceText(clsTmplSpecDecl->getSourceRange(), outputFormatHelper.GetString());
113  }
114 
115  } else if(const auto* vd = result.Nodes.getNodeAs<VarTemplateDecl>("vd")) {
116  OutputFormatHelper outputFormatHelper = InsertInstantiatedTemplate(*vd);
117 
118  const auto endOfCond = FindLocationAfterSemi(GetEndLoc(vd), result);
119  InsertIndentedText(endOfCond, outputFormatHelper);
120  }
121 }
