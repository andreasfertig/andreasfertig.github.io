206 OutputFormatHelper& CodeGenerator::LambdaScopeHandler::GetBuffer(OutputFormatHelper& outputFormatHelper) const
207 {
208  // Find the most outer element to place the lambda class definition. For example, if we have this:
209  // Test( [&]() {} );
210  // The lambda's class definition needs to be placed _before_ the CallExpr to Test.
211  for(auto& l : mStack) {
212  switch(l.callerType()) {
213  case LambdaCallerType::CallExpr: [[fallthrough]];
214  case LambdaCallerType::VarDecl: [[fallthrough]];
215  case LambdaCallerType::ReturnStmt: [[fallthrough]];
216  case LambdaCallerType::OperatorCallExpr: [[fallthrough]];
217  case LambdaCallerType::MemberCallExpr: [[fallthrough]];
218  case LambdaCallerType::BinaryOperator: [[fallthrough]];
219  case LambdaCallerType::CXXMethodDecl: return l.buffer();
220  default: break;
221  }
222  }
223 
224  return outputFormatHelper;
225 }

280 void CodeGenerator::InsertArg(const CXXForRangeStmt* rangeForStmt)
281 {
282  auto& langOpts{GetLangOpts(*rangeForStmt->getLoopVariable())};
283  const bool onlyCpp11{not langOpts.CPlusPlus17};
284 
285  auto* rwStmt = const_cast<CXXForRangeStmt*>(rangeForStmt);
286 
287  std::vector<Stmt*> outerScopeStmts{};
288  std::vector<Stmt*> bodyStmts{};
289 
290  // C++20 init-statement
291  AddStmt(outerScopeStmts, rangeForStmt->getInit());
292 
293  // range statement
294  AddStmt(outerScopeStmts, rangeForStmt->getRangeStmt());
295 
296  if(not onlyCpp11) {
297  AddStmt(outerScopeStmts, rangeForStmt->getBeginStmt());
298  AddStmt(outerScopeStmts, rangeForStmt->getEndStmt());
299  }
300 
301  // add the loop variable to the body
302  AddStmt(bodyStmts, rangeForStmt->getLoopVarStmt());
303 
304  // add the body itself, without the CompoundStmt
305  AddBodyStmts(bodyStmts, rwStmt->getBody());
306 
307  const auto& ctx = rangeForStmt->getLoopVariable()->getASTContext();
308 
309  // In case of a range-based for-loop inside an unevaluated template the begin and end statements are not present. In
310  // this case just add a nullptr.
311  Decl* decls[2]{rwStmt->getBeginStmt() ? rwStmt->getBeginStmt()->getSingleDecl() : nullptr,
312  rwStmt->getEndStmt() ? rwStmt->getEndStmt()->getSingleDecl() : nullptr};
313 
314  auto dgRef = DeclGroupRef::Create(const_cast<ASTContext&>(ctx), decls, 2);
315 
316  auto* declStmt = [&]() -> DeclStmt* {
317  if(onlyCpp11) {
318  return new(ctx) DeclStmt(dgRef, rangeForStmt->getBeginLoc(), rangeForStmt->getEndLoc());
319  }
320 
321  return nullptr;
322  }();
323 
324  ArrayRef<Stmt*> innerScopeStmtsRef{bodyStmts};
325  auto* innerScope =
326  asthelpers::mkCompoundStmt(innerScopeStmtsRef, rangeForStmt->getBeginLoc(), rangeForStmt->getEndLoc());
327 
328  auto* forStmt = new(ctx) ForStmt(ctx,
329  declStmt,
330  rwStmt->getCond(),
331  rwStmt->getLoopVariable(),
332  rwStmt->getInc(),
333  innerScope,
334  rangeForStmt->getBeginLoc(),
335  rangeForStmt->getEndLoc(),
336  rangeForStmt->getEndLoc());
337 
338  AddStmt(outerScopeStmts, forStmt);
339 
340  ArrayRef<Stmt*> outerScopeStmtsRef{outerScopeStmts};
341  auto* outerScope =
342  asthelpers::mkCompoundStmt(outerScopeStmtsRef, rangeForStmt->getBeginLoc(), rangeForStmt->getEndLoc());
343 
344  InsertArg(outerScope);
345 
346  mOutputFormatHelper.AppendNewLine();
347 }

389 void CodeGenerator::InsertArg(const VarTemplateDecl* stmt)
390 {
391  const auto* templatedDecl = stmt->getTemplatedDecl();
392 
393  // Insert only the primary template here. The specializations are inserted via their instantiated
394  // VarTemplateSpecializationDecl which resolved to a VarDecl. It looks like whether the variable has an initializer
395  // or not can be used to distinguish between the primary template and one appearing in a templated class.
396  RETURN_IF(not templatedDecl->hasInit());
397 
398  // VarTemplatedDecl's can have lambdas as initializers. Push a VarDecl on the stack, otherwise the lambda would
399  // appear in the middle of template<....> and the variable itself.
400  {
401  LAMBDA_SCOPE_HELPER(Decltype); // Needed for P0315Checker
402  mLambdaStack.back().setInsertName(true);
403 
404  InsertTemplateParameters(*stmt->getTemplateParameters());
405  }
406 
407  LAMBDA_SCOPE_HELPER(VarDecl);
408 
409  InsertArg(templatedDecl);
410 
411  OnceTrue first{};
412  for(const auto* spec : stmt->specializations()) {
413  if(TSK_ExplicitSpecialization != spec->getSpecializationKind()) {
414  if(first) {
415  mOutputFormatHelper.AppendNewLine();
416  }
417 
418  InsertArg(spec);
419  }
420  }

547 /// Get the name of a \c FieldDecl in case this \c FieldDecl is part of a lambda. The name has to be retrieved from the
548 /// capture fields or can be \c __this.
549 static Optional<std::string> GetFieldDeclNameForLambda(const FieldDecl& fieldDecl, const CXXRecordDecl& cxxRecordDecl)
550 {
551  if(cxxRecordDecl.isLambda()) {
552  llvm::DenseMap<const VarDecl*, FieldDecl*> captures{};
553  FieldDecl* thisCapture{};
554 
555  cxxRecordDecl.getCaptureFields(captures, thisCapture);
556 
557  if(&fieldDecl == thisCapture) {
558  return std::string{kwInternalThis};
559  } else {
560  for(const auto& [key, value] : captures) {
561  if(&fieldDecl == value) {
562  return GetName(*key);
563  }
564  }
565  }
566  }
567 
568  return {};
569 }

572 void CodeGenerator::InsertArg(const MemberExpr* stmt)
573 {
574  const auto* base = stmt->getBase();
575  const bool skipBase{[&] {
576  if(const auto* implicitCast = dyn_cast_or_null<ImplicitCastExpr>(base)) {
577  if(CastKind::CK_UncheckedDerivedToBase == implicitCast->getCastKind()) {
578  // if this calls a protected function we cannot cast it to the base, this would not compile
579  if(isa<CXXThisExpr>(implicitCast->IgnoreImpCasts())) {
580  return true;
581  }
582  }
583  }
584 
585  return false;
586  }()};
587 
588  if(skipBase) {
589  mOutputFormatHelper.Append(kwCCommentStartSpace);
590  }
591 
592  InsertArg(base);
593 
594  const auto* meDecl = stmt->getMemberDecl();
595  bool skipTemplateArgs{false};
596  const auto name = [&]() -> std::string {
597  // Handle a special case where we have a lambda static invoke operator. In that case use the appropriate
598  // using retType as return type
599  if(const auto* m = dyn_cast_or_null<CXXMethodDecl>(meDecl)) {
600  if(const auto* rd = m->getParent(); rd and rd->isLambda() and isa<CXXConversionDecl>(m)) {
601  skipTemplateArgs = true;
602 
603  return StrCat(kwOperatorSpace, GetLambdaName(*rd), "::"sv, BuildRetTypeName(*rd));
604  }
605  }
606 
607  // This is at least the case for lambdas, where members are created by capturing a structured binding. See #181.
608  else if(const auto* fd = dyn_cast_or_null<FieldDecl>(meDecl)) {
609  if(const auto* cxxRecordDecl = dyn_cast_or_null<CXXRecordDecl>(fd->getParent())) {
610  if(const auto& fieldName = GetFieldDeclNameForLambda(*fd, *cxxRecordDecl)) {
611  return fieldName.getValue();
612  }
613  }
614  }
615 
616  // Special case. If this is a CXXConversionDecl it might be:
617  // a) a template so we need the template arguments from this type
618  // b) in a namespace and need want to preserve that one.
619  if(const auto* convDecl = dyn_cast_or_null<CXXConversionDecl>(meDecl)) {
620  return StrCat(kwOperatorSpace, GetName(convDecl->getConversionType()));
621  }
622 
623  return stmt->getMemberNameInfo().getName().getAsString();
624  }();
625 
626  mOutputFormatHelper.Append(ArrowOrDot(stmt->isArrow()));
627 
628  if(skipBase) {
629  mOutputFormatHelper.Append(kwSpaceCCommentEndSpace);
630  }
631 
632  mOutputFormatHelper.Append(name);
633 
634  RETURN_IF(skipTemplateArgs);
635 
636  if(const auto cxxMethod = dyn_cast_or_null<CXXMethodDecl>(meDecl)) {
637  if(const auto* tmplArgs = cxxMethod->getTemplateSpecializationArgs()) {
638  OutputFormatHelper ofm{};
639 
640  ofm.Append('<');
641 
642  bool haveArg{false};
643  OnceFalse needsComma{};
644  for(const auto& arg : tmplArgs->asArray()) {
645 
646  if(arg.getKind() == TemplateArgument::Integral) {
647  ofm.AppendComma(needsComma);
648 
649  ofm.Append(arg.getAsIntegral());
650  haveArg = true;
651  } else {
652 
653  break;
654  }
655  }
656 
657  if(haveArg) {
658  mOutputFormatHelper.Append(ofm, ">"sv);
659 
660  } else {
661  InsertTemplateArgs(*tmplArgs);
662  }
663  }
664  }
665 }

979  }

995 bool CodeGenerator::InsertLambdaStaticInvoker(const CXXMethodDecl* cxxMethodDecl)
996 {
997  if(not(cxxMethodDecl and cxxMethodDecl->isLambdaStaticInvoker())) {
998  return false;
999  }

1002  // operator gives the same result as invoking the function pointer (see [expr.prim.lambda.closure] p9). When it
1003  // comes to block local statics having a body for both functions reveals a difference. This special code
1004  // generates a forwarding call from the call operator to the static invoker. However, the compiler does better
1005  // here. As this way we end up with copies of the parameters which is hard to avoid.
1006

1012  }

1021  if(cxxMethodDecl->getDescribedTemplate()) {
1022  InsertTemplateParameters(*cxxMethodDecl->getDescribedTemplate()->getTemplateParameters(),
1023  TemplateParamsOnly::Yes);
1024  }
1025  /*else if(decl.isFunctionTemplateSpecialization()) {
1026  InsertTemplateSpecializationHeader();
1027  }*/
1028  }
1029

1031  mOutputFormatHelper.AppendParameterList(cxxMethodDecl->parameters(),
1032  OutputFormatHelper::NameOnly::Yes,
1033  OutputFormatHelper::GenMissingParamName::Yes);
1034  });
1035 
1036  mOutputFormatHelper.AppendSemiNewLine();
1037  mOutputFormatHelper.CloseScope(OutputFormatHelper::NoNewLineBefore::Yes);
1038  mOutputFormatHelper.AppendNewLine();
1039 
1040  return true;
1041 }
