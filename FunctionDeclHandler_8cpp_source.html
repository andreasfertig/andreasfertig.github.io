51 void FunctionDeclHandler::run(const MatchFinder::MatchResult& result)
52 {
53  if(const auto* funcDecl = result.Nodes.getNodeAs<FunctionDecl>("funcDecl")) {
54  const auto columnNr = GetSM(result).getSpellingColumnNumber(GetBeginLoc(funcDecl)) - 1;
55  OutputFormatHelper outputFormatHelper{columnNr};
56  CodeGenerator codeGenerator{outputFormatHelper};
57 
58  codeGenerator.InsertArg(funcDecl);
59 
60  // Find the correct ending of the source range. In case of a declaration we need to find the ending semi,
61  // otherwise the provided source range is correct.
62  const auto sr =
63  GetSourceRangeAfterSemi(funcDecl->getSourceRange(),
64  result,
65  funcDecl->doesThisDeclarationHaveABody() ? RequireSemi::No : RequireSemi::Yes);
66 
67  // DPrint("fd rw: %d %s\n", (sr.getBegin() == sr.getEnd()), outputFormatHelper.GetString());
68 
69  mRewrite.ReplaceText(sr, outputFormatHelper.GetString());
