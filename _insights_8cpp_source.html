112 class FindIncludes : public PPCallbacks
113 {
114  SourceManager& mSm;
115  Preprocessor& mPP;
116  std::vector<IncludeData>& mIncludes;
117 
118 public:
119  FindIncludes(SourceManager& sm, Preprocessor& pp, std::vector<IncludeData>& incData)
120  : PPCallbacks{}
121  , mSm{sm}
122  , mPP{pp}
123  , mIncludes{incData}
124  {
125  }
126 
127  void InclusionDirective(SourceLocation hashLoc,
128  const Token& /*IncludeTok*/,
129  StringRef fileName,
130  bool isAngled,
131  CharSourceRange /*FilenameRange*/,
132  OptionalFileEntryRef /*file*/,
133  StringRef /*SearchPath*/,
134  StringRef /*RelativePath*/,
135  const Module* /*Imported*/,
136  SrcMgr::CharacteristicKind /*FileType*/) override
137  {
138  auto expansionLoc = mSm.getExpansionLoc(hashLoc);
139 
140  if(expansionLoc.isInvalid() or mSm.isInSystemHeader(expansionLoc)) {
141  return;
142  }
143 
144  // XXX: distinguish between include and import via the IncludeTok
145  if(isAngled) {
146  mIncludes.emplace_back(expansionLoc, StrCat("#include <"sv, fileName, ">\n"sv));
147 
148  } else {
149  mIncludes.emplace_back(expansionLoc, StrCat("#include \""sv, fileName, "\"\n"sv));
150  }
151  }
152 
153  void MacroDefined(const Token& macroNameTok, const MacroDirective* md) override
154  {
155  const auto loc = md->getLocation();
156  if(not mSm.isWrittenInMainFile(loc)) {
157  return;
158  }
159 
160  auto name = mPP.getSpelling(macroNameTok);
161 
162  if(not name.starts_with("INSIGHTS_"sv)) {
163  return;
164  }
165 
166  mIncludes.emplace_back(loc, StrCat("#define "sv, name, "\n"sv));
167  }
168 };

171 {
172  Rewriter& mRewriter;
173  std::vector<IncludeData>& mIncludes;
174 
175 public:
176  explicit CppInsightASTConsumer(Rewriter& rewriter, std::vector<IncludeData>& includes)
177  : ASTConsumer{}
178  , mRewriter{rewriter}
179  , mIncludes{includes}
180  {
181  if(GetInsightsOptions().UseShow2C) {
182  gInsightsOptions.ShowLifetime = true;
183  }
184 
185  if(GetInsightsOptions().ShowLifetime) {
186  gInsightsOptions.UseShowInitializerList = true;
187  }
188  }
189 
190  void HandleTranslationUnit(ASTContext& context) override
191  {
192  gAST = &context;
193  auto& sm = context.getSourceManager();
194 
195  auto isExpansionInSystemHeader = [&sm](const Decl* d) {
196  auto expansionLoc = sm.getExpansionLoc(d->getLocation());
197 
198  return expansionLoc.isInvalid() or sm.isInSystemHeader(expansionLoc);
199  };
200 
201  const auto& mainFileId = sm.getMainFileID();
202 
203  mRewriter.ReplaceText({sm.getLocForStartOfFile(mainFileId), sm.getLocForEndOfFile(mainFileId)}, "");
204 
205  OutputFormatHelper outputFormatHelper{};
206  CodeGeneratorVariant codeGenerator{outputFormatHelper};
207 
208  auto include = mIncludes.begin();
209 
210  auto insertBlankLineIfRequired = [&](std::optional<SourceLocation>& lastLoc, SourceLocation nextLoc) {
211  if(lastLoc.has_value() and
212  (2 <= (sm.getSpellingLineNumber(nextLoc) - sm.getSpellingLineNumber(lastLoc.value())))) {
213  outputFormatHelper.AppendNewLine();
214  }
215 
216  lastLoc = nextLoc;
217  };
218 
219  for(std::optional<SourceLocation> lastLoc{}; const auto* d : context.getTranslationUnitDecl()->decls()) {
220  if(isExpansionInSystemHeader(d)) {
221  continue;
222  }
223 
224  // includes before this decl
225  for(; (mIncludes.end() != include) and (include->first < d->getLocation()); include = std::next(include)) {
226  insertBlankLineIfRequired(lastLoc, include->first);
227  outputFormatHelper.Append(include->second);
228  }
229 
230  // ignore includes inside this decl
231  include = std::find_if_not(include, mIncludes.end(), [&](auto& inc) {
232  return ((inc.first >= d->getLocation()) and (inc.first <= d->getEndLoc()));
233  });
234 
235  if(isa<LinkageSpecDecl>(d) and d->isImplicit()) {
236  continue;
237 
238  // Only handle explicit specializations here. Implicit ones are handled by the `VarTemplateDecl`
239  // itself.
240  } else if(const auto* vdspec = dyn_cast_or_null<VarTemplateSpecializationDecl>(d);
241  vdspec and (TSK_ExplicitSpecialization != vdspec->getSpecializationKind())) {
242  continue;
243  }
244 
245  insertBlankLineIfRequired(lastLoc, d->getLocation());
246 
247  codeGenerator->InsertArg(d);
248  }
249 
250  std::string insightsIncludes{};
251 
252  if(GetInsightsOptions().ShowCoroutineTransformation) {
253  insightsIncludes.append(
254  R"(/*************************************************************************************
255  * NOTE: The coroutine transformation you've enabled is a hand coded transformation! *
256  * Most of it is _not_ present in the AST. What you see is an approximation. *
257  *************************************************************************************/
258 )"sv);
259  } else if(GetInsightsOptions().UseShow2C or GetInsightsOptions().ShowLifetime) {
260  insightsIncludes.append(
261  R"(/*************************************************************************************
262  * NOTE: This an educational hand-rolled transformation. Things can be incorrect or *
263  * buggy. *
264  *************************************************************************************/
265 )"sv);
266  }
267 
268  // Check whether we had static local variables which we transformed. Then for the placement-new we need to
269  // include the header <new>.
270  std::string inserts{};
271  for(const auto& [active, value] : gGlobalInserts) {
272  if(not active) {
273  continue;
274  }
275 
276  inserts.append(value);
277  inserts.append("\n"sv);
278  }
279 
280  if(not inserts.empty()) {
281  insightsIncludes.append(inserts);
282  insightsIncludes.append("\n");
283  }
284 
285  outputFormatHelper.InsertAt(0, insightsIncludes);
286 
287  mRewriter.InsertText(sm.getLocForStartOfFile(mainFileId), outputFormatHelper.GetString());
288 
289  if(GetInsightsOptions().UseShow2C) {
290  const auto& fileEntry = sm.getFileEntryForID(mainFileId);
291  auto cxaStart = EmitGlobalVariableCtors();
292  const auto cxaLoc = sm.translateFileLineCol(fileEntry, fileEntry->getSize(), 1);
293 
294  mRewriter.InsertText(cxaLoc, cxaStart);
295  }
296  }
297 };

341 int main(int argc, const char** argv)
342 {
343  // Headers go first
344  using enum GlobalInserts;
345  AddGLobalInsertMapEntry(HeaderNew,
346  "#include <new> // for thread-safe static's placement new\n#include <stdint.h> // for "
347  "uint64_t under Linux/GCC"sv);
348  AddGLobalInsertMapEntry(HeaderException, "#include <exception> // for noexcept transformation"sv);
349  AddGLobalInsertMapEntry(HeaderUtility, "#include <utility> // std::move"sv);
350  AddGLobalInsertMapEntry(HeaderStddef, "#include <stddef.h> // NULL and more"sv);
351  AddGLobalInsertMapEntry(HeaderAssert, "#include <assert.h> // _Static_assert"sv);
352 
353  // Now all the forward declared functions
354  AddGLobalInsertMapEntry(FuncCxaStart, "void __cxa_start(void);"sv);
355  AddGLobalInsertMapEntry(FuncCxaAtExit, "void __cxa_atexit(void);"sv);
356  AddGLobalInsertMapEntry(FuncMalloc, "void* malloc(unsigned int);"sv);
357  AddGLobalInsertMapEntry(FuncFree, R"(extern "C" void free(void*);)"sv);
358  AddGLobalInsertMapEntry(FuncMemset, R"(extern "C" void* memset(void*, int, unsigned int);)"sv);
359  AddGLobalInsertMapEntry(FuncMemcpy, R"(void* memcpy(void*, const void*, unsigned int);)"sv);
360  AddGLobalInsertMapEntry(
361  FuncCxaVecNew,
362  R"(extern "C" void* __cxa_vec_new(void*, unsigned int, unsigned int, unsigned int, void* (*)(void*), void* (*)(void*));)"sv);
363  AddGLobalInsertMapEntry(
364  FuncCxaVecCtor,
365  R"(extern "C" void* __cxa_vec_ctor(void*, unsigned int, unsigned int, unsigned int, void* (*)(void*), void* (*)(void*));)"sv);
366  AddGLobalInsertMapEntry(
367  FuncCxaVecDel,
368  R"(extern "C" void __cxa_vec_delete(void *, unsigned int, unsigned int, void* (*destructor)(void *) );)"sv);
369  AddGLobalInsertMapEntry(
370  FuncCxaVecDtor,
371  R"(extern "C" void __cxa_vec_dtor(void *, unsigned int, unsigned int, void* (*destructor)(void *) );)"sv);
372 
373  llvm::sys::PrintStackTraceOnErrorSignal(argv[0]);
374  llvm::cl::HideUnrelatedOptions(gInsightCategory);
375  llvm::cl::SetVersionPrinter(&PrintVersion);
376 
377  auto opExpected = CommonOptionsParser::create(argc, argv, gInsightCategory);
378 
379  if(auto err = opExpected.takeError()) {
380  llvm::errs() << toString(std::move(err)) << "\n";
381  return 1;
382  }
383 
384  // In STDINMode, we override the file content with the <stdin> input.
385  // Since `tool.mapVirtualFile` takes `StringRef`, we define `Code` outside of
386  // the if-block so that `Code` is not released after the if-block.
387  std::unique_ptr<llvm::MemoryBuffer> inMemoryCode{};
388 
389  CommonOptionsParser& op{opExpected.get()};
390  ClangTool tool(op.getCompilations(), op.getSourcePathList());
391 
392  if(gStdinMode) {
393  if(op.getSourcePathList().size() != 1) {
394  llvm::errs() << "Expect exactly one file path in STDINMode.\n"sv;
395  return 1;
396  }
397 
398  llvm::ErrorOr<std::unique_ptr<llvm::MemoryBuffer>> codeOrErr = llvm::MemoryBuffer::getSTDIN();
399 
400  if(const std::error_code errorCode = codeOrErr.getError()) {
401  llvm::errs() << errorCode.message() << "\n";
402  return 1;
403  }
404 
405  inMemoryCode = std::move(codeOrErr.get());
406 
407  if(inMemoryCode->getBufferSize() == 0) {
408  Error("empty file\n");
409  return 1; // Skip empty files.
410  }
411 
412  llvm::StringRef sourceFilePath = op.getSourcePathList().front();
413  tool.mapVirtualFile(sourceFilePath, inMemoryCode->getBuffer());
414  }
415 
416  auto prependArgument = [&](auto arg) {
417  tool.appendArgumentsAdjuster(getInsertArgumentAdjuster(arg, ArgumentInsertPosition::BEGIN));
418  };
419 
420  // Special handling to spare users to figure out what include paths to add.
421 
422  // For some reason, Clang on Apple seems to require an additional hint for the C++ headers.
423 #ifdef __APPLE__
424  gUseLibCpp = true;
425 #endif /* __APPLE__ */
426 
427  if(gUseLibCpp) {
428  prependArgument(INSIGHTS_LLVM_INCLUDE_DIR);
429  prependArgument("-stdlib=libc++");
430 
431 #if IS_CLANG_NEWER_THAN(15)
432  prependArgument("-fexperimental-library");
433 #endif
434 
435 #ifdef __APPLE__
436  prependArgument("-nostdinc++"); // macos Monterey
437 #endif /* __APPLE__ */
438  }
439 
440  prependArgument(INSIGHTS_CLANG_RESOURCE_INCLUDE_DIR);
441  prependArgument(INSIGHTS_CLANG_RESOURCE_DIR);
442 
443  if(GetInsightsOptions().UseShow2C) {
444  EnableGlobalInsert(FuncCxaStart);
445  EnableGlobalInsert(FuncCxaAtExit);
446  }
447 
448  return tool.run(newFrontendActionFactory<CppInsightFrontendAction>().get());
449 }
