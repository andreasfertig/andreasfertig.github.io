107  void HandleTranslationUnit(ASTContext& context) override
108  {
109  gAST = &context;
110  mMatcher.matchAST(context);
111 
112  // Check whether we had static local variables which we transformed. Then for the placement-new we need to
113  // include the header <new>.
114  if(CodeGenerator::NeedToInsertNewHeader() || CodeGenerator::NeedToInsertExceptionHeader()) {
115  const auto& sm = context.getSourceManager();
116  const auto& mainFileId = sm.getMainFileID();
117  const auto loc = sm.translateFileLineCol(sm.getFileEntryForID(mainFileId), 1, 1);
118 
119  if(CodeGenerator::NeedToInsertNewHeader()) {
120  mRewriter.InsertText(
121  loc,
122  "#include <new> // for thread-safe static's placement new\n#include <stdint.h> // for "
123  "uint64_t under Linux/GCC\n"sv);
124  } else {
125  mRewriter.InsertText(loc, "#include <exception> // for noexcept transformation\n"sv);
126  }
127  }
128  }

185 int main(int argc, const char** argv)
186 {
187  llvm::sys::PrintStackTraceOnErrorSignal(argv[0]);
188  llvm::cl::HideUnrelatedOptions(gInsightCategory);
189  llvm::cl::SetVersionPrinter(&PrintVersion);
190 
191  CommonOptionsParser op(argc, argv, gInsightCategory);
192  ClangTool tool(op.getCompilations(), op.getSourcePathList());
193 
194  llvm::StringRef sourceFilePath = op.getSourcePathList().front();
195  // In STDINMode, we override the file content with the <stdin> input.
196  // Since `tool.mapVirtualFile` takes `StringRef`, we define `Code` outside of
197  // the if-block so that `Code` is not released after the if-block.
198  std::unique_ptr<llvm::MemoryBuffer> inMemoryCode{};
199 
200  if(gStdinMode) {
201  assert(op.getSourcePathList().size() == 1 && "Expect exactly one file path in STDINMode.");
202  llvm::ErrorOr<std::unique_ptr<llvm::MemoryBuffer>> codeOrErr = llvm::MemoryBuffer::getSTDIN();
203  if(const std::error_code errorCode = codeOrErr.getError()) {
204  llvm::errs() << errorCode.message() << "\n";
205  return 1;
206  }
207  inMemoryCode = std::move(codeOrErr.get());
208  if(inMemoryCode->getBufferSize() == 0) {
209  Error("empty file\n");
210  return 1; // Skip empty files.
211  }
212 
213  tool.mapVirtualFile(sourceFilePath, inMemoryCode->getBuffer());
214  }
215 
216  auto prependArgument = [&](auto arg) {
217  tool.appendArgumentsAdjuster(getInsertArgumentAdjuster(arg, ArgumentInsertPosition::BEGIN));
218  };
219 
220  // Special handling to spare users to figure out what include paths to add.
221 
222  // For some reason, Clang on Apple seems to require an additional hint for the C++ headers.
223 #ifdef __APPLE__
224  gUseLibCpp = true;
225 #endif /* __APPLE__ */
226 
227  if(gUseLibCpp) {
228  prependArgument(INSIGHTS_LLVM_INCLUDE_DIR);
229  prependArgument("-stdlib=libc++");
230  }
231 
232  prependArgument(INSIGHTS_CLANG_RESOURCE_INCLUDE_DIR);
233  prependArgument(INSIGHTS_CLANG_RESOURCE_DIR);
234 
235  return tool.run(newFrontendActionFactory<CppInsightFrontendAction>().get());
236 }
