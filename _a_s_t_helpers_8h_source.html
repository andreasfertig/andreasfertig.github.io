61 ///! A helper type to have a container for ArrayRef
62 struct StmtsContainer
63 {
64  SmallVector<Stmt*, 64> mStmts{};
65 
66  StmtsContainer() = default;
67  StmtsContainer(std::initializer_list<const Stmt*> stmts)
68  {
69  for(const auto& stmt : stmts) {
70  Add(stmt);
71  }
72  }
73 
74  void clear() { mStmts.clear(); }
75 
76  void Add(const Stmt* stmt)
77  {
78  if(stmt) {
79  mStmts.push_back(const_cast<Stmt*>(stmt));
80  }
81  }
82 
83  void AddBodyStmts(Stmt* body);
84 
85  operator ArrayRef<Stmt*>() { return mStmts; }
86 };
