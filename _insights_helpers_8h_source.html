262 /// \brief Track the scope we are currently in to build a properly scoped variable.
263 ///
264 /// The AST only knows about absolute scopes (namespace, struct, class), as once a declaration is parsed it is either in
265 /// a scope or not. Each request to give me the namespace automatically leads to the entire scope the item is in. This
266 /// makes it hard to have constructs like this: \code struct One
267 /// {
268 /// static const int o{};
269 ///
270 /// struct Two
271 /// {
272 /// static const int d = o;
273 /// };
274 ///
275 /// static const int a = Two::d;
276 ///};
277 /// \endcode
278 ///
279 /// Here the initializer of \c a is in the scope \c One::Two::d. At this point the qualification \c Two::d is enought.
280 ///
281 /// \c ScopeHelper tracks whether we are currently in a class or namespace and simply remove the path we already in from
282 /// the scope.
283 struct ScopeHelper : public StackListEntry<ScopeHelper>
284 {
285  ScopeHelper(const size_t len)
286  : mLength{len}
287  {
288  }
289 
290  const size_t mLength; //!< Length of the scope as it was _before_ this declaration was appended.
291 };

294 /// \brief The ScopeHandler tracks the current scope.
295 ///
296 /// The \c ScopeHandler tracks the current scope, knows about all the parts and is able to remove the current scope part
297 /// from a name.
298 class ScopeHandler
299 {
300 public:
301  ScopeHandler(const Decl* d);
302 
303  ~ScopeHandler();
304 
305  /// \brief Remove the current scope from a string.
306  ///
307  /// The default is that the entire scope is replaced. Suppose we are
308  /// currently in N::X and having a symbol N::X::y then N::X:: is removed. However, there is a special case, where
309  /// the last item is skipped.
310  static std::string RemoveCurrentScope(std::string name);
311 
312 private:
313  using ScopeStackType = StackList<ScopeHelper>;
314 
315  ScopeStackType& mStack; //!< Access to the global \c ScopeHelper stack.
316  ScopeHelper mHelper; //!< The \c ScopeHelper this item refers to.
317 
318  static ScopeStackType mGlobalStack; //!< Global stack to keep track of the scope elements.
319  static std::string mScope; //!< The entire scope we are already in.
320 };
