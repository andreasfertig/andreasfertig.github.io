224 /// \brief Track the scope we are currently in to build a properly scoped variable.
225 ///
226 /// The AST only knows about absolute scopes (namespace, struct, class), as once a declaration is parsed it is either in
227 /// a scope or not. Each request to give me the namespace automatically leads to the entire scope the item is in. This
228 /// makes it hard to have constructs like this: \code struct One
229 /// {
230 /// static const int o{};
231 ///
232 /// struct Two
233 /// {
234 /// static const int d = o;
235 /// };
236 ///
237 /// static const int a = Two::d;
238 ///};
239 /// \endcode
240 ///
241 /// Here the initializer of \c a is in the scope \c One::Two::d. At this point the qualification \c Two::d is enought.
242 ///
243 /// \c ScopeHelper tracks whether we are currently in a class or namespace and simply remove the path we already in from
244 /// the scope.
245 struct ScopeHelper : public StackListEntry<ScopeHelper>
246 {
247  ScopeHelper(const size_t len)
248  : mLength{len}
249  {
250  }
251 
252  const size_t mLength; //!< Length of the scope as it was _before_ this declaration was appended.
253 };
254 //-----------------------------------------------------------------------------
255 
256 /// \brief The ScopeHandler tracks the current scope.
257 ///
258 /// The \c ScopeHandler tracks the current scope, knows about all the parts and is able to remove the current scope part
259 /// from a name.
260 class ScopeHandler
261 {
262 public:
263  ScopeHandler(const Decl* d);
264 
265  ~ScopeHandler();
266 
267  /// \brief Remove the current scope from a string.
268  ///
269  /// The default is that the entire scope is replaced. Suppose we are
270  /// currently in N::X and having a symbol N::X::y then N::X:: is removed. However, there is a special case, where
271  /// the last item is skipped.
272  static std::string RemoveCurrentScope(std::string name);
273 
274 private:
275  using ScopeStackType = StackList<ScopeHelper>;
276 
277  ScopeStackType& mStack; //!< Access to the global \c ScopeHelper stack.
278  ScopeHelper mHelper; //!< The \c ScopeHelper this item refers to.
279 
280  static ScopeStackType mGlobalStack; //!< Global stack to keep track of the scope elements.
281  static std::string mScope; //!< The entire scope we are already in.
282 };
