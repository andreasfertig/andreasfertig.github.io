223 /// \brief Track the scope we are currently in to build a properly scoped variable.
224 ///
225 /// The AST only knows about absolute scopes (namespace, struct, class), as once a declaration is parsed it is either in
226 /// a scope or not. Each request to give me the namespace automatically leads to the entire scope the item is in. This
227 /// makes it hard to have constructs like this: \code struct One
228 /// {
229 /// static const int o{};
230 ///
231 /// struct Two
232 /// {
233 /// static const int d = o;
234 /// };
235 ///
236 /// static const int a = Two::d;
237 ///};
238 /// \endcode
239 ///
240 /// Here the initializer of \c a is in the scope \c One::Two::d. At this point the qualification \c Two::d is enought.
241 ///
242 /// \c ScopeHelper tracks whether we are currently in a class or namespace and simply remove the path we already in from
243 /// the scope.
244 struct ScopeHelper : public StackListEntry<ScopeHelper>
245 {
246  ScopeHelper(const size_t len)
247  : mLength{len}
248  {
249  }
250 
251  const size_t mLength; //!< Length of the scope as it was _before_ this declaration was appended.
252 };
253 //-----------------------------------------------------------------------------
254 
255 /// \brief The ScopeHandler tracks the current scope.
256 ///
257 /// The \c ScopeHandler tracks the current scope, knows about all the parts and is able to remove the current scope part
258 /// from a name.
259 class ScopeHandler
260 {
261 public:
262  ScopeHandler(const Decl* d);
263 
264  ~ScopeHandler();
265 
266  /// \brief Remove the current scope from a string.
267  ///
268  /// The default is that the entire scope is replaced. Suppose we are
269  /// currently in N::X and having a symbol N::X::y then N::X:: is removed. However, there is a special case, where
270  /// the last item is skipped.
271  static std::string RemoveCurrentScope(std::string name);
272 
273 private:
274  using ScopeStackType = StackList<ScopeHelper>;
275 
276  ScopeStackType& mStack; //!< Access to the global \c ScopeHelper stack.
277  ScopeHelper mHelper; //!< The \c ScopeHelper this item refers to.
278 
279  static inline ScopeStackType mGlobalStack; //!< Global stack to keep track of the scope elements.
280  static inline std::string mScope; //!< The entire scope we are already in.
281 };

395 template<class T>
396 class MyOptional : public std::optional<T>
397 {
398 
399 public:
400  using std::optional<T>::optional;
401 
402  template<class Return>
403  MyOptional<Return> and_then(std::function<MyOptional<Return>(T)> func)
404  requires(not std::is_pointer_v<T>)
405  {
406  if(not this->has_value()) {
407  return {};
408  }
409 
410  return func(this->value());
411  }
412 
413  template<class Return>
414  MyOptional<Return> and_then(std::function<MyOptional<Return>(std::remove_pointer_t<T>&)> func)
415  requires(std::is_pointer_v<T>)
416  {
417  if(not this->has_value()) {
418  return {};
419  }
420 
421  if(this->value() == nullptr) {
422  return {};
423  } else {
424  return func(*this->value());
425  }
426  }
427 
428  template<class Return>
429  MyOptional<Return> and_not(std::function<MyOptional<Return>(T)> func)
430  {
431  if(not this->has_value()) {
432  return {};
433  }
434 
435  if(not func(this->value()).has_value()) {
436  return *this;
437  }
438 
439  return {};
440  }
441 };
