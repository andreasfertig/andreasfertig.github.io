50  OutputFormatHelper outputFormatHelper{};
51 
52  CodeGenerator codeGenerator{outputFormatHelper};
53  codeGenerator.InsertArg(cxxRecordDecl);
54 
55  if(auto sourceRange = cxxRecordDecl->getSourceRange(); not IsMacroLocation(sourceRange)) {
56  if(IsAnonymousStructOrUnion(cxxRecordDecl)) {
57  sourceRange.setEnd(sourceRange.getEnd().getLocWithOffset(2)); // 2 is just what worked
58  }
59 
60  mRewrite.ReplaceText(GetSourceRangeAfterSemi(sourceRange, result, RequireSemi::Yes),
61  outputFormatHelper.GetString());
62 
63  } else {
64  // We're just interested in the start location, -1 work(s|ed)
65  const auto startLoc =
66  GetSourceRangeAfterSemi(sourceRange, result, RequireSemi::No).getBegin().getLocWithOffset(-1);
67 
68  mRewrite.InsertText(startLoc, outputFormatHelper.GetString());
69  }

71  } else if(const auto* namespaceDecl = result.Nodes.getNodeAs<NamespaceDecl>("namespaceDecl")) {
72  OutputFormatHelper outputFormatHelper{};

74  CodeGenerator codeGenerator{outputFormatHelper};
