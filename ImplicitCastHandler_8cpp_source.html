21 ImplicitCastHandler::ImplicitCastHandler(Rewriter& rewrite, MatchFinder& matcher)
22 : InsightsBase(rewrite)
23 {
24  static const auto implicitCastMatch =
25  anyOf(isExpansionInSystemHeader(),
26  isMacroOrInvalidLocation(),
27  isTemplate,
28  isAutoAncestor,
29  hasAncestor(functionDecl()),
30  hasAncestor(userDefinedLiteral()),
31  hasAncestor(implicitCastExpr(hasMatchingCast())), /* will be catch by the walk down */
32  /* we do not like to introduce casts in parameter decls. This happens when
33  we have a template parameter with a base class. CharLiteralTest.cpp */
34  hasAncestor(parmVarDecl()));
35 
36  matcher.addMatcher(implicitCastExpr(unless(implicitCastMatch), hasMatchingCast()).bind("implicitCast"), this);
37 }
