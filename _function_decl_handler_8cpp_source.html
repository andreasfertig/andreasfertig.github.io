54 void FunctionDeclHandler::run(const MatchFinder::MatchResult& result)
55 {
56  if(const auto* funcDecl = result.Nodes.getNodeAs<FunctionDecl>("funcDecl")) {
57  const auto columnNr = GetSM(result).getSpellingColumnNumber(funcDecl->getBeginLoc()) - 1;
58  OutputFormatHelper outputFormatHelper{columnNr};
59  CodeGenerator codeGenerator{outputFormatHelper};
60 
61  codeGenerator.InsertArg(funcDecl);
62 
63  // Find the correct ending of the source range. In case of a declaration we need to find the ending semi,
64  // otherwise the provided source range is correct.
65  const auto sr = [&] {
66  auto funcRange =
67  GetSourceRangeAfterSemi(funcDecl->getSourceRange(),
68  result,
69  funcDecl->doesThisDeclarationHaveABody() ? RequireSemi::No : RequireSemi::Yes);
70 
71  // Adjust the begin location, if this decl has an attribute
72  if(funcDecl->hasAttrs()) {
73  // Find the first attribute with a valid source-location
74  for(const auto& attr : funcDecl->attrs()) {
75  if(const auto location = attr->getLocation(); location.isValid()) {
76 
77  // only use the begin location of the attribute, if it is before the one of the function decl.
78  if(funcRange.getBegin() > location) {
79  // the -3 are a guess that seems to work
80  funcRange.setBegin(location.getLocWithOffset(-3));
81  }
82 
83  return funcRange;
84  }
85  }
86  }
87 
88  return funcRange;
89  }();
90 
91  // DPrint("fd rw: %d %s\n", (sr.getBegin() == sr.getEnd()), outputFormatHelper.GetString());
92 
93  mRewrite.ReplaceText(sr, outputFormatHelper.GetString());
94  }
95 }
