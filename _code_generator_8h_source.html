183  template<typename T>
184  void InsertTemplateArgs(const ArrayRef<T>& array)
185  {
186  mOutputFormatHelper.Append('<');
187 
188  ForEachArg(array, [&](const auto& arg) { InsertTemplateArg(arg); });
189 
190  /* put as space between to closing brackets: >> -> > > */
191  if(mOutputFormatHelper.GetString().back() == '>') {
192  mOutputFormatHelper.Append(' ');
193  }
194 
195  mOutputFormatHelper.Append('>');
196  }
