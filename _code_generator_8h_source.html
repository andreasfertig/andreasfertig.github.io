169  template<typename T>
170  void InsertTemplateArgs(const T& t)
171  {
172  if constexpr(std::is_same_v<T, FunctionDecl>) {
173  if(const auto* tmplArgs = t.getTemplateSpecializationArgs()) {
174  InsertTemplateArgs(*tmplArgs);
175  }
176  } else if constexpr(std::is_same_v<T, VarTemplateSpecializationDecl>) {
177  InsertTemplateArgs(t.getTemplateArgs());
178 
179  } else if constexpr(requires { t.template_arguments(); }) {
180  if constexpr(std::is_same_v<DeclRefExpr, T>) {
181  if(0 == t.getNumTemplateArgs()) {
182  return;
183  }
184  }
185 
186  InsertTemplateArgs(t.template_arguments());
187 
188  } else if constexpr(requires { t.asArray(); }) {
189  InsertTemplateArgs(t.asArray());
190  }
191  }
