323 static auto*
324 FunctionBase(std::string_view name, QualType returnType, const params_vector& parameters, DeclContext* declCtx)
325 {
326  auto& ctx = GetGlobalAST();
327  SmallVector<QualType, 8> argTypes{};
328 
329  for(const auto& [_, type] : parameters) {
330  argTypes.push_back(type);
331  }
332 
333  FunctionDecl* fdd = FunctionDecl::Create(
334  const_cast<ASTContext&>(ctx),
335  declCtx,
336  {},
337  {},
338  &ctx.Idents.get(name),
339  ctx.getFunctionType(returnType, ArrayRef<QualType>{argTypes}, FunctionProtoType::ExtProtoInfo{}),
340  nullptr,
341  SC_None, // SC_Static,
342  false,
343  false,
344  false,
345  ConstexprSpecKind::Unspecified,
346  nullptr);
347  fdd->setImplicit(true);
348 
349  SmallVector<ParmVarDecl*, 8> paramVarDecls{};
350 
351  for(const auto& [name, type] : parameters) {
352  ParmVarDecl* param = Parameter(fdd, name, type);
353  param->setScopeInfo(0, 0);
354  paramVarDecls.push_back(param);
355  }
356 
357  fdd->setParams(paramVarDecls);
358 
359  return fdd;
360 }

543 static CallExpr* CallConstructor(QualType ctorType,
544  DeclRefExpr* lhsDeclRef,
545  Expr* lhsMemberExpr,
546  ArrayRef<Expr*> callParams,
547  DoCast doCast,
548  AsReference asReference)
549 {
550  if(nullptr == lhsMemberExpr) {
551  lhsMemberExpr = lhsDeclRef;
552  }
553 
554  auto* callCtor =
555  CreateFunctionDecl(StrCat("Constructor_"sv, GetName(ctorType)), {{kwInternalThis, lhsDeclRef->getType()}});
556 
557  SmallVector<Expr*, 8> modCallParams{};
558 
559  if(DoCast::Yes == doCast) {
560  modCallParams.push_back(StaticCast(lhsDeclRef->getType(), lhsDeclRef, false));
561 
562  } else {
563  if(AsReference::Yes == asReference) {
564  modCallParams.push_back(Ref(lhsMemberExpr));
565  } else {
566  modCallParams.push_back(lhsMemberExpr);
567  }
568  }
569 
570  modCallParams.append(callParams.begin(), callParams.end());
571 
572  return Call(callCtor, modCallParams);
573 }
