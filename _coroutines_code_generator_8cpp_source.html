380 /// \brief Find a \c SuspendsExpr's in a coroutine body statement for early transformation.
381 ///
382 /// Traverse the whole CoroutineBodyStmt to find all appearing \c VarDecl. These need to be rerouted to the
383 /// coroutine frame and hence prefixed by something like __f->. For that reason we only look for \c VarDecls
384 /// directly appearing in the body, \c CallExpr will be skipped.
385 class CoroutineASTTransformer : public StmtVisitor<CoroutineASTTransformer>
386 {
387  StmtsContainer mBodyStmts{};
388  Stmt* mPrevStmt{}; // used to insert the suspendexpr
389  CoroutineASTData& mASTData;
390  Stmt* mStaged{};
391  bool mSkip{};
392  bool mFinalSuspend{};
393  size_t& mSuspendsCount;
394  llvm::DenseMap<VarDecl*, MemberExpr*> mVarNamePrefix{};
395 
396 public:
397  CoroutineASTTransformer(CoroutineASTData& coroutineASTData,
398  size_t& suspendsCounter,
399  Stmt* stmt,
400  llvm::DenseMap<VarDecl*, MemberExpr*> varNamePrefix,
401  Stmt* prev = nullptr)
402  : mPrevStmt{prev}
403  , mASTData{coroutineASTData}
404  , mSuspendsCount{suspendsCounter}
405  , mVarNamePrefix{varNamePrefix}
406  {
407  if(nullptr == mPrevStmt) {
408  mPrevStmt = stmt;
409  }
410 
411  Visit(stmt);
412  }
413 
414  void Visit(Stmt* stmt)
415  {
416  if(stmt) {
417  StmtVisitor<CoroutineASTTransformer>::Visit(stmt);
418  }
419  }

472  {

650  void VisitCoroutineBodyStmt(CoroutineBodyStmt* stmt)
651  {
652  auto* varDecl = stmt->getPromiseDecl();
653 
654  mASTData.mPromiseField = AddField(mASTData, GetName(*varDecl), varDecl->getType());
655  auto* me = asthelpers::mkMemberExpr(mASTData.mFrameAccessDeclRef, mASTData.mPromiseField);
656 
657  mVarNamePrefix.insert(std::make_pair(varDecl, me));
658 
659  auto& ctx = GetGlobalAST();
660 
661  // add the suspend index variable
662  mASTData.mSuspendIndexField = AddField(mASTData, SUSPEND_INDEX_NAME, ctx.IntTy);
663  mASTData.mSuspendIndexAccess =
664  asthelpers::mkMemberExpr(mASTData.mFrameAccessDeclRef, mASTData.mSuspendIndexField);
665 
666  // https://timsong-cpp.github.io/cppwp/n4861/dcl.fct.def.coroutine#5.3
667  mASTData.mInitialAwaitResumeCalledField = AddField(mASTData, INITIAL_AWAIT_SUSPEND_CALLED_NAME, ctx.BoolTy);
668  mASTData.mInitialAwaitResumeCalledAccess =
669  asthelpers::mkMemberExpr(mASTData.mFrameAccessDeclRef, mASTData.mInitialAwaitResumeCalledField);
670 
671  for(auto* param : stmt->getParamMoves()) {
672  if(auto* declStmt = dyn_cast_or_null<DeclStmt>(param)) {
673  if(auto* varDecl2 = dyn_cast_or_null<VarDecl>(declStmt->getSingleDecl())) {
674  // For the captured parameters we need to find the ParmVarDecl instead of the newly created VarDecl
675  if(auto* declRef = FindDeclRef(varDecl2->getAnyInitializer())) {
676  auto* varDecl = dyn_cast<ParmVarDecl>(declRef->getDecl());
677 
678  auto* field = AddField(mASTData, GetName(*varDecl), varDecl->getType());
679  auto* me = asthelpers::mkMemberExpr(mASTData.mFrameAccessDeclRef, field);
680 
681  mVarNamePrefix.insert(std::make_pair(const_cast<ParmVarDecl*>(varDecl), me));
682  }
683  }
684  }
685  }

688

690  Visit(stmt->getReturnValue());
691  Visit(stmt->getReturnValueInit());

693  Visit(stmt->getReturnStmtOnAllocFailure());
694

696

723  mFSMName = [&] {
724  OutputFormatHelper ofm{};
725  CodeGenerator codeGenerator{ofm};
726 
727  // Coroutines can be templates and then we end up with the same FSM name but different template parameters.
728  // XXX: This will fail with NTTP's like 3.14
729  if(const auto* args = fd.getTemplateSpecializationArgs()) {
730  ofm.Append('_');
731  OnceFalse needsUnderscore{};
732 
733  for(const auto& arg : args->asArray()) {
734  if(needsUnderscore) {
735  ofm.Append('_');
736  }
737 
738  codeGenerator.InsertTemplateArg(arg);
739  }
740  }
741 
742  auto& str = ofm.GetString();
743  ReplaceAll(str, "<"sv, ""sv);
744  ReplaceAll(str, ":"sv, ""sv);
745  ReplaceAll(str, ">"sv, ""sv);
746  ReplaceAll(str, ","sv, ""sv);
747  ReplaceAll(str, " "sv, ""sv);
748 
749  if(fd.isOverloadedOperator()) {
750  return StrCat(MakeLineColumnName(ctx.getSourceManager(), stmt->getBeginLoc(), "operator_"sv), str);
751  } else {
752  return StrCat(GetName(fd), str);
753  }
754  }();
755 
756  mFrameName = BuildInternalVarName(StrCat(mFSMName, "Frame"sv));
757 
758  // Insert a made up struct which holds the "captured" parameters stored in the coroutine frame
759  mASTData.mFrameType = asthelpers::mkCXXRecordDecl(mFrameName);
760  mASTData.mFrameType->startDefinition();
761 
762  // A "normal" struct has itself attached as a Decl. To make everything work do the same thing here
763  auto* selfDecl = asthelpers::mkCXXRecordDecl(mFrameName);
764  selfDecl->setAccess(AS_public);
765  mASTData.mFrameType->addDecl(selfDecl);
766 
767  auto* frameAccess = asthelpers::mkVarDecl(CORO_FRAME_NAME, GetFrameType());
768  mASTData.mFrameAccessDeclRef = asthelpers::mkDeclRefExpr(frameAccess);
769 
770  // The coroutine frame starts with two function pointers to the resume and destroy function. See:
771  // https://gcc.gnu.org/legacy-ml/gcc-patches/2020-01/msg01096.html:
772  // "The ABI mandates that pointers into the coroutine frame point to an area
773  // begining with two function pointers (to the resume and destroy functions
774  // described below); these are immediately followed by the "promise object"
775  // described in the standard."
776  //
777  // and
778  // https://llvm.org/docs/Coroutines.html#id72 "Coroutine Representation"
779  auto* resumeFnFd =
780  asthelpers::mkFunctionDecl(hlpResumeFn, ctx.VoidTy, {{CORO_FRAME_NAME, ctx.getPointerType(GetFrameType())}});
781  auto resumeFnType = ctx.getPointerType(resumeFnFd->getType());
782  mASTData.mResumeFnField = AddField(hlpResumeFn, resumeFnType);
783 
784  auto* destroyFnFd =
785  asthelpers::mkFunctionDecl(hlpDestroyFn, ctx.VoidTy, {{CORO_FRAME_NAME, ctx.getPointerType(GetFrameType())}});
786  auto destroyFnType = ctx.getPointerType(destroyFnFd->getType());
787  mASTData.mDestroyFnField = AddField(hlpDestroyFn, destroyFnType);
788 
789  // Allocated the made up frame
790  mOutputFormatHelper.AppendCommentNewLine("Allocate the frame including the promise"sv);
791  mOutputFormatHelper.AppendCommentNewLine("Note: The actual parameter new is __builtin_coro_size"sv);
792 
793  auto* coroFrameVar = asthelpers::mkVarDecl(CORO_FRAME_NAME, GetFramePointerType());
794 
795  CXXReinterpretCastExpr* reicast =
796  CXXReinterpretCastExpr::Create(ctx,
797  GetFramePointerType(),
798  VK_LValue,
799  CK_BitCast,
800  stmt->getAllocate(),
801  nullptr,
802  ctx.getTrivialTypeSourceInfo(GetFramePointerType()),
803  {},
804  {},
805  {});
806 
807  coroFrameVar->setInit(reicast);
808 
809  InsertArg(coroFrameVar);
810 
811  // P0057R8: [dcl.fct.def.coroutine] p8: get_return_object_on_allocation_failure indicates that new may return a
812  // nullptr. In this case return get_return_object_on_allocation_failure.
813  if(stmt->getReturnStmtOnAllocFailure()) {
814  auto* nptr = new(ctx) CXXNullPtrLiteralExpr({});
815  auto* bop = asthelpers::mkBinaryOperator(nptr, mASTData.mFrameAccessDeclRef, BO_EQ, ctx.BoolTy);
816 
817  // Create an IfStmt.
818  StmtsContainer bodyStmts{stmt->getReturnStmtOnAllocFailure()};
819  auto* ifStmt = asthelpers::mkIfStmt(bop, bodyStmts);
820 
821  mOutputFormatHelper.AppendNewLine();
822  InsertArg(ifStmt);
823  }
824 
825  CoroutineASTTransformer{
826  mASTData, mSuspendsCounter, const_cast<CoroutineBodyStmt*>(stmt), llvm::DenseMap<VarDecl*, MemberExpr*>{}};
827 
828  // set initial suspend count to zero.
829  auto* setSuspendIndexToZero = asthelpers::mkAssign(
830  mASTData.mFrameAccessDeclRef, mASTData.mSuspendIndexField, asthelpers::mkIntegerLiteral32(0));
831  InsertArgWithNull(setSuspendIndexToZero);
832 
833  // https://timsong-cpp.github.io/cppwp/n4861/dcl.fct.def.coroutine#5.3
834  auto* initializeInitialAwaitResume = asthelpers::mkAssign(
835  mASTData.mFrameAccessDeclRef, mASTData.mInitialAwaitResumeCalledField, asthelpers::mkBoolLiteral(false));
836  InsertArgWithNull(initializeInitialAwaitResume);
837 
838  // Move the parameters first
839  for(auto* param : stmt->getParamMoves()) {
840  if(const auto* declStmt = dyn_cast_or_null<DeclStmt>(param)) {
841  if(const auto* varDecl = dyn_cast_or_null<VarDecl>(declStmt->getSingleDecl())) {
842  const auto varName = GetName(*varDecl);
843 
844  mOutputFormatHelper.AppendNewLine(CORO_FRAME_ACCESS,
845  varName,
846  " = "sv,
847  "std::forward<"sv,
848  GetName(varDecl->getType()),
849  ">("sv,
850  varName,
851  ");"sv);
852  }
853  }
854  }
855 
856  std::vector<Expr*> exprs{};
857 
858  // According to https://eel.is/c++draft/dcl.fct.def.coroutine#5.7 the promise_type constructor can have
859  // parameters. If so, they must be equal to the coroutines function parameters.
860  // The code here performs a _simple_ lookup for a matching ctor without using Clang's overload resolution.
861  ArrayRef<ParmVarDecl*> funParams = fd.parameters();
862  std::vector<ParmVarDecl*> funParamStorage{};
863  QualType cxxMethodType{};
864 
865  if(const auto* cxxMethodDecl = dyn_cast_or_null<CXXMethodDecl>(&fd)) {
866  funParamStorage.reserve(funParams.size() + 1);
867 
868  cxxMethodType = cxxMethodDecl->getThisObjectType();
869 
870  // In case we have a member function the first parameter is a reference to this. The following code injects
871  // this parameter.
872  funParamStorage.push_back(ParmVarDecl::Create(const_cast<ASTContext&>(ctx),
873  const_cast<FunctionDecl*>(&fd),
874  {},
875  {},
876  &ctx.Idents.get(CORO_FRAME_ACCESS_THIS),
877  cxxMethodType,
878  nullptr,
879  SC_None,
880  nullptr));
881 
882  // XXX: use ranges once available
883  std::copy(funParams.begin(), funParams.end(), std::back_inserter(funParamStorage));
884 
885  funParams = funParamStorage;
886  }
887 
888  auto getNonRefType = [&](auto* var) -> QualType {
889  if(const auto* et = var->getType().getNonReferenceType()->template getAs<ElaboratedType>()) {
890  return et->getNamedType();
891  } else {
892  return QualType(var->getType().getNonReferenceType().getTypePtrOrNull(), 0);
893  }
894  };
895 
896  for(auto* promiseTypeRecordDecl = mASTData.mPromiseField->getType()->getAsCXXRecordDecl();
897  auto* ctor : promiseTypeRecordDecl->ctors()) {
898 
899  // XXX: use ranges once available
900  if(not std::equal(
901  ctor->param_begin(), ctor->param_end(), funParams.begin(), funParams.end(), [&](auto& a, auto& b) {
902  return getNonRefType(a) == getNonRefType(b);
903  })) {
904  continue;
905  }
906 
907  // In case of a promise ctor which takes this as the first argument, that parameter needs to be deferences,
908  // as it can only be taken as a reference.
909  OnceTrue derefFirstParam{};
910 
911  if(not ctor->param_empty() and
912  (getNonRefType(ctor->getParamDecl(0)) == QualType(cxxMethodType.getTypePtrOrNull(), 0))) {
913  if(0 == mASTData.mThisExprs.size()) {
914  mASTData.mThisExprs.push_back(
915 #if IS_CLANG_NEWER_THAN(17)
916  CXXThisExpr::Create(ctx, {}, ctx.getPointerType(cxxMethodType), false)
917 #else
918  new(ctx) CXXThisExpr{{}, ctx.getPointerType(cxxMethodType), false}
919 #endif
920 
921  );
922  }
923  } else {
924  (void)static_cast<bool>(derefFirstParam); // set it to false
925  }
926 
927  for(const auto& fparam : funParams) {
928  if(derefFirstParam) {
929  exprs.push_back(
930  asthelpers::mkUnaryOperator(asthelpers::mkDeclRefExpr(fparam), UO_Deref, fparam->getType()));
931 
932  } else {
933  exprs.push_back(asthelpers::mkMemberExpr(mASTData.mFrameAccessDeclRef, fparam));
934  }
935  }
936 
937  if(funParams.size()) {
938  // The <new> header needs to be included.
939  mHaveCoroutine = true;
940  }
941 
942  break; // We've found what we were looking for
943  }
944 
945  if(mASTData.mThisExprs.size()) {
946  mOutputFormatHelper.AppendNewLine(CORO_FRAME_ACCESS_THIS, " = this;"sv);
947  }
948 
949  // Now call the promise ctor, as it may access some of the parameters it comes at this point.
950  mOutputFormatHelper.AppendNewLine();
951  mOutputFormatHelper.AppendCommentNewLine("Construct the promise."sv);
952  auto* me = asthelpers::mkMemberExpr(mASTData.mFrameAccessDeclRef, mASTData.mPromiseField);
953  auto* asRef = asthelpers::mkUnaryOperator(me, UO_AddrOf, mASTData.mPromiseField->getType());
954 
955  auto* ctorArgs = new(ctx) InitListExpr{ctx, {}, static_cast<ArrayRef<Expr*>>(exprs), {}};
956 
957  CXXNewExpr* newFrame =
958  CXXNewExpr::Create(ctx,
959  false,
960  nullptr,
961  nullptr,
962  true,
963  false,
964  {asRef},
965  SourceRange{},
966 #if IS_CLANG_NEWER_THAN(15)
967  std::optional<Expr*>{},
968 #else
969  Optional<Expr*>{},
970 #endif
971  CXXNewExpr::ListInit,
972  ctorArgs,
973  ctx.getPointerType(mASTData.mPromiseField->getType()),
974  ctx.getTrivialTypeSourceInfo(ctx.getPointerType(mASTData.mPromiseField->getType())),
975  SourceRange{},
976  SourceRange{});
977 
978  InsertArgWithNull(newFrame);
979 
980  // Add parameters from the original function to the list
981 
982  // P0057R8: [dcl.fct.def.coroutine] p5: before initial_suspend and at tops 1
983 #if not IS_CLANG_NEWER_THAN(14)
984  mOutputFormatHelper.AppendNewLine();
985  InsertArg(stmt->getResultDecl());
986 #endif
987 
988  // Make a call to the made up state machine function for the initial suspend
989  mOutputFormatHelper.AppendNewLine();
990 
991  // [dcl.fct.def.coroutine]
992  mOutputFormatHelper.AppendCommentNewLine("Forward declare the resume and destroy function."sv);
993 
994  auto* fsmFuncDecl = CreateFunctionDecl(StrCat(mFSMName, "Resume"sv), {{CORO_FRAME_NAME, GetFramePointerType()}});
995  InsertArg(fsmFuncDecl);
996  auto* deallocFuncDecl =
997  CreateFunctionDecl(StrCat(mFSMName, "Destroy"sv), {{CORO_FRAME_NAME, GetFramePointerType()}});
998  InsertArg(deallocFuncDecl);
999 
1000  mOutputFormatHelper.AppendNewLine();
1001 
1002  mOutputFormatHelper.AppendCommentNewLine("Assign the resume and destroy function pointers."sv);
1003 
1004  auto* assignResumeFn = asthelpers::mkAssign(
1005  mASTData.mFrameAccessDeclRef, mASTData.mResumeFnField, asthelpers::mkFunctionReference(fsmFuncDecl));
1006  InsertArgWithNull(assignResumeFn);
1007 
1008  auto* assignDestroyFn = asthelpers::mkAssign(
1009  mASTData.mFrameAccessDeclRef, mASTData.mDestroyFnField, asthelpers::mkFunctionReference(deallocFuncDecl));
1010  InsertArgWithNull(assignDestroyFn);
1011  mOutputFormatHelper.AppendNewLine();
1012 
1013  mOutputFormatHelper.AppendCommentNewLine(
1014  R"A(Call the made up function with the coroutine body for initial suspend.
1015  This function will be called subsequently by coroutine_handle<>::resume()
1016  which calls __builtin_coro_resume(__handle_))A"sv);
1017 
1018  auto* callCoroFSM = asthelpers::mkCallExpr(fsmFuncDecl, fsmFuncDecl->getType(), {mASTData.mFrameAccessDeclRef});
1019  InsertArgWithNull(callCoroFSM);
1020 
1021  mOutputFormatHelper.AppendNewLine();
1022  mOutputFormatHelper.AppendNewLine();
1023 
1024  InsertArg(stmt->getReturnStmt());
1025 
1026  mOutputFormatHelper.AppendSemiNewLine();
1027 
1028  mOutputFormatHelper.CloseScope(OutputFormatHelper::NoNewLineBefore::Yes);
1029  mOutputFormatHelper.AppendNewLine();
1030  mOutputFormatHelper.AppendNewLine();
1031

1033  StmtsContainer fsmFuncBodyStmts{stmt};
1034 
1035  mOutputFormatHelper.AppendCommentNewLine("This function invoked by coroutine_handle<>::resume()"sv);
1036  SetFunctionBody(fsmFuncDecl, fsmFuncBodyStmts);
1037  InsertArg(fsmFuncDecl);
1038 
1039  mASTData.mDoInsertInDtor = true; // As we have a coroutine insert the frame when this object goes out of scope.
1040 
1041 #if 0 // Preserve for later. Technically the destructor for the entire frame that's made up below takes care of
1042  // everything.
1043 
1044  // A destructor is only present, if they promise_type or one of its members is non-trivially destructible.
1045  if(auto* dtor = mASTData.mPromiseField->getType()->getAsCXXRecordDecl()->getDestructor()) {
1046  deallocFuncBodyStmts.Add(asthelpers::mkInsightsComment("Deallocating the coroutine promise type"sv));
1047

1049  auto* deallocPromise = asthelpers::mkMemberExpr(promiseAccess, dtor, false);
1050  auto* dtorCall = asthelpers::mkMemberCallExpr(deallocPromise, dtor->getType());
1051  deallocFuncBodyStmts.Add(dtorCall);
1052 
1053  } else {
1054  deallocFuncBodyStmts.Add(
1055  asthelpers::mkInsightsComment("promise_type is trivially destructible, no dtor required."sv));
1056  }
1057 #endif
1058

1060  // member in the frame. The deallocation function:
1061  // https://devblogs.microsoft.com/oldnewthing/20210331-00/?p=105028
1062  mOutputFormatHelper.AppendNewLine();
1063  mOutputFormatHelper.AppendCommentNewLine("This function invoked by coroutine_handle<>::destroy()"sv);
1064 
1065  StmtsContainer deallocFuncBodyStmts{asthelpers::mkInsightsComment("destroy all variables with dtors"sv)};
1066 
1067  auto* dtorFuncDecl = asthelpers::mkFunctionDecl(
1068  StrCat("~"sv, GetName(*mASTData.mFrameType)), ctx.VoidTy, {{CORO_FRAME_NAME, GetFramePointerType()}});
1069  auto* deallocPromise = asthelpers::mkMemberExpr(mASTData.mFrameAccessDeclRef, dtorFuncDecl);
1070  auto* dtorCall = asthelpers::mkMemberCallExpr(deallocPromise, GetFrameType());
1071  deallocFuncBodyStmts.Add(dtorCall);
1072

1074  deallocFuncBodyStmts.Add(asthelpers::mkInsightsComment(
1075  "Note: The actual argument to delete is __builtin_coro_frame with the promise as parameter"sv));
1076

1084 void CoroutinesCodeGenerator::InsertArg(const CoroutineBodyStmt* stmt)
1085 {
1086  auto& ctx = GetGlobalAST();
1087 
1088  // insert a made up switch for continuing a resume
1089  SwitchStmt* sstmt = SwitchStmt::Create(ctx, nullptr, nullptr, mASTData.mSuspendIndexAccess, {}, {});
1090 
1091  // insert 0 with break for consistency
1092  auto* initialSuspendCase = asthelpers::mkCaseStmt(0, new(ctx) BreakStmt(SourceLocation{}));
1093  StmtsContainer switchBodyStmts{initialSuspendCase};
1094 
1095  for(const auto& i : NumberIterator{mSuspendsCounter}) {
1096  switchBodyStmts.Add(asthelpers::mkCaseStmt(i + 1, asthelpers::mkGotoStmt(BuildResumeLabelName(i + 1))));
1097  }
1098 
1099  auto* switchBody = asthelpers::mkCompoundStmt(switchBodyStmts);
1100  sstmt->setBody(switchBody);
1101 
1102  StmtsContainer funcBodyStmts{asthelpers::mkInsightsComment("Create a switch to get to the correct resume point"sv),
1103  sstmt,
1104  stmt->getInitSuspendStmt()};
1105 
1106  // insert the init suspend expr
1107  mState = eState::InitialSuspend;
1108 
1109  if(mASTData.mThisExprs.size()) {
1110  AddField(kwInternalThis, mASTData.mThisExprs.at(0)->getType());
1111  }
1112 
1113  mInsertVarDecl = false;
1114  mSupressRecordDecls = true;
1115 
1116  for(const auto* c : stmt->getBody()->children()) {
1117  funcBodyStmts.Add(c);
1118  }
1119 
1120  // InsertArg(stmt->getFallthroughHandler());
1121 
1122  auto* gotoFinalSuspend = asthelpers::mkGotoStmt(FINAL_SUSPEND_NAME);
1123  funcBodyStmts.Add(gotoFinalSuspend);
1124 
1125  auto* body = [&]() -> Stmt* {
1126  auto* tryBody = asthelpers::mkCompoundStmt(funcBodyStmts);
1127 
1128  // First open the try-catch block, as we get an error when jumping across such blocks with goto
1129  if(const auto* exceptionHandler = stmt->getExceptionHandler()) {
1130  // If we encounter an exceptionbefore inital_suspend's await_suspend was called we re-throw the
1131  // exception.
1132  auto* throwExpr = new(ctx) CXXThrowExpr(nullptr, {}, {}, false);
1133  auto* ifStmt = asthelpers::mkNotIfStmt(mASTData.mInitialAwaitResumeCalledAccess, throwExpr);
1134 
1135  StmtsContainer catchBodyStmts{ifStmt, exceptionHandler};
1136  auto* catchAll = new(ctx) CXXCatchStmt({}, nullptr, asthelpers::mkCompoundStmt(catchBodyStmts));
1137 
1138  return CXXTryStmt::Create(ctx, {}, tryBody, {catchAll});
1139  }
1140 
1141  return tryBody;
1142  }();
1143 
1144  InsertArg(body);
1145 
1146  mOutputFormatHelper.AppendNewLine();
1147 
1148  auto* finalSuspendLabel = asthelpers::mkLabelStmt(FINAL_SUSPEND_NAME);
1149  InsertArg(finalSuspendLabel);
1150  mState = eState::FinalSuspend;
1151  InsertArg(stmt->getFinalSuspendStmt());
1152 
1153  // disable prefixing names and types
1154  mInsertVarDecl = true;
1155 }

1289  /// that returns 'false':
1290  /// -- execution of the coroutine is suspended
1291  /// -- the 'suspend' expression is evaluated
1292  /// -- if the 'suspend' expression returns 'false', the coroutine is
1293  /// resumed
1294  /// -- otherwise, control passes back to the resumer.

1297  /// expression.

1331  }

1337

1339  bodyStmts.Add(asthelpers::mkReturnStmt());
1340  }

1343

1345

1351  }

1354  auto* ifReady = asthelpers::mkNotIfStmt(stmt->getReadyExpr(), ifSuspend);
1355

1362  mOutputFormatHelper.AppendNewLine();
1363  }
1364 
1365  RETURN_IF(eState::FinalSuspend == mState);
1366

1368  InsertArg(suspendLabel);
1369

1377  mOutputFormatHelper.Append(CORO_FRAME_ACCESS, fieldName, hlpAssing);
1378 
1379  AddField(fieldName, resumeExpr->getType());

1381  }
1382
