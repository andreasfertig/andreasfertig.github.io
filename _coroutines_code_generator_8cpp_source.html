298  /// Traverse the whole CoroutineBodyStmt to find all appearing \c VarDecl. These need to be rerouted to the
299  /// coroutine frame and hence prefixed by something like __f->. For that reason we only look for \c VarDecls
300  /// directly appearing in the body, \c CallExpr will be skipped.
301  void FindAllVarDecls(const Stmt* stmt)
302  {
303  if(nullptr == stmt) {
304  return;
305  }
306 
307  // Take this expressions to rewrite access when the coroutine is a class method.
308  if(const auto* thisExpr = dyn_cast_or_null<CXXThisExpr>(stmt)) {
309  mThisExprs.push_back(thisExpr);
310 
311  } else if(const auto* declStmt = dyn_cast_or_null<DeclStmt>(stmt)) {
312  const auto* singleDecl = declStmt->getSingleDecl();
313 
314  if(const auto* vd = dyn_cast_or_null<VarDecl>(singleDecl)) {
315  push_back(vd);
316  return;
317 
318  } else if(const auto* recordDecl = dyn_cast_or_null<CXXRecordDecl>(singleDecl)) {
319  mRecordDecls.push_back(recordDecl);
320  }
321  }
322 
323  for(const auto* child : stmt->children()) {
324  FindAllVarDecls(child);
325  }
326  }

437 void CoroutinesCodeGenerator::InsertCoroutine(const FunctionDecl& fd, const CoroutineBodyStmt* stmt)
438 {
439  mOutputFormatHelper.OpenScope();
440 
441  const auto* promiseDecl = stmt->getPromiseDecl();
442  mDeclsHolder.push_back(promiseDecl);
443 
444  auto& ctx = GetGlobalAST();
445 
446  mFSMName = [&] {
447  OutputFormatHelper ofm{};
448  CodeGenerator codeGenerator{ofm};
449 
450  // Coroutines can be templates and then we end up with the same FSM name but different template parameters.
451  // XXX: This will fail with NTTP's like 3.14
452  if(const auto* args = fd.getTemplateSpecializationArgs()) {
453  ofm.Append('_');
454  OnceFalse needsUnderscore{};
455 
456  for(const auto& arg : args->asArray()) {
457  if(needsUnderscore) {
458  ofm.Append('_');
459  }
460 
461  codeGenerator.InsertTemplateArg(arg);
462  }
463  }
464 
465  auto& str = ofm.GetString();
466  ReplaceAll(str, "<"sv, ""sv);
467  ReplaceAll(str, ":"sv, ""sv);
468  ReplaceAll(str, ">"sv, ""sv);
469  ReplaceAll(str, ","sv, ""sv);
470  ReplaceAll(str, " "sv, ""sv);
471 
472  if(fd.isOverloadedOperator()) {
473  return StrCat(MakeLineColumnName(ctx.getSourceManager(), stmt->getBeginLoc(), "operator_"sv), str);
474  } else {
475  return StrCat(GetName(fd), str);
476  }
477  }();
478 
479  mFrameName = BuildInternalVarName(StrCat(mFSMName, "Frame"sv));
480 
481  // Insert a made up struct which holds the "captured" parameters stored in the coroutine frame
482  mASTData.mFrameType = asthelpers::mkCXXRecordDecl(mFrameName);
483  mASTData.mFrameType->startDefinition();
484 
485  // A "normal" struct has itself attached as a Decl. To make everything work do the same thing here
486  auto* selfDecl = asthelpers::mkCXXRecordDecl(mFrameName);
487  selfDecl->setAccess(AS_public);
488  mASTData.mFrameType->addDecl(selfDecl);
489 
490  auto* frameAccess = asthelpers::mkVarDecl(CORO_FRAME_NAME, GetFrameType());
491  mASTData.mFrameAccessDeclRef = asthelpers::mkDeclRefExpr(frameAccess);
492 
493  // The coroutine frame starts with two function pointers to the resume and destroy function. See:
494  // https://gcc.gnu.org/legacy-ml/gcc-patches/2020-01/msg01096.html and https://llvm.org/docs/Coroutines.html#id72
495  // "Coroutine Representation"
496  auto* resumeFnFd =
497  asthelpers::mkFunctionDecl(hlpResumeFn, ctx.VoidTy, {{CORO_FRAME_NAME, ctx.getPointerType(GetFrameType())}});
498  auto resumeFnType = ctx.getPointerType(resumeFnFd->getType());
499  mASTData.mResumeFnField = AddField(hlpResumeFn, resumeFnType);
500 
501  auto* destroyFnFd =
502  asthelpers::mkFunctionDecl(hlpDestroyFn, ctx.VoidTy, {{CORO_FRAME_NAME, ctx.getPointerType(GetFrameType())}});
503  auto destroyFnType = ctx.getPointerType(destroyFnFd->getType());
504  mASTData.mDestroyFnField = AddField(hlpDestroyFn, destroyFnType);
505 
506  SKIP_NAME_PREFIX_FOR_SCOPE
507  {
508  mASTData.mPromiseField = AddField(GetName(*promiseDecl), promiseDecl->getType());
509  // add the suspend index variable
510  mASTData.mSuspendIndexField = AddField(SUSPEND_INDEX_NAME, ctx.IntTy);
511  mASTData.mSuspendIndexAccess =
512  asthelpers::mkMemberExpr(mASTData.mFrameAccessDeclRef, mASTData.mSuspendIndexField);
513 
514  // https://timsong-cpp.github.io/cppwp/n4861/dcl.fct.def.coroutine#5.3
515  mASTData.mInitialAwaitResumeCalledField = AddField(INITIAL_AWAIT_SUSPEND_CALLED_NAME, ctx.BoolTy);
516  mASTData.mInitialAwaitResumeCalledAccess =
517  asthelpers::mkMemberExpr(mASTData.mFrameAccessDeclRef, mASTData.mInitialAwaitResumeCalledField);
518  }
519 
520  for(auto* param : stmt->getParamMoves()) {
521  if(const auto* declStmt = dyn_cast_or_null<DeclStmt>(param)) {
522  if(const auto* varDecl = dyn_cast_or_null<VarDecl>(declStmt->getSingleDecl())) {
523  // For the captured parameters we need to find the ParmVarDecl instead of the newly created VarDecl
524  if(const auto* declRef = FindDeclRef(varDecl->getAnyInitializer())) {
525  varDecl = dyn_cast<ParmVarDecl>(declRef->getDecl());
526  }
527 
528  mDeclsHolder.push_back(varDecl);
529  }
530  }
531  }
532 
533  // Allocated the made up frame
534  mOutputFormatHelper.AppendCommentNewLine("Allocate the frame including the promise"sv);
535 
536  auto* coroFrameVar = asthelpers::mkVarDecl(CORO_FRAME_NAME, GetFramePointerType());
537 
538  CXXReinterpretCastExpr* reicast =
539  CXXReinterpretCastExpr::Create(ctx,
540  GetFramePointerType(),
541  VK_LValue,
542  CK_BitCast,
543  stmt->getAllocate(),
544  nullptr,
545  ctx.getTrivialTypeSourceInfo(GetFramePointerType()),
546  {},
547  {},
548  {});
549 
550  coroFrameVar->setInit(reicast);
551 
552  InsertArg(coroFrameVar);
553 
554  // P0057R8: [dcl.fct.def.coroutine] p8: get_return_object_on_allocation_failure indicates that new may return a
555  // nullptr. In this case return get_return_object_on_allocation_failure.
556  if(stmt->getReturnStmtOnAllocFailure()) {
557  auto* nptr = new(ctx) CXXNullPtrLiteralExpr({});
558  auto* bop = asthelpers::mkBinaryOperator(nptr, mASTData.mFrameAccessDeclRef, BO_EQ, ctx.BoolTy);
559 
560  // Create an IfStmt.
561  StmtsContainer bodyStmts{stmt->getReturnStmtOnAllocFailure()};
562  auto* ifStmt = asthelpers::mkIfStmt(bop, bodyStmts);
563 
564  mOutputFormatHelper.AppendNewLine();
565  InsertArg(ifStmt);
566  }
567 
568  // set initial suspend count to zero.
569  auto* setSuspendIndexToZero = asthelpers::mkAssign(
570  mASTData.mFrameAccessDeclRef, mASTData.mSuspendIndexField, asthelpers::mkIntegerLiteral32(0));
571  InsertArgWithNull(setSuspendIndexToZero);
572 
573  // https://timsong-cpp.github.io/cppwp/n4861/dcl.fct.def.coroutine#5.3
574  auto* initializeInitialAwaitResume = asthelpers::mkAssign(
575  mASTData.mFrameAccessDeclRef, mASTData.mInitialAwaitResumeCalledField, asthelpers::mkBoolLiteral(false));
576  InsertArgWithNull(initializeInitialAwaitResume);
577 
578  // Move the parameters first
579  for(auto* param : stmt->getParamMoves()) {
580  if(const auto* declStmt = dyn_cast_or_null<DeclStmt>(param)) {
581  if(const auto* varDecl = dyn_cast_or_null<VarDecl>(declStmt->getSingleDecl())) {
582  const auto varName = GetName(*varDecl);
583 
584  mOutputFormatHelper.AppendNewLine(CORO_FRAME_ACCESS,
585  varName,
586  " = "sv,
587  "std::forward<"sv,
588  GetName(varDecl->getType()),
589  ">("sv,
590  varName,
591  ");"sv);
592  }
593  }
594  }
595 
596  mDeclsHolder.FindAllVarDecls(stmt->getBody());
597 
598  // cache the entries. At least in a co_return co_await + co_await we see the operands multiple times
599  llvm::DenseMap<const Stmt*, bool> exprs{};
600 
601  // -1 because of the final suspend which needs no label
602  auto FindAllSuspends = [&](const Stmt* stmt, auto& self) {
603  if(nullptr == stmt) {
604  return 0;
605  }
606 
607  int count = 0;
608 
609  if(isa<CoroutineSuspendExpr>(stmt)) {
610  if(not Contains(exprs, stmt)) {
611  ++count;
612  exprs.insert(std::make_pair(stmt, true));
613  }
614  }
615 
616  for(const auto* child : stmt->children()) {
617  count += self(child, self);
618  }
619 
620  return count;
621  };
622 
623  mSuspendsCounter = FindAllSuspends(fd.getBody(), FindAllSuspends) - 1;
624 
625  SCOPED_VAR_TOGGLE(mDeclsHolder.EmulateEmpty)
626  {
627  for(const auto* thisExpr : mDeclsHolder.mThisExprs) {
628  mOutputFormatHelper.AppendNewLine(CORO_FRAME_ACCESS_THIS, " = this;"sv);
629  }
630 
631  // Now call the promise ctor, as it may access some of the parameters it comes at this point.
632  mOutputFormatHelper.AppendNewLine();
633  mOutputFormatHelper.AppendCommentNewLine("Construct the promise."sv);
634  // XXX: Can the promise have parameters?
635  auto* me = asthelpers::mkMemberExpr(mASTData.mFrameAccessDeclRef, mASTData.mPromiseField);
636  auto* asRef = asthelpers::mkUnaryOperator(me, UO_AddrOf, mASTData.mPromiseField->getType());
637 
638  CXXNewExpr* newFrame =
639  CXXNewExpr::Create(ctx,
640  false,
641  nullptr,
642  nullptr,
643  true,
644  false,
645  {asRef},
646  SourceRange{},
647  Optional<Expr*>{},
648  CXXNewExpr::ListInit,
649  new(ctx) InitListExpr{{}},
650  ctx.getPointerType(mASTData.mPromiseField->getType()),
651  ctx.getTrivialTypeSourceInfo(ctx.getPointerType(mASTData.mPromiseField->getType())),
652  SourceRange{},
653  SourceRange{});
654 
655  InsertArgWithNull(newFrame);
656

1161

1165  // \code
1166  // co_return co_await simpleReturn(v) + co_await simpleReturn(v);
1167  // \endcode
1168  //
1169  // If we just insert \c getPromiseCall we end up having a expanded coroutine suspend expression in a call, which
1170  // would look like this: \code
1171  // __f->__promise.return_value(__f->__promise_10_24 = simpleReturn(__f->v); ... );
1172  // \endcode
1173  // Which is invalid code. To avoid this, we traverse with \c getOperand first the operand only. During the

1186  codeGenerator.InsertArg(op);
1187 
1188  mBinaryExprs.insert(codeGenerator.mBinaryExprs.begin(), codeGenerator.mBinaryExprs.end());
1189  }
1190 
1191  if(const auto fieldsAfter = std::distance(mASTData.mFrameType->field_begin(), mASTData.mFrameType->field_end());
1192  fieldsAfter != fieldsBefore) {
1193  mOutputFormatHelper.AppendNewLine(outputFormatHelper.GetString());

1200  InsertArg(stmt->getPromiseCall());

1202
