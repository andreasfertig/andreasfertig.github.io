103 /// \brief Find a \c SuspendsExpr's in a coroutine body statement for early transformation.
104 ///
105 /// Traverse the whole CoroutineBodyStmt to find all appearing \c VarDecl. These need to be rerouted to the
106 /// coroutine frame and hence prefixed by something like __f->. For that reason we only look for \c VarDecls
107 /// directly appearing in the body, \c CallExpr will be skipped.
108 class CoroutineASTTransformer : public StmtVisitor<CoroutineASTTransformer>
109 {
110  StmtsContainer mBodyStmts{};
111  Stmt* mPrevStmt{}; // used to insert the suspendexpr
112  CoroutineASTData& mASTData;
113  Stmt* mStaged{};
114  bool mSkip{};
115  bool mFinalSuspend{};
116  size_t& mSuspendsCount;
117  llvm::DenseMap<VarDecl*, MemberExpr*> mVarNamePrefix{};
118 
119 public:
120  CoroutineASTTransformer(CoroutineASTData& coroutineASTData,
121  size_t& suspendsCounter,
122  Stmt* stmt,
123  llvm::DenseMap<VarDecl*, MemberExpr*> varNamePrefix,
124  Stmt* prev = nullptr)
125  : mPrevStmt{prev}
126  , mASTData{coroutineASTData}
127  , mSuspendsCount{suspendsCounter}
128  , mVarNamePrefix{varNamePrefix}
129  {
130  if(nullptr == mPrevStmt) {
131  mPrevStmt = stmt;
132  }
133 
134  Visit(stmt);
135  }

137  void Visit(Stmt* stmt)
138  {
139  if(stmt) {
140  StmtVisitor<CoroutineASTTransformer>::Visit(stmt);
141  }
142  }

144  void VisitCompoundStmt(CompoundStmt* stmt)
145  {
146  for(auto* child : stmt->body()) {
147  mStaged = child;
148  Visit(child);
149 
150  if(not mSkip) {
151  mBodyStmts.Add(child);
152  }
153 
154  mSkip = false;
155  }
156 
157  auto* comp = mkCompoundStmt(mBodyStmts);
158 
159  ReplaceNode(mPrevStmt, stmt, comp);
160 
161  mBodyStmts.clear();
162  }

194  void VisitForStmt(ForStmt* stmt)
195  {
196  // technically because of the init the entire for loop should be put into a dedicated scope
197  Visit(stmt->getInit());
198 
199  // Special case. A VarDecl in init will be added to the body of the function and the actual init is left
200  // untouched. Work some magic to put it in the right place.
201  if(mSkip) {
202  auto* oldInit = stmt->getInit();
203  auto* newInit = mBodyStmts.mStmts.back();
204  mBodyStmts.mStmts.pop_back();
205 
206  ReplaceNode(stmt, oldInit, newInit);
207 
208  mSkip = false;
209  }
210 
211  Visit(stmt->getCond());
212 
213  Visit(stmt->getInc());
214 
215  CoroutineASTTransformer{mASTData, mSuspendsCount, stmt->getBody(), mVarNamePrefix, stmt};
216 
217  mSkip = false;
218  }

229  void VisitDeclRefExpr(DeclRefExpr* stmt)
230  {
231  if(auto* vd = dyn_cast_or_null<VarDecl>(stmt->getDecl())) {
232  RETURN_IF(not vd->isLocalVarDeclOrParm() or not Contains(mVarNamePrefix, vd));
233 
234  auto* memberExpr = mVarNamePrefix[vd];
235 
236  ReplaceNode(mPrevStmt, stmt, memberExpr);
237  }
238  }
239 
240  void VisitDeclStmt(DeclStmt* stmt)
241  {
242  for(auto* decl : stmt->decls()) {
243  if(auto* varDecl = dyn_cast_or_null<VarDecl>(decl)) {
244  // add this point a placement-new would be appropriate for at least some cases.
245 
246  auto* field = AddField(mASTData, GetName(*varDecl), varDecl->getType());
247  auto* me = AccessMember(mASTData.mFrameAccessDeclRef, field);
248  auto* assign = Assign(me, field, varDecl->getInit());
249 
250  mVarNamePrefix.insert(std::make_pair(varDecl, me));
251 
252  Visit(varDecl->getInit());
253 
254  mSkip = true;
255  mBodyStmts.Add(assign);
256 
257  } else if(const auto* recordDecl = dyn_cast_or_null<CXXRecordDecl>(decl)) {
258  mASTData.mFrameType->addDecl(const_cast<CXXRecordDecl*>(recordDecl));
259  }
260  }
261  }

263  void VisitCXXThisExpr(CXXThisExpr* stmt)
264  {
265  auto* fieldDecl = mkFieldDecl(mASTData.mFrameType, kwInternalThis, stmt->getType());
266  auto* indirectThisMemberExpr = AccessMember(mASTData.mFrameAccessDeclRef, fieldDecl);
267 
268  ReplaceNode(mPrevStmt, stmt, indirectThisMemberExpr);
269 
270  if(0 == mASTData.mThisExprs.size()) {
271  mASTData.mThisExprs.push_back(stmt);
272  }
273  }

317  void VisitCoawaitExpr(CoawaitExpr* stmt)
318  {
319  if(not mFinalSuspend) {
320  ++mSuspendsCount;
321  }
322 
323  if(const bool returnsVoid{stmt->getResumeExpr()->getType()->isVoidType()}; returnsVoid) {
324  Visit(stmt->getOperand());
325 
326  // in the void return case there is nothing to do, because this expression (potentially) is not nested.
327  return;
328  }
329 
330  mBodyStmts.Add(stmt);
331 
332  // Note: Add the this pointer to the name isn't the best but s quick approach
333  const std::string name{StrCat(CORO_FRAME_ACCESS, BuildSuspendVarName(stmt->getOpaqueValue()), "_res"sv)};
334 
335  auto* resultVar = Variable(name, stmt->getType());
336  auto* resultVarDeclRef = mkDeclRefExpr(resultVar);
337 
338  ReplaceNode(mPrevStmt, stmt, resultVarDeclRef);
339 
340  Visit(stmt->getCommonExpr());
341  Visit(stmt->getOperand());
342  Visit(stmt->getSuspendExpr());
343  Visit(stmt->getReadyExpr());
344  Visit(stmt->getResumeExpr());
345  }

347  void VisitCoroutineBodyStmt(CoroutineBodyStmt* stmt)
348  {
349  auto* varDecl = stmt->getPromiseDecl();
350 
351  mASTData.mPromiseField = AddField(mASTData, GetName(*varDecl), varDecl->getType());
352  auto* me = AccessMember(mASTData.mFrameAccessDeclRef, mASTData.mPromiseField);
353 
354  mVarNamePrefix.insert(std::make_pair(varDecl, me));
355 
356  auto& ctx = GetGlobalAST();
357 
358  // add the suspend index variable
359  mASTData.mSuspendIndexField = AddField(mASTData, SUSPEND_INDEX_NAME, ctx.IntTy);
360  mASTData.mSuspendIndexAccess = AccessMember(mASTData.mFrameAccessDeclRef, mASTData.mSuspendIndexField);
361 
362  // https://timsong-cpp.github.io/cppwp/n4861/dcl.fct.def.coroutine#5.3
363  mASTData.mInitialAwaitResumeCalledField = AddField(mASTData, INITIAL_AWAIT_SUSPEND_CALLED_NAME, ctx.BoolTy);
364  mASTData.mInitialAwaitResumeCalledAccess =
365  AccessMember(mASTData.mFrameAccessDeclRef, mASTData.mInitialAwaitResumeCalledField);
366 
367  for(auto* param : stmt->getParamMoves()) {
368  if(auto* declStmt = dyn_cast_or_null<DeclStmt>(param)) {
369  if(auto* varDecl2 = dyn_cast_or_null<VarDecl>(declStmt->getSingleDecl())) {
370  // For the captured parameters we need to find the ParmVarDecl instead of the newly created VarDecl
371  if(auto* declRef = FindDeclRef(varDecl2->getAnyInitializer())) {
372  auto* varDecl = dyn_cast<ParmVarDecl>(declRef->getDecl());
373 
374  auto* field = AddField(mASTData, GetName(*varDecl), varDecl->getType());
375  auto* me = AccessMember(mASTData.mFrameAccessDeclRef, field);
376 
377  mVarNamePrefix.insert(std::make_pair(const_cast<ParmVarDecl*>(varDecl), me));
378  }
379  }
380  }
381  }
382 
383  Visit(stmt->getBody());
384 
385  Visit(stmt->getReturnStmt());
386  Visit(stmt->getReturnValue());
387  Visit(stmt->getReturnValueInit());
388  Visit(stmt->getExceptionHandler());
389  Visit(stmt->getReturnStmtOnAllocFailure());
390 
391  Visit(stmt->getInitSuspendStmt());
392 
393  // final suspend point doesn't need a label
394  mFinalSuspend = true;
395  Visit(stmt->getFinalSuspendStmt());
396  mFinalSuspend = false;
397  }

399  void VisitStmt(Stmt* stmt)
400  {
401  auto* tmp = mPrevStmt;
402  mPrevStmt = stmt;
403 
404  for(auto* child : stmt->children()) {
405  Visit(child);
406  }
407 
408  mPrevStmt = tmp;
409  }

413 void CoroutinesCodeGenerator::InsertCoroutine(const FunctionDecl& fd, const CoroutineBodyStmt* stmt)
414 {
415  mOutputFormatHelper.OpenScope();
416 
417  auto& ctx = GetGlobalAST();
418 
419  mFSMName = [&] {
420  OutputFormatHelper ofm{};
421  CodeGeneratorVariant codeGenerator{ofm};
422 
423  // Coroutines can be templates and then we end up with the same FSM name but different template parameters.
424  // XXX: This will fail with NTTP's like 3.14
425  if(const auto* args = fd.getTemplateSpecializationArgs()) {
426  ofm.Append('_');
427 
428  for(OnceFalse needsUnderscore{}; const auto& arg : args->asArray()) {
429  if(needsUnderscore) {
430  ofm.Append('_');
431  }
432 
433  codeGenerator->InsertTemplateArg(arg);
434  }
435  }
436 
437  auto& str = ofm.GetString();
438  ReplaceAll(str, "<"sv, ""sv);
439  ReplaceAll(str, ":"sv, ""sv);
440  ReplaceAll(str, ">"sv, ""sv);
441  ReplaceAll(str, ","sv, ""sv);
442  ReplaceAll(str, " "sv, ""sv);
443 
444  if(fd.isOverloadedOperator()) {
445  return StrCat(MakeLineColumnName(ctx.getSourceManager(), stmt->getBeginLoc(), "operator_"sv), str);
446  } else {
447  return StrCat(GetName(fd), str);
448  }
449  }();
450 
451  mFrameName = BuildInternalVarName(StrCat(mFSMName, "Frame"sv));
452 
453  // Insert a made up struct which holds the "captured" parameters stored in the coroutine frame
454  mASTData.mFrameType = Struct(mFrameName);
455  mASTData.mFrameAccessDeclRef = mkVarDeclRefExpr(CORO_FRAME_NAME, GetFrameType());
456 
457  // The coroutine frame starts with two function pointers to the resume and destroy function. See:
458  // https://gcc.gnu.org/legacy-ml/gcc-patches/2020-01/msg01096.html:
459  // "The ABI mandates that pointers into the coroutine frame point to an area
460  // begining with two function pointers (to the resume and destroy functions
461  // described below); these are immediately followed by the "promise object"
462  // described in the standard."
463  //
464  // and
465  // https://llvm.org/docs/Coroutines.html#id72 "Coroutine Representation"
466  auto* resumeFnFd = Function(hlpResumeFn, VoidTy(), {{CORO_FRAME_NAME, GetFramePointerType()}});
467  auto resumeFnType = Ptr(resumeFnFd->getType());
468  mASTData.mResumeFnField = AddField(hlpResumeFn, resumeFnType);
469 
470  auto* destroyFnFd = Function(hlpDestroyFn, VoidTy(), {{CORO_FRAME_NAME, GetFramePointerType()}});
471  auto destroyFnType = Ptr(destroyFnFd->getType());
472  mASTData.mDestroyFnField = AddField(hlpDestroyFn, destroyFnType);
473 
474  // Allocated the made up frame
475  mOutputFormatHelper.AppendCommentNewLine("Allocate the frame including the promise"sv);
476  mOutputFormatHelper.AppendCommentNewLine("Note: The actual parameter new is __builtin_coro_size"sv);
477 
478  auto* coroFrameVar = Variable(CORO_FRAME_NAME, GetFramePointerType());
479 
480  CXXReinterpretCastExpr* reicast =
481  CXXReinterpretCastExpr::Create(ctx,
482  GetFramePointerType(),
483  VK_LValue,
484  CK_BitCast,
485  stmt->getAllocate(),
486  nullptr,
487  ctx.getTrivialTypeSourceInfo(GetFramePointerType()),
488  {},
489  {},
490  {});
491 
492  coroFrameVar->setInit(reicast);
493 
494  InsertArg(coroFrameVar);
495 
496  // P0057R8: [dcl.fct.def.coroutine] p8: get_return_object_on_allocation_failure indicates that new may return a
497  // nullptr. In this case return get_return_object_on_allocation_failure.
498  if(stmt->getReturnStmtOnAllocFailure()) {
499  auto* nptr = new(ctx) CXXNullPtrLiteralExpr({});
500 
501  // Create an IfStmt.
502  StmtsContainer bodyStmts{stmt->getReturnStmtOnAllocFailure()};
503  auto* ifStmt = If(Equal(nptr, mASTData.mFrameAccessDeclRef), bodyStmts);
504 
505  mOutputFormatHelper.AppendNewLine();
506  InsertArg(ifStmt);
507  }
508 
509  CoroutineASTTransformer{
510  mASTData, mSuspendsCounter, const_cast<CoroutineBodyStmt*>(stmt), llvm::DenseMap<VarDecl*, MemberExpr*>{}};
511 
512  // set initial suspend count to zero.
513  auto* setSuspendIndexToZero = Assign(mASTData.mFrameAccessDeclRef, mASTData.mSuspendIndexField, Int32(0));
514  InsertArgWithNull(setSuspendIndexToZero);
515 
516  // https://timsong-cpp.github.io/cppwp/n4861/dcl.fct.def.coroutine#5.3
517  auto* initializeInitialAwaitResume =
518  Assign(mASTData.mFrameAccessDeclRef, mASTData.mInitialAwaitResumeCalledField, Bool(false));
519  InsertArgWithNull(initializeInitialAwaitResume);
520 
521  // Move the parameters first
522  for(auto* param : stmt->getParamMoves()) {
523  if(const auto* declStmt = dyn_cast_or_null<DeclStmt>(param)) {
524  if(const auto* varDecl = dyn_cast_or_null<VarDecl>(declStmt->getSingleDecl())) {
525  const auto varName = GetName(*varDecl);
526 
527  mOutputFormatHelper.AppendNewLine(CORO_FRAME_ACCESS,
528  varName,
529  " = "sv,
530  "std::forward<"sv,
531  GetName(varDecl->getType()),
532  ">("sv,
533  varName,
534  ");"sv);
535  }
536  }
537  }
538 
539  // According to https://eel.is/c++draft/dcl.fct.def.coroutine#5.7 the promise_type constructor can have
540  // parameters. If so, they must be equal to the coroutines function parameters.
541  // The code here performs a _simple_ lookup for a matching ctor without using Clang's overload resolution.
542  ArrayRef<ParmVarDecl*> funParams = fd.parameters();
543  SmallVector<ParmVarDecl*, 16> funParamStorage{};
544  QualType cxxMethodType{};
545 
546  if(const auto* cxxMethodDecl = dyn_cast_or_null<CXXMethodDecl>(&fd)) {
547  funParamStorage.reserve(funParams.size() + 1);
548 
549  cxxMethodType = cxxMethodDecl->
550 #if IS_CLANG_NEWER_THAN(17)
551  getFunctionObjectParameterType()
552 #else
553  getThisObjectType()
554 #endif
555  ;
556 
557  // In case we have a member function the first parameter is a reference to this. The following code injects
558  // this parameter.
559  funParamStorage.push_back(Parameter(&fd, CORO_FRAME_ACCESS_THIS, cxxMethodType));
560 
561  ranges::copy(funParams, std::back_inserter(funParamStorage));
562 
563  funParams = funParamStorage;
564  }
565 
566  auto getNonRefType = [&](auto* var) -> QualType {
567  if(const auto* et = var->getType().getNonReferenceType()->template getAs<ElaboratedType>()) {
568  return et->getNamedType();
569  } else {
570  return QualType(var->getType().getNonReferenceType().getTypePtrOrNull(), 0);
571  }
572  };
573 
574  SmallVector<Expr*, 16> exprs{};
575 
576  for(auto* promiseTypeRecordDecl = mASTData.mPromiseField->getType()->getAsCXXRecordDecl();
577  auto* ctor : promiseTypeRecordDecl->ctors()) {
578 
579  if(not ranges::equal(
580  ctor->parameters(), funParams, [&](auto& a, auto& b) { return getNonRefType(a) == getNonRefType(b); })) {
581  continue;
582  }
583 
584  // In case of a promise ctor which takes this as the first argument, that parameter needs to be deferences,
585  // as it can only be taken as a reference.
586  OnceTrue derefFirstParam{};
587 
588  if(not ctor->param_empty() and
589  (getNonRefType(ctor->getParamDecl(0)) == QualType(cxxMethodType.getTypePtrOrNull(), 0))) {
590  if(0 == mASTData.mThisExprs.size()) {
591  mASTData.mThisExprs.push_back(
592 #if IS_CLANG_NEWER_THAN(17)
593  CXXThisExpr::Create(ctx, {}, Ptr(cxxMethodType), false)
594 #else
595  new(ctx) CXXThisExpr{{}, Ptr(cxxMethodType), false}
596 #endif
597 
598  );
599  }
600  } else {
601  (void)static_cast<bool>(derefFirstParam); // set it to false
602  }
603 
604  for(const auto& fparam : funParams) {
605  if(derefFirstParam) {
606  exprs.push_back(Dref(mkDeclRefExpr(fparam)));
607 
608  } else {
609  exprs.push_back(AccessMember(mASTData.mFrameAccessDeclRef, fparam));
610  }
611  }
612 
613  if(funParams.size()) {
614  // The <new> header needs to be included.
615  EnableGlobalInsert(GlobalInserts::HeaderNew);
616  }
617 
618  break; // We've found what we were looking for
619  }
620 
621  if(mASTData.mThisExprs.size()) {
622  mOutputFormatHelper.AppendNewLine(CORO_FRAME_ACCESS_THIS, " = this;"sv);
623  }
624 
625  // Now call the promise ctor, as it may access some of the parameters it comes at this point.
626  mOutputFormatHelper.AppendNewLine();
627  mOutputFormatHelper.AppendCommentNewLine("Construct the promise."sv);
628  auto* me = AccessMember(mASTData.mFrameAccessDeclRef, mASTData.mPromiseField);
629 
630  auto* ctorArgs = new(ctx) InitListExpr{ctx, {}, exprs, {}};
631 
632  CXXNewExpr* newFrame = New({AddrOf(me)}, ctorArgs, mASTData.mPromiseField->getType());
633 
634  InsertArgWithNull(newFrame);
635 
636  // Add parameters from the original function to the list
637 
638  // P0057R8: [dcl.fct.def.coroutine] p5: before initial_suspend and at tops 1
639 
640  // Make a call to the made up state machine function for the initial suspend
641  mOutputFormatHelper.AppendNewLine();
642 
643  // [dcl.fct.def.coroutine]
644  mOutputFormatHelper.AppendCommentNewLine("Forward declare the resume and destroy function."sv);
645 
646  auto* fsmFuncDecl = CreateCoroFunctionDecl(StrCat(mFSMName, "Resume"sv), GetFramePointerType());
647  InsertArg(fsmFuncDecl);
648  auto* deallocFuncDecl = CreateCoroFunctionDecl(StrCat(mFSMName, "Destroy"sv), GetFramePointerType());

733 void CoroutinesCodeGenerator::InsertArg(const CoroutineBodyStmt* stmt)
734 {
735  // insert a made up switch for continuing a resume
736  SwitchStmt* sstmt = Switch(mASTData.mSuspendIndexAccess);
737 
738  // insert 0 with break for consistency
739  auto* initialSuspendCase = Case(0, Break());
740  StmtsContainer switchBodyStmts{initialSuspendCase};
741 
742  for(const auto& i : NumberIterator{mSuspendsCounter}) {
743  switchBodyStmts.Add(Case(i + 1, Goto(BuildResumeLabelName(i + 1))));
744  }
745 
746  auto* switchBody = mkCompoundStmt(switchBodyStmts);
747  sstmt->setBody(switchBody);
748 
749  StmtsContainer funcBodyStmts{
750  Comment("Create a switch to get to the correct resume point"sv), sstmt, stmt->getInitSuspendStmt()};
751 
752  // insert the init suspend expr
753  mState = eState::InitialSuspend;
754 
755  if(mASTData.mThisExprs.size()) {
756  AddField(kwInternalThis, mASTData.mThisExprs.at(0)->getType());
757  }
758 
759  mInsertVarDecl = false;
760  mSupressRecordDecls = true;
761 
762  for(const auto* c : stmt->getBody()->children()) {
763  funcBodyStmts.Add(c);
764  }
765 
766  // InsertArg(stmt->getFallthroughHandler());
767 
768  auto* gotoFinalSuspend = Goto(FINAL_SUSPEND_NAME);
769  funcBodyStmts.Add(gotoFinalSuspend);
770 
771  auto* body = [&]() -> Stmt* {
772  auto* tryBody = mkCompoundStmt(funcBodyStmts);
773 
774  // First open the try-catch block, as we get an error when jumping across such blocks with goto
775  if(const auto* exceptionHandler = stmt->getExceptionHandler()) {
776  // If we encounter an exceptionbefore inital_suspend's await_suspend was called we re-throw the
777  // exception.
778  auto* ifStmt = If(Not(mASTData.mInitialAwaitResumeCalledAccess), Throw());
779 
780  StmtsContainer catchBodyStmts{ifStmt, exceptionHandler};
781 
782  return Try(tryBody, Catch(catchBodyStmts));
783  }
784 
785  return tryBody;
786  }();
787 
788  InsertArg(body);
789 
790  mOutputFormatHelper.AppendNewLine();
791 
792  auto* finalSuspendLabel = Label(FINAL_SUSPEND_NAME);
793  InsertArg(finalSuspendLabel);
794  mState = eState::FinalSuspend;
795  InsertArg(stmt->getFinalSuspendStmt());
796 
797  // disable prefixing names and types
798  mInsertVarDecl = true;
799 }

902 void CoroutinesCodeGenerator::InsertArg(const CoroutineSuspendExpr* stmt)
903 {
904  mOutputFormatHelper.AppendNewLine();
905  InsertInstantiationPoint(GetGlobalAST().getSourceManager(), stmt->getKeywordLoc(), [&] {
906  if(isa<CoawaitExpr>(stmt)) {
907  return kwCoAwaitSpace;
908  } else {
909  return kwCoYieldSpace;
910  }
911  }());
912 
913  mPosBeforeSuspendExpr = mOutputFormatHelper.CurrentPos();
914 
915  /// Represents an expression that might suspend coroutine execution;
916  /// either a co_await or co_yield expression.
917  ///
918  /// Evaluation of this expression first evaluates its 'ready' expression. If
919  /// that returns 'false':
920  /// -- execution of the coroutine is suspended
921  /// -- the 'suspend' expression is evaluated
922  /// -- if the 'suspend' expression returns 'false', the coroutine is
923  /// resumed
924  /// -- otherwise, control passes back to the resumer.
925  /// If the coroutine is not suspended, or when it is resumed, the 'resume'
926  /// expression is evaluated, and its result is the result of the overall
927  /// expression.
928 
929  // mOutputFormatHelper.AppendNewLine("// __builtin_coro_save() // frame->suspend_index = n");
930 
931  // For why, see the implementation of CoroutinesCodeGenerator::InsertArg(const ImplicitCastExpr* stmt)
932  mSupressCasts = true;
933 
934  auto* il = Int32(++mSuspendsCount);
935  auto* bop = Assign(mASTData.mSuspendIndexAccess, mASTData.mSuspendIndexField, il);
936 
937  // Find out whether the return type is void or bool. In case of bool, we need to insert an if-statement, to
938  // suspend only, if the return value was true.
939  // Technically only void, bool, or std::coroutine_handle<Z> is allowed. [expr.await] p3.7
940  const bool returnsVoid{stmt->getSuspendExpr()->getType()->isVoidType()};
941 
942  // XXX: check if getResumeExpr is marked noexcept. Otherwise we need additional expcetion handling?
943  // CGCoroutine.cpp:229
944 
945  StmtsContainer bodyStmts{};
946  Expr* initializeInitialAwaitResume = nullptr;
947 
948  auto addInitialAwaitSuspendCalled = [&] {
949  if(eState::FinalSuspend != mState) {
950  bodyStmts.Add(bop);
951 
952  if(eState::InitialSuspend == mState) {
953  mState = eState::Body;
954  // https://timsong-cpp.github.io/cppwp/n4861/dcl.fct.def.coroutine#5.3
955  initializeInitialAwaitResume =
956  Assign(mASTData.mFrameAccessDeclRef, mASTData.mInitialAwaitResumeCalledField, Bool(true));
957  bodyStmts.Add(initializeInitialAwaitResume);
958  }
959  }
960  };
961 
962  if(returnsVoid) {
963  bodyStmts.Add(stmt->getSuspendExpr());
964  addInitialAwaitSuspendCalled();
965 
966  if(eState::FinalSuspend != mState) {
967  bodyStmts.Add(Return());
968  }
969 
970  InsertArg(If(Not(stmt->getReadyExpr()), bodyStmts));
971 
972  } else {
973  addInitialAwaitSuspendCalled();
974 
975  if(eState::FinalSuspend != mState) {
976  bodyStmts.Add(Return());
977  }
978 
979  auto* ifSuspend = If(stmt->getSuspendExpr(), bodyStmts);
980 
981  InsertArg(If(Not(stmt->getReadyExpr()), ifSuspend));
982  }
983 
984  if(not returnsVoid and initializeInitialAwaitResume) {
985  // At this point we technically haven't called initial suspend
986  InsertArgWithNull(initializeInitialAwaitResume);
987  mOutputFormatHelper.AppendNewLine();
988  }
989 
990  RETURN_IF(eState::FinalSuspend == mState);
991 
992  auto* suspendLabel = Label(BuildResumeLabelName(mSuspendsCount));
993  InsertArg(suspendLabel);
994 
995  const auto* resumeExpr = stmt->getResumeExpr();
996 
997  if(not resumeExpr->getType()->isVoidType()) {
998  const auto* sourceExpr = stmt->getOpaqueValue()->getSourceExpr();
999 
1000  if(const auto& s = FindValue(mOpaqueValues, sourceExpr)) {
1001  const auto fieldName{StrCat(std::string_view{s.value()}.substr(CORO_FRAME_ACCESS.size()), "_res"sv)};
1002  mOutputFormatHelper.Append(CORO_FRAME_ACCESS, fieldName, hlpAssing);
1003 
1004  AddField(fieldName, resumeExpr->getType());
1005  }
1006  }
1007 
1008  InsertArg(resumeExpr);
1009 }
