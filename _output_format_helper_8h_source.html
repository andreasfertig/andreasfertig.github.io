150  /// \brief Append a argument list to the buffer.
151  ///
152  /// This function takes care of the delimiting ',' between the parameters. The lambda \c lambda is called to each
153  /// argument after the comma was inserted.
154  /// Usage:
155  /// \code
156  /// ForEachArg(parameters, [&](const auto& p) {
157  /// // do something with p
158  /// });
159  /// \endcode
160  template<typename T, typename Lambda>
161  static inline void ForEachArg(const T& arguments, OutputFormatHelper& outputFormatHelper, Lambda&& lambda)
162  {
163  OnceFalse needsComma{};
164  for(const auto& arg : arguments) {
165  if constexpr(is_same_v<const TemplateArgument&, decltype(arg)>) {
166  if((TemplateArgument::Pack == arg.getKind()) && (0 == arg.pack_size())) {
167  break;
168  }
169  }
170 
171  if(needsComma) {
172  outputFormatHelper.Append(", ");
173  }
174 
175  lambda(arg);
176  }
177  }
178 
179  void InsertIfDefTemplateGuard() { AppendNewLine("#ifdef INSIGHTS_USE_TEMPLATE"); }
180  void InsertEndIfTemplateGuard() { AppendNewLine("#endif"); }
