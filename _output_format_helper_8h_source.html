166  /// \brief Append a argument list to the buffer.
167  ///
168  /// This function takes care of the delimiting ',' between the parameters. The lambda \c lambda is called to each
169  /// argument after the comma was inserted.
170  /// Usage:
171  /// \code
172  /// ForEachArg(parameters, [&](const auto& p) {
173  /// // do something with p
174  /// });
175  /// \endcode
176  inline void ForEachArg(const auto& arguments, /*XXX: invocable*/ auto&& lambda)
177  {
178  OnceFalse needsComma{};
179  for(const auto& arg : arguments) {
180  if constexpr(std::is_same_v<const TemplateArgument&, decltype(arg)>) {
181  if((TemplateArgument::Pack == arg.getKind()) && (0 == arg.pack_size())) {
182  break;
183  }
184  }
185 
186  AppendComma(needsComma);
187 
188  lambda(arg);
189  }
190  }
191

193  void InsertEndIfTemplateGuard() { AppendNewLine("#endif"sv); }
