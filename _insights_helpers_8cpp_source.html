57 std::string ScopeHandler::RemoveCurrentScope(std::string name)
58 {
59  if(mScope.length()) {
60  auto findAndReplace = [&name](const std::string& scope) {
61  if(const auto startPos = name.find(scope, 0); std::string::npos != startPos) {
62  if(const auto pos = startPos + scope.length();
63  (pos > name.length()) || (name[pos] != '*')) { // keep member points (See #374)
64  name.replace(startPos, scope.length(), ""sv);
65  return true;
66  }
67  }
68 
69  return false;
70  };
71 
72  // The default is that we can replace the entire scope. Suppose we are currently in N::X and having a symbol
73  // N::X::y then N::X:: is removed.
74  if(not findAndReplace(mScope)) {
75 
76  // A special case where we need to remove the scope without the last item.
77  std::string tmp{mScope};
78  tmp.resize(mGlobalStack.back().mLength);
79 
80  findAndReplace(tmp);
81  }
82  }
83 
84  return name;
85 }

126 static void BuildNamespace(std::string& fullNamespace, const NestedNameSpecifier* stmt)
127 {
128  if(!stmt) {
129  return;
130  }
131 
132  if(const auto* prefix = stmt->getPrefix()) {
133  BuildNamespace(fullNamespace, prefix);
134  }
135 
136  switch(stmt->getKind()) {
137  case NestedNameSpecifier::Identifier: fullNamespace.append(stmt->getAsIdentifier()->getName()); break;
138 
139  case NestedNameSpecifier::Namespace:
140  if(stmt->getAsNamespace()->isAnonymousNamespace()) {
141  return;
142  }
143 
144  fullNamespace.append(stmt->getAsNamespace()->getName());
145  break;
146 
147  case NestedNameSpecifier::NamespaceAlias: fullNamespace.append(stmt->getAsNamespaceAlias()->getName()); break;
148 
149  case NestedNameSpecifier::TypeSpecWithTemplate: fullNamespace.append(kwTemplateSpace); [[fallthrough]];
150 
151  case NestedNameSpecifier::TypeSpec:
152  fullNamespace.append(GetUnqualifiedScopelessName(stmt->getAsType(), InsightsSuppressScope::Yes));
153  // The template parameters are already contained in the type we inserted above.
154  break;
155 
156  default: break;
157  }
158 
159  fullNamespace.append("::"sv);
160 }

203 SourceLocation FindLocationAfterSemi(const SourceLocation loc,
204  const ast_matchers::MatchFinder::MatchResult& result,
205  RequireSemi requireSemi)
206 {
207  auto findLocation = [&](const tok::TokenKind tKind) {
208  return clang::Lexer::findLocationAfterToken(loc, tKind, GetSM(result), GetLangOpts(result), false);
209  };
210 
211  if(const auto locEnd{findLocation(tok::semi)}; locEnd.isValid()) {
212  return locEnd;
213 
214  } else if(RequireSemi::Yes == requireSemi) {
215  // if we do not find a ; then it can possibly be a brace init like this:
216  // auto x {23};
217  // Try to find the right curly which seems to also contain the semi.
218  if(const auto locEnd2{findLocation(tok::r_brace)}; locEnd2.isValid()) {
219  return locEnd2;
220  }
221  // if we do not find a ; then it can possibly be a paren init like this:
222  // int x(23);
223  // Try to find the right paren which seems to also contain the semi.
224  else if(const auto locEnd3{findLocation(tok::r_paren)}; locEnd3.isValid()) {
225  return locEnd3;
226  }
227  }
228 
229  return loc;
230 }

368 // own implementation due to lambdas
369 std::string GetDeclContext(const DeclContext* ctx, WithTemplateParameters withTemplateParameters)
370 {
371  OutputFormatHelper mOutputFormatHelper{};
372  SmallVector<const DeclContext*, 8> contexts{};
373 
374  while(ctx) {
375  if(isa<NamedDecl>(ctx)) {
376  contexts.push_back(ctx);
377  }
378  ctx = ctx->getParent();
379  }
380 
381  for(const auto* declContext : llvm::reverse(contexts)) {
382  if(const auto* classTmplSpec = dyn_cast<ClassTemplateSpecializationDecl>(declContext)) {
383  mOutputFormatHelper.Append(classTmplSpec->getName());
384 
385  CodeGenerator codeGenerator{mOutputFormatHelper};
386  codeGenerator.InsertTemplateArgs(*classTmplSpec);
387 
388  } else if(const auto* nd = dyn_cast<NamespaceDecl>(declContext)) {
389  if(nd->isAnonymousNamespace() || nd->isInline()) {
390  continue;
391  }
392 
393  mOutputFormatHelper.Append(nd->getName());
394 
395  } else if(const auto* rd = dyn_cast<RecordDecl>(declContext)) {
396  if(!rd->getIdentifier()) {
397  continue;
398  }
399 
400  mOutputFormatHelper.Append(rd->getName());
401 
402  // A special case at least for out-of-line static member variables of a class template. They need to carry
403  // the template parameters of the class template.
404  if(WithTemplateParameters::Yes == withTemplateParameters /*declContext->isNamespace() || declContext->getLexicalParent()->isNamespace() || declContext->getLexicalParent()->isTranslationUnit()*/) {
405  if(const auto* cxxRecordDecl = dyn_cast_or_null<CXXRecordDecl>(rd)) {
406  if(const auto* classTmpl = cxxRecordDecl->getDescribedClassTemplate()) {
407  const auto* tmplParams = classTmpl->getTemplateParameters();
408 
409  CodeGenerator codeGenerator{mOutputFormatHelper};
410  mOutputFormatHelper.Append('<');
411 
412  mOutputFormatHelper.ForEachArg(*tmplParams, [&](const auto* param) {
413  mOutputFormatHelper.Append(GetName(*param));
414  if(param->isParameterPack()) {
415  mOutputFormatHelper.Append("..."sv);
416  }
417  });
418 
419  mOutputFormatHelper.Append('>');
420  }
421  }
422  }
423 
424  } else if(dyn_cast<FunctionDecl>(declContext)) {
425  continue;
426 
427  } else if(const auto* ed = dyn_cast<EnumDecl>(declContext)) {
428  if(!ed->isScoped()) {
429  continue;
430  }
431 
432  mOutputFormatHelper.Append(ed->getName());
433 
434  } else {
435  mOutputFormatHelper.Append(cast<NamedDecl>(declContext)->getName());
436  }
437 
438  mOutputFormatHelper.Append("::"sv);
439  }
440 
441  return mOutputFormatHelper.GetString();
442 }

1006 std::string GetTypeNameAsParameter(const QualType& t, std::string_view varName, const Unqualified unqualified)
1007 {
1008  const bool isFunctionPointer = HasTypeWithSubType<ReferenceType, FunctionProtoType>(t);
1009  const bool isArrayRef = HasTypeWithSubType<ReferenceType, ArrayType>(t);
1010  // Special case for Issue81, auto returns an array-ref and to catch auto deducing an array (Issue106)
1011  const bool isAutoType = (nullptr != dyn_cast_or_null<AutoType>(t.getTypePtrOrNull()));
1012  const auto pointerToArrayBaseType = isAutoType ? t->getContainedAutoType()->getDeducedType() : t;
1013  const bool isPointerToArray = HasTypeWithSubType<PointerType, ArrayType>(pointerToArrayBaseType);
1014 
1015  std::string typeName = details::GetName(t, unqualified);
1016 
1017  // Sometimes we get char const[2]. If we directly insert the typename we end up with char const__var[2] which is not
1018  // a valid type name. Hence check for this condition and, if necessary, insert a space before __var.
1019  auto getSpaceOrEmpty = [&](const std::string_view& needle) -> std::string_view {
1020  if(not Contains(typeName, needle)) {
1021  return " ";
1022  }
1023 
1024  return {};
1025  };
1026 
1027  if(t->isArrayType() && !t->isLValueReferenceType()) {
1028  const auto space = getSpaceOrEmpty(" ["sv);
1029  InsertBefore(typeName, "["sv, StrCat(space, varName));
1030 
1031  } else if(isArrayRef) {
1032  const bool isRValueRef{HasTypeWithSubType<RValueReferenceType, ArrayType>(t)};
1033  const std::string_view contains{isRValueRef ? "(&&" : "(&"};
1034 
1035  if(Contains(typeName, contains)) {
1036  InsertAfter(typeName, contains, varName);
1037  } else {
1038  const std::string_view insertBefore{isRValueRef ? "&&[" : "&["};
1039 
1040  InsertBefore(typeName, insertBefore, "("sv);
1041 
1042  // check whether we are dealing with a function or an array
1043  if(Contains(typeName, contains)) {
1044  InsertAfter(typeName, contains, StrCat(varName, ")"sv));
1045  } else {
1046  InsertAfter(typeName, typeName, StrCat(" "sv, varName));
1047  }
1048  }
1049 
1050  } else if(isFunctionPointer) {
1051  const bool isRValueRef{HasTypeWithSubType<RValueReferenceType, FunctionProtoType>(t)};
1052  const std::string_view contains{isRValueRef ? "(&&" : "(&"};
1053 
1054  if(Contains(typeName, contains)) {
1055  InsertAfter(typeName, contains, varName);
1056  } else {
1057  typeName += StrCat(" "sv, varName);
1058  }
1059 
1060  } else if(isa<MemberPointerType>(t)) {
1061  InsertAfter(typeName, "::*"sv, varName);
1062 
1063  } else if(isPointerToArray) {
1064  if(Contains(typeName, "(*"sv)) {
1065  InsertAfter(typeName, "(*"sv, varName);
1066  } else if(Contains(typeName, "*"sv)) {
1067  InsertBefore(typeName, "*"sv, "("sv);
1068  InsertAfter(typeName, "*"sv, StrCat(varName, ")"sv));
1069  }
1070  } else if(t->isFunctionPointerType()) {
1071  if(Contains(typeName, "(*"sv)) {
1072  InsertAfter(typeName, "(*"sv, varName);
1073  } else {
1074  typeName += StrCat(" "sv, varName);
1075  }
1076 
1077  } else if(!t->isArrayType() && !varName.empty()) {
1078  typeName += StrCat(" "sv, varName);
1079  }
1080 
1081  return typeName;
1082 }

1168  * \code
