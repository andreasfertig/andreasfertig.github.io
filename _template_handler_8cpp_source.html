43 TemplateHandler::TemplateHandler(Rewriter& rewrite, MatchFinder& matcher)
44 : InsightsBase(rewrite)
45 {
46  matcher.addMatcher(
47  functionDecl(allOf(unless(isExpansionInSystemHeader()),
48  unless(isMacroOrInvalidLocation()),
49  unless(hasAncestor(namespaceDecl())),
50  hasParent(functionTemplateDecl(unless(hasParent(classTemplateSpecializationDecl())),
51  unless(hasAncestor(cxxRecordDecl())))),
52  isTemplateInstantiationPlain()))
53  .bind("func"),
54  this);
55 
56  // match typical use where a class template is defined and it is used later.
57  matcher.addMatcher(
58  classTemplateSpecializationDecl(
59  unless(anyOf(isExpansionInSystemHeader(), hasAncestor(namespaceDecl()), hasAncestor(cxxRecordDecl()))),
60  hasParent(classTemplateDecl().bind("decl")))
61  .bind("class"),
62  this);
63 
64  // special case, where a class template is defined and somewhere else we request an explicit instantiation
65  matcher.addMatcher(classTemplateSpecializationDecl(unless(anyOf(isExpansionInSystemHeader(),
66  hasAncestor(namespaceDecl()),
67  hasParent(classTemplateDecl()),
68  isExplicitTemplateSpecialization())))
69  .bind("class"),
70  this);
71 
72  matcher.addMatcher(
73  varTemplateDecl(
74  unless(anyOf(isExpansionInSystemHeader(), hasAncestor(namespaceDecl()), hasParent(classTemplateDecl()))))
75  .bind("vd"),
76  this);
77 }

80 // Special treatment for function templates which end with a semi-colon but have a valid body. In theory getting the end
81 // of the location should be equivalent to what is done in FunctionDeclHandler.cpp. However, this approach fails for
82 // example for AutoHandler2Test.cpp.
83 static SourceLocation FindLocationAfterRBrace(const SourceLocation loc,
84  const ast_matchers::MatchFinder::MatchResult& result,
85  const bool mustHaveSemi)
86 {
87  if(not mustHaveSemi) {
88  auto nToken = clang::Lexer::findNextToken(loc, GetSM(result), result.Context->getLangOpts());
89 
90  if(nToken.hasValue()) {
91  if(auto& rToken = nToken.getValue(); rToken.getKind() == tok::semi) {
92  return rToken.getLocation();
93  }
94  }
95  }
96 
97  return FindLocationAfterSemi(loc, result).getLocWithOffset(1);
98 }

101 void TemplateHandler::run(const MatchFinder::MatchResult& result)
102 {
103  if(const auto* functionDecl = result.Nodes.getNodeAs<FunctionDecl>("func")) {
104  if(not functionDecl->getBody() && not isa<CXXDeductionGuideDecl>(functionDecl)) {
105  return;
106  }
107 
108  // Figure out whether we are looking at a CXXDeductionGuideDecl. If so, ensure that the deduction guide comes
109  // after the primary template declaration.
110  const Decl* endLocDecl = functionDecl;
111  if(const auto* deductionGuide = dyn_cast_or_null<CXXDeductionGuideDecl>(functionDecl)) {
112  const auto* deducedTemplate = deductionGuide->getDeducedTemplate();
113 
114  // deduction guide must follow after the template declaration
115  if(deducedTemplate->getEndLoc() > deductionGuide->getEndLoc()) {
116  endLocDecl = deducedTemplate;
117  }
118  }
119 
120  OutputFormatHelper outputFormatHelper = InsertInstantiatedTemplate(functionDecl);
121  const auto endOfCond = FindLocationAfterRBrace(
122  GetEndLoc(endLocDecl),
123  result,
124  isa<CXXDeductionGuideDecl>(
125  functionDecl)); // if we lift the "not getBody()" restriction above we need to take this in account
126  // here, as then we require a semi at the end.
127 
128  InsertIndentedText(endOfCond, outputFormatHelper);
129 
130  } else if(const auto* clsTmplSpecDecl = result.Nodes.getNodeAs<ClassTemplateSpecializationDecl>("class")) {
131  // skip classes/struct's without a definition
132  if(not clsTmplSpecDecl->hasDefinition()) {
133  return;
134  }
135 
136  OutputFormatHelper outputFormatHelper = InsertInstantiatedTemplate(clsTmplSpecDecl);
137 
138  if(const auto* clsTmplDecl = result.Nodes.getNodeAs<ClassTemplateDecl>("decl")) {
139  const auto endOfCond = FindLocationAfterSemi(GetEndLoc(clsTmplDecl), result);
140  InsertIndentedText(endOfCond, outputFormatHelper);
141 
142  } else { // explicit specialization, we have to remove the specialization
143  mRewrite.ReplaceText(clsTmplSpecDecl->getSourceRange(), outputFormatHelper.GetString());
144  }
145 
146  } else if(const auto* vd = result.Nodes.getNodeAs<VarTemplateDecl>("vd")) {
147  OutputFormatHelper outputFormatHelper = InsertInstantiatedTemplate(vd);
148  const auto endOfCond = FindLocationAfterSemi(GetEndLoc(vd), result);
149 
150  mRewrite.ReplaceText({vd->getSourceRange().getBegin(), endOfCond.getLocWithOffset(1)},
151  outputFormatHelper.GetString());
152  }
153 }
