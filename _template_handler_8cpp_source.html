29 /// \brief Find a \c DeclRefExpr belonging to a \c DecompisitionDecl
30 class CoroutineBodyFinder : public ConstStmtVisitor<CoroutineBodyFinder>
31 {
32  bool mIsCoroutine{};
33 
34 public:
35  CoroutineBodyFinder() = default;
36 
37  void VisitCoroutineBodyStmt(const CoroutineBodyStmt*) { mIsCoroutine = true; }
38 
39  void VisitStmt(const Stmt* stmt)
40  {
41  if(isa<CoroutineBodyStmt>(stmt)) {
42  mIsCoroutine = true;
43  return;
44  }
45 
46  for(const auto* child : stmt->children()) {
47  if(child) {
48  Visit(child);
49  }
50 
51  RETURN_IF(mIsCoroutine);
52  }
53  }
54 
55  bool Find(const Stmt* stmt)
56  {
57  if(stmt) {
58  VisitStmt(stmt);
59  }
60 
61  return mIsCoroutine;
62  }
63 };

93 TemplateHandler::TemplateHandler(Rewriter& rewrite, MatchFinder& matcher)
94 : InsightsBase(rewrite)
95 {
96  matcher.addMatcher(functionDecl(allOf(unless(isExpansionInSystemHeader()),
97  unless(isMacroOrInvalidLocation()),
98  hasParent(functionTemplateDecl(hasParent(translationUnitDecl()))),
99  isTemplateInstantiationPlain()))
100  .bind(idFunc),
101  this);
102 
103  matcher.addMatcher(functionTemplateDecl(hasThisTUParent).bind(idFunc2), this);
104 
105  // match typical use where a class template is defined and it is used later.
106  matcher.addMatcher(classTemplateSpecializationDecl(
107  unless(isExpansionInSystemHeader()),
108  allOf(hasParent(classTemplateDecl(hasParent(translationUnitDecl())).bind(idDecl)),
109  isExplicitTemplateSpecialization()))
110  .bind(idClass),
111  this);
112 
113  matcher.addMatcher(classTemplateDecl(hasThisTUParent).bind(idDecl), this);
114 
115  matcher.addMatcher(conceptDecl(hasThisTUParent).bind(idDecl), this);
116 
117  // special case, where a class template is defined and somewhere else we request an explicit instantiation
118  matcher.addMatcher(
119  classTemplateSpecializationDecl(hasThisTUParent, unless(isExplicitTemplateSpecialization())).bind(idClass),
120  this);
121 
122  matcher.addMatcher(varTemplateDecl(hasThisTUParent).bind(idVar), this);
123 }

126 // Special treatment for function templates which end with a semi-colon but have a valid body. In theory getting the end
127 // of the location should be equivalent to what is done in FunctionDeclHandler.cpp. However, this approach fails for
128 // example for AutoHandler2Test.cpp.
129 static SourceLocation FindLocationAfterRBrace(const SourceLocation loc,
130  const ast_matchers::MatchFinder::MatchResult& result,
131  const bool mustHaveSemi)
132 {
133  if(not mustHaveSemi) {
134  auto nToken = clang::Lexer::findNextToken(loc, GetSM(result), result.Context->getLangOpts());
135 
136  if(nToken.hasValue()) {
137  if(auto& rToken = nToken.getValue(); rToken.getKind() == tok::semi) {
138  return rToken.getLocation();
139  }
140  }
141  }
142 
143  return FindLocationAfterSemi(loc, result).getLocWithOffset(1);
144 }

147 void TemplateHandler::run(const MatchFinder::MatchResult& result)
148 {
149  if(const auto* functionDecl = result.Nodes.getNodeAs<FunctionDecl>(idFunc)) {
150  RETURN_IF(not functionDecl->getBody() and not isa<CXXDeductionGuideDecl>(functionDecl));
151 
152  // Figure out whether we are looking at a CXXDeductionGuideDecl. If so, ensure that the deduction guide comes
153  // after the primary template declaration.
154  const Decl* endLocDecl = functionDecl;
155  if(const auto* deductionGuide = dyn_cast_or_null<CXXDeductionGuideDecl>(functionDecl)) {
156  const auto* deducedTemplate = deductionGuide->getDeducedTemplate();
157 
158  // deduction guide must follow after the template declaration
159  if(deducedTemplate->getEndLoc() > deductionGuide->getEndLoc()) {
160  endLocDecl = deducedTemplate;
161  }
162  }
163 
164  OutputFormatHelper outputFormatHelper = InsertInstantiatedTemplate(functionDecl);
165  const auto endOfCond = FindLocationAfterRBrace(
166  endLocDecl->getEndLoc(),
167  result,
168  isa<CXXDeductionGuideDecl>(
169  functionDecl)); // if we lift the "not getBody()" restriction above we need to take this in account
170  // here, as then we require a semi at the end.
171 
172  InsertIndentedText(endOfCond, outputFormatHelper);
173 
174  } else if(const auto* functionTemplateDecl = result.Nodes.getNodeAs<FunctionTemplateDecl>(idFunc2)) {
175  RETURN_IF(isa<CXXDeductionGuideDecl>(functionTemplateDecl->getTemplatedDecl()));
176 
177  // For now, don't transform the primary template of a coroutine
178  RETURN_IF(CoroutineBodyFinder{}.Find(functionTemplateDecl->getTemplatedDecl()->getBody()));
179 
180  OutputFormatHelper outputFormatHelper{};
181 
182  CodeGenerator codeGenerator{outputFormatHelper};
183  codeGenerator.InsertPrimaryTemplate(functionTemplateDecl);
184 
185  mRewrite.ReplaceText(functionTemplateDecl->getSourceRange(), outputFormatHelper);
186 
187  } else if(const auto* clsTmplSpecDecl = result.Nodes.getNodeAs<ClassTemplateSpecializationDecl>(idClass)) {
188  // skip classes/struct's without a definition
189  RETURN_IF(not clsTmplSpecDecl->hasDefinition());
190 
191  OutputFormatHelper outputFormatHelper = InsertInstantiatedTemplate(clsTmplSpecDecl);
192 
193  if(const auto* clsTmplDecl = result.Nodes.getNodeAs<ClassTemplateDecl>(idDecl)) {
194  const auto endOfCond = FindLocationAfterSemi(clsTmplDecl->getEndLoc(), result);
195  InsertIndentedText(endOfCond, outputFormatHelper);
196 
197  } else { // explicit specialization, we have to remove the specialization
198  mRewrite.ReplaceText(clsTmplSpecDecl->getSourceRange(), outputFormatHelper);
199  }
200 
201  } else if(const auto* clsTmplDecl = result.Nodes.getNodeAs<ClassTemplateDecl>(idDecl)) {
202 
203  OutputFormatHelper outputFormatHelper{};
204 
205  CodeGenerator codeGenerator{outputFormatHelper};
206  codeGenerator.InsertArg(clsTmplDecl);
207 
208  mRewrite.ReplaceText(
209  {clsTmplDecl->getSourceRange().getBegin(), FindLocationAfterSemi(clsTmplDecl->getEndLoc(), result)},
210  outputFormatHelper);
211 
212  } else if(const auto* conceptDecl = result.Nodes.getNodeAs<ConceptDecl>(idDecl)) {
213 
214  OutputFormatHelper outputFormatHelper{};
215 
216  CodeGenerator codeGenerator{outputFormatHelper};
217  codeGenerator.InsertArg(conceptDecl);
218 
219  mRewrite.ReplaceText({conceptDecl->getSourceRange().getBegin(),
220  FindLocationAfterSemi(conceptDecl->getEndLoc(), result).getLocWithOffset(1)},
221  outputFormatHelper);
222 
223  } else if(const auto* vd = result.Nodes.getNodeAs<VarTemplateDecl>(idVar)) {
224  OutputFormatHelper outputFormatHelper = InsertInstantiatedTemplate(vd);
225 
226  // Keep the primary template in case it has no initializer
227  RETURN_IF(outputFormatHelper.empty());
228 
229  if(auto sourceRange = vd->getSourceRange(); not IsMacroLocation(sourceRange)) {
230  const auto endOfCond = FindLocationAfterSemi(vd->getEndLoc(), result);
231 
232  mRewrite.ReplaceText({sourceRange.getBegin(), endOfCond.getLocWithOffset(1)}, outputFormatHelper);
233 
234  } else {
235  // We're just interested in the start location, -1 work(s|ed)
236  const auto startLoc =
237  GetSourceRangeAfterSemi(sourceRange, result, RequireSemi::No).getBegin().getLocWithOffset(-1);
238 
239  InsertIndentedText(startLoc, outputFormatHelper);
240  }
241  }
242 }
