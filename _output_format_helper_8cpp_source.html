24 void OutputFormatHelper::AppendParameterList(const ArrayRef<ParmVarDecl*> parameters,
25  const NameOnly nameOnly,
26  const GenMissingParamName genMissingParamName)
27 {
28  int count{};
29 
30  ForEachArg(parameters, [&](const auto& p) {
31  auto name{GetName(*p)};
32 
33  // A special case for CXXInheritedCtor. A user can omit the parameters name, but wihtout a name the call to the
34  // base constructor may look like calling the default constructor. In such a case we create a name.
35  if((GenMissingParamName::Yes == genMissingParamName) && (0 == name.length())) {
36  name = BuildInternalVarName(StrCat("param", count));
37  ++count;
38  }
39 
40  // Get the attributes and insert them, if there are any
41  CodeGenerator codeGenerator{*this};
42  codeGenerator.InsertAttributes(p->attrs());
43 
44  if(NameOnly::No == nameOnly) {
45  const auto& type{p->getType()};
46 
47  Append(GetTypeNameAsParameter(type, name));
48  } else {
49  Append(name);
50 
51  if(isa<PackExpansionType>(p->getType())) {
52  Append(kwElipsis);
53  }
54  }
55  });

73 void OutputFormatHelper::RemoveIndent()
74 {
75  /* After a newline we are already indented by one level to much. Try to decrease it. */
76  if(0 != mDefaultIndent) {
77  // go the string backwards and find the first non-whitespace character
78  const auto res = std::find_if(
79  std::rbegin(mOutput), std::rbegin(mOutput) + SCOPE_INDENT, [](const char& c) { return ' ' != c; });
80 
81  // check if the string did end with at least one whitespace
82  if(const auto& end = std::rbegin(mOutput); res != end) {
83  // remove the whitespaces at the end of the string
84  mOutput.resize(mOutput.size() - std::distance(end, res));
85  }
86  }
87 }
