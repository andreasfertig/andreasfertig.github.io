57 void OutputFormatHelper::RemoveIndent()
58 {
59  /* After a newline we are already indented by one level to much. Try to decrease it. */
60  if(0 != mDefaultIndent) {
61  // go the string backwards and find the first non-whitespace character
62  const auto res = std::find_if(
63  std::rbegin(mOutput), std::rbegin(mOutput) + SCOPE_INDENT, [](const char& c) { return ' ' != c; });
64 
65  // check if the string did end with at least one whitespace
66  if(const auto& end = std::rbegin(mOutput); res != end) {
67  // remove the whitespaces at the end of the string
68  mOutput.resize(mOutput.size() - std::distance(end, res));
69  }
