54 inline std::string Normalize(const APValue& arg)
55 {
56  switch(arg.getKind()) {
57  case APValue::Int: return Normalize(arg.getInt());
58  case APValue::Float: {
59  std::string str{};
60  ::llvm::raw_string_ostream stream{str};
61 
62  arg.getFloat().print(stream);
63  str.pop_back();
64 
65  if(std::string::npos == str.find('.')) {
66  /* in case it is a number like 10.0 toString() seems to leave out the .0. However, as this distinguished
67  * between an integer and a floating point literal we need that dot. */
68  str.append(".0");
69  }
70 
71  return str;
72  }
73 
74  default: break;
75  }
76 
77  return std::string{"unsupported APValue"};
78 }

93 template<class T>
94 inline decltype(auto) Normalize(const T& arg)
95 {
96  // Handle bool's first, we like their string representation.
97  if constexpr(std::is_same_v<std::remove_cvref_t<T>, bool>) {
98  return details::ConvertToBoolString(arg);
99 
100  } else if constexpr(std::is_integral_v<T>) {
101  return std::to_string(arg);
102 
103  } else {
104  return (arg);
105  }
106 }
