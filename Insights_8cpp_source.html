122 int main(int argc, const char** argv)
123 {
124  llvm::sys::PrintStackTraceOnErrorSignal(argv[0]);
125  llvm::cl::HideUnrelatedOptions(gInsightCategory);
126  llvm::cl::SetVersionPrinter(&PrintVersion);
127 
128  CommonOptionsParser op(argc, argv, gInsightCategory);
129  ClangTool tool(op.getCompilations(), op.getSourcePathList());
130 
131  llvm::StringRef sourceFilePath = op.getSourcePathList().front();
132  // In STDINMode, we override the file content with the <stdin> input.
133  // Since `tool.mapVirtualFile` takes `StringRef`, we define `Code` outside of
134  // the if-block so that `Code` is not released after the if-block.
135  std::unique_ptr<llvm::MemoryBuffer> inMemoryCode{};
136 
137  if(gStdinMode) {
138  assert(op.getSourcePathList().size() == 1 && "Expect exactly one file path in STDINMode.");
139  llvm::ErrorOr<std::unique_ptr<llvm::MemoryBuffer>> codeOrErr = llvm::MemoryBuffer::getSTDIN();
140  if(const std::error_code errorCode = codeOrErr.getError()) {
141  llvm::errs() << errorCode.message() << "\n";
142  return 1;
143  }
144  inMemoryCode = std::move(codeOrErr.get());
145  if(inMemoryCode->getBufferSize() == 0) {
146  Error("empty file\n");
147  return 1; // Skip empty files.
148  }
149 
150  tool.mapVirtualFile(sourceFilePath, inMemoryCode->getBuffer());
151  }
152 
153  return tool.run(newFrontendActionFactory<CppInsightFrontendAction>().get());
154 }
