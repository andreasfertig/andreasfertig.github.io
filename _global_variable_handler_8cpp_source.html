28 GlobalVariableHandler::GlobalVariableHandler(Rewriter& rewrite, MatchFinder& matcher)
29 : InsightsBase(rewrite)
30 {
31  matcher.addMatcher(
32  varDecl(unless(anyOf(
33  isExpansionInSystemHeader(),
34  isInvalidLocation(),
35  hasAncestor(varTemplateDecl()),
36  hasAncestor(functionDecl()),
37  hasAncestor(cxxRecordDecl()),
38  hasAncestor(typeAliasDecl()),
39  hasAncestor(cxxMethodDecl()),
40  parmVarDecl(),
41  // don't match a VarDecl within a VarDecl. Happens for example in lambdas.
42  hasAncestor(varDecl()),
43  varTemplateSpecDecl(),
44  // A DecompositionDecl in global scope is different in the AST than one in a function for example.
45  // Try to find out whether this VarDecl is the result of a DecompositionDecl, if so bail out.
46  hasInitializer(ignoringImpCasts(
47  callExpr(hasAnyArgument(ignoringParenImpCasts(declRefExpr(to(decompositionDecl()))))))),
48  // don't replace anything in templates
49  isTemplate)))
50  .bind("varDecl"),
51  this);
52 }

55 void GlobalVariableHandler::run(const MatchFinder::MatchResult& result)
56 {
57  if(const auto* matchedDecl = result.Nodes.getNodeAs<VarDecl>("varDecl")) {
