35 GlobalVariableHandler::GlobalVariableHandler(Rewriter& rewrite, MatchFinder& matcher)
36 : InsightsBase(rewrite)
37 {
38  matcher.addMatcher(
39  varDecl(unless(anyOf(
40  isExpansionInSystemHeader(),
41  isInvalidLocation(),
42  hasAncestor(varTemplateDecl()),
43  hasAncestor(functionDecl()),
44  hasAncestor(cxxRecordDecl()),
45  hasAncestor(namespaceDecl()),
46  hasAncestor(typeAliasDecl()),
47  hasAncestor(cxxMethodDecl()),
48  parmVarDecl(),
49  // don't match a VarDecl within a VarDecl. Happens for example in lambdas.
50  hasAncestor(varDecl()),
51  varTemplateSpecDecl(),
52  // A DecompositionDecl in global scope is different in the AST than one in a function for example.
53  // Try to find out whether this VarDecl is the result of a DecompositionDecl, if so bail out.
54  hasInitializer(ignoringImpCasts(
55  callExpr(hasAnyArgument(ignoringParenImpCasts(declRefExpr(to(decompositionDecl()))))))),
56  // don't replace anything in templates
57  isTemplate)))
58  .bind("varDecl"),
59  this);
60 }

63 void GlobalVariableHandler::run(const MatchFinder::MatchResult& result)
64 {
65  if(const auto* matchedDecl = result.Nodes.getNodeAs<VarDecl>("varDecl")) {
66  OutputFormatHelper outputFormatHelper{};
67 
68  const auto sr = GetSourceRangeAfterSemi(matchedDecl->getSourceRange(), result, RequireSemi::Yes);
69 
70  // Check whether we already have rewritten this location. If so, insert the text after the location. This is the
71  // case for:
72  // - an anonymous struct declared with TU as root
73  // - a out-of-line static member variable of a class template
74  if(not IsMacroLocation(matchedDecl->getSourceRange()) && (mRewrite.buffer_begin() != mRewrite.buffer_end()) &&
75  (isTemplateInstantiation(matchedDecl->getTemplateSpecializationKind()) ||
76  IsAnonymousStructOrUnion(matchedDecl->getType()->getAsCXXRecordDecl()))) {
77  if(auto&& text = mRewrite.getRewrittenText(sr); not text.empty()) {
78  const char* data = result.SourceManager->getCharacterData(matchedDecl->getSourceRange().getBegin());
79 
80  StringRef sref{data, static_cast<size_t>(std::distance(text.begin(), text.end()))};
81 
82  if(not std::equal(text.begin(), text.end(), sref.begin(), sref.end())) {
83  outputFormatHelper.Append(std::move(text));
84  }
85  }
86  }
87 
88  CodeGenerator codeGenerator{outputFormatHelper};
89  codeGenerator.InsertArg(matchedDecl);
90 
91  if(IsMacroLocation(sr)) {
92  // Special case for AnonymousStructInMacroTest.cpp (#290) where a macro gets expanded
93  mRewrite.ReplaceText(GetSM(result).getImmediateExpansionRange(matchedDecl->getSourceRange().getBegin()),
94  outputFormatHelper.GetString());
95 
96  } else {
97  mRewrite.ReplaceText(sr, outputFormatHelper.GetString());
98  }
99  }
100 }
