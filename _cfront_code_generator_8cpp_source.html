92 static std::string GetSpecialMemberName(const ValueDecl* vd, QualType type)
93 {
94  if(const auto* md = dyn_cast_or_null<CXXMethodDecl>(vd)) {
95  // GetName below would return a[4] for example. To avoid the array part we go for the underlying type.
96  if(const auto* ar = dyn_cast_or_null<ArrayType>(type)) {
97  type = ar->getElementType();
98  }
99 
100  auto rdName = GetName(type);
101 
102  if(const auto* ctor = dyn_cast_or_null<CXXConstructorDecl>(md)) {
103  if(ctor->isCopyConstructor()) {
104  return StrCat("CopyConstructor_"sv, rdName);
105  } else if(ctor->isMoveConstructor()) {
106  return StrCat("MoveConstructor_"sv, rdName);
107  }
108 
109  return StrCat("Constructor_"sv, rdName);
110 
111  } else if(isa<CXXDestructorDecl>(md)) {
112  return StrCat("Destructor_"sv, rdName);
113  }
114  }
115 
116  return GetName(*vd);
117 }

185 void CfrontCodeGenerator::InsertArg(const CXXDeleteExpr* stmt)
186 {
187  auto destroyedType = stmt->getDestroyedType();
188  auto* arg = const_cast<Expr*>(stmt->getArgument());
189 
190  StmtsContainer bodyStmts{};
191 
192  if(const auto hasDtor{HasDtor(destroyedType)}; stmt->isArrayForm() and hasDtor) {
193  bodyStmts.Add(CallVecDelete(arg, destroyedType));
194  } else if(hasDtor) {
195  bodyStmts.Add(Call(StrCat("Destructor_"sv, GetName(destroyedType)), {arg}));
196  }
197 
198  EnableGlobalInsert(GlobalInserts::FuncFree);
199 
200  bodyStmts.Add(Call("free"sv, {arg}));
201 
202  InsertArg(If(arg, {bodyStmts}));
203  mInsertSemi = false; // Since we tamper with a CompoundStmt we signal _not_ to insert the usual semi
204 }

207 static auto* CallVecNewOrCtor(std::string_view ctorName,
208  Expr* objectParam,
209  QualType allocatedType,
210  Expr* arraySizeExpr,
211  std::string_view funName)
212 {
213  // According to "Inside the C++ Object Model" the dtor is required in case one of the array element ctors fails the
214  // already constructed one must be destroyed.
215  // See also: https://itanium-cxx-abi.github.io/cxx-abi/abi.html#array-ctor
216  auto dtorName = [&]() {
217  if(HasDtor(allocatedType)) {
218  return GetSpecialMemberName(allocatedType->getAsCXXRecordDecl()->getDestructor());
219  }
220 
221  return std::string{kwNull};
222  }();
223 
224  SmallVector<Expr*, 6> args{objectParam,
225  Sizeof(allocatedType),
226  arraySizeExpr,
227  Int32(0), // XXX what is the correct value?
228  CastToVoidFunPtr(ctorName),
229  CastToVoidFunPtr(dtorName)};
230 
231  return Call(funName, args);
232 }

252 void CfrontCodeGenerator::InsertArg(const CXXNewExpr* cstmt)
253 {
254  auto* stmt = const_cast<CXXNewExpr*>(cstmt);
255 
256  auto allocatedType = stmt->getAllocatedType();
257  auto ctorName = StrCat("Constructor_"sv, GetName(allocatedType));
258 
259  if(stmt->isArray()) {
260  EnableGlobalInsert(GlobalInserts::FuncMalloc);
261 
262  auto* arraySizeExpr = stmt->getArraySize().value();
263  auto* callMalloc = Call("malloc"sv, {Mul(Sizeof(allocatedType), arraySizeExpr)});
264 
265  Expr* assign{};
266 
267  // According to "Inside the C++ Object Model" a trivial and literal type does use plain malloc as it does not
268  // require a ctor/dtor.
269  if(not HasCtor(allocatedType) and not HasDtor(allocatedType)) {
270  assign = Cast(callMalloc, Ptr(allocatedType));
271 
272  } else {
273  if(allocatedType->getAsCXXRecordDecl()->ctors().empty()) {
274  EnableGlobalInsert(GlobalInserts::HeaderStddef);
275 
276  ctorName = kwNull;
277  }
278 
279  assign = Cast(CallVecNew(ctorName, callMalloc, allocatedType, arraySizeExpr), Ptr(allocatedType));
280  }
281 
282  InsertArg(assign);
283 
284  if(stmt->hasInitializer() and allocatedType->isBuiltinType()) { // Ignore CXXConstructExpr
285  // The resulting code here is slightly different from the cfront code. For
286  //
287  // int* p = new int(2);
288  //
289  // C++ Insights generates:
290  //
291  // int* p = (int*)malloc(sizeof(int));
292  // if(p) { *p = 2; }
293  //
294  // However, due to Cs requirement to only declare variables and delay initialization the cfront code is
295  //
296  // int* p;
297  // if(p = (int*)malloc(sizeof(int))) { *p = 2; }
298  mOutputFormatHelper.AppendSemiNewLine();
299 
300  if(auto* inits = dyn_cast_or_null<InitListExpr>(stmt->getInitializer())) {
301  auto* expr = [&]() -> const Expr* {
302  if(auto* vd = dyn_cast_or_null<VarDecl>(mLastDecl)) {
303  return mkDeclRefExpr(vd);
304  }
305 
306  return mLastExpr;
307  }();
308 
309  StmtsContainer bodyStmts{};
310 
311  for(auto idx : NumberIterator(inits->getNumInits())) {
312  bodyStmts.Add(Assign(ArraySubscript(expr, idx, allocatedType), inits->getInit(idx)));
313  }
314 
315  InsertArg(If(expr, {bodyStmts}));
316 
317  mInsertSemi = false; // Since we tamper with a CompoundStmt we signal _not_ to insert the usual semi
318  }
319  }
320 
321  return;
322  }
323 
324  auto* mallocCall = [&]() -> Expr* {
325  if(stmt->getNumPlacementArgs()) {
326  return stmt->getPlacementArg(0);
327  }
328 
329  EnableGlobalInsert(GlobalInserts::FuncMalloc);
330 
331  return Call("malloc"sv, {Sizeof(allocatedType)});
332  }();
333 
334  mallocCall = Cast(mallocCall, Ptr(allocatedType));
335 
336  if(allocatedType->isBuiltinType()) {
337  InsertArg(mallocCall);
338 
339  if(stmt->hasInitializer()) {
340  // The resulting code here is slightly different from the cfront code. For
341  //
342  // int* p = new int(2);
343  //
344  // C++ Insights generates:
345  //
346  // int* p = (int*)malloc(sizeof(int));
347  // if(p) { *p = 2; }
348  //
349  // However, due to Cs requirement to only declare variables and delay initialization the cfront code is
350  //
351  // int* p;
352  // if(p = (int*)malloc(sizeof(int))) { *p = 2; }
353  mOutputFormatHelper.AppendSemiNewLine();
354 
355  auto* varDeclRefExpr = mkDeclRefExpr(dyn_cast_or_null<VarDecl>(mLastDecl));
356 
357  StmtsContainer bodyStmts{Assign(Dref(varDeclRefExpr), stmt->getInitializer())};
358 
359  InsertArg(If(varDeclRefExpr, {bodyStmts}));
360 
361  mInsertSemi = false; // Since we tamper with a CompoundStmt we signal _not_ to insert the usual semi
362  }
363 
364  return;
365  }
366 
367  SmallVector<Expr*, 16> args{mallocCall};
368  if(auto* ncCtorExpr = const_cast<CXXConstructExpr*>(stmt->getConstructExpr())) {
369  args.append(ncCtorExpr->arg_begin(), ncCtorExpr->arg_end());
370  }
371 
372  InsertArg(Call(ctorName, args));
373 }

395 void CfrontCodeGenerator::InsertCXXMethodDecl(const CXXMethodDecl* stmt, SkipBody)
396 {
397  OutputFormatHelper initOutputFormatHelper{};
398  initOutputFormatHelper.SetIndent(mOutputFormatHelper, OutputFormatHelper::SkipIndenting::Yes);
399 
400  auto recordDeclType = GetRecordDeclType(stmt);
401  if(stmt->isConst()) {
402  recordDeclType.addConst();
403  }
404 
405  auto parentType = Ptr(recordDeclType);
406  auto* body = stmt->getBody();
407  StmtsContainer bodyStmts{};
408 
409  if(body) {
410  bodyStmts.AddBodyStmts(body);
411  }
412 
413  auto retType = stmt->getReturnType();
414 
415  auto processBaseClassesAndFields = [&](const CXXRecordDecl* parent) {
416  auto* thisOfParent = mkVarDeclRefExpr(kwInternalThis, parentType);
417 
418  auto insertFields = [&](const RecordDecl* rd) {
419  for(auto* fieldDecl : rd->fields()) {
420  if(const auto* cxxRecordDecl = fieldDecl->getType()->getAsCXXRecordDecl()) {
421  // We don't need any checks like isDefaultConstructible. We would not reach this
422  // point in any other case.
423 
424  auto lvalueRefType = GetGlobalAST().getLValueReferenceType(parentType);
425  auto* lhsMemberExpr = AccessMember(kwInternalThis, fieldDecl, lvalueRefType);
426  auto* rhsMemberExpr = AccessMember("__rhs"sv, fieldDecl, lvalueRefType);
427 
428  // Add call to ctor
429  bodyStmts.AddBodyStmts(Call(GetSpecialMemberName(stmt, GetRecordDeclType(cxxRecordDecl)),
430  {Ref(lhsMemberExpr), Ref(rhsMemberExpr)}));
431 
432  } else {
433  auto* rhsMemberExpr = AccessMember("__rhs"sv, fieldDecl, parentType);
434 
435  bodyStmts.AddBodyStmts(Assign(thisOfParent, fieldDecl, rhsMemberExpr));
436  }
437  }
438  };
439 
440  for(const auto& base : parent->bases()) {
441  const auto rd = base.getType()->getAsRecordDecl();
442 
443  insertFields(rd);
444 
445  auto* lhsCast = StaticCast(GetRecordDeclType(rd), thisOfParent, true);
446  auto* rhsDeclRef = mkVarDeclRefExpr("__rhs"sv, parentType);
447  auto* rhsCast = StaticCast(GetRecordDeclType(rd), rhsDeclRef, true);
448 
449  auto* callAssignBase = Call(GetSpecialMemberName(stmt, GetRecordDeclType(rd)), {lhsCast, rhsCast});
450 
451  bodyStmts.AddBodyStmts(callAssignBase);
452  }
453 
454  // insert own fields
455  insertFields(parent);
456  };
457 
458  // Skip ctor for trivial types
459  if(const auto* ctorDecl = dyn_cast_or_null<CXXConstructorDecl>(stmt)) {
460 
461  const auto* parent = stmt->getParent();
462 
463  if(not stmt->doesThisDeclarationHaveABody()) {
464  if(IsCopyOrMoveCtor(ctorDecl)) {
465  if(const bool showSpecialMemberFunc{stmt->isUserProvided() or stmt->isExplicitlyDefaulted()};
466  not showSpecialMemberFunc) {
467  return;
468  }
469 
470  processBaseClassesAndFields(parent);
471  }
472  } else if(ctorDecl->isDefaultConstructor()) {
473  auto insertFields = [&](const RecordDecl* rd) {
474  for(auto* fieldDecl : rd->fields()) {
475  if(auto* initializer = fieldDecl->getInClassInitializer();
476  initializer and fieldDecl->hasInClassInitializer()) {
477  const bool isConstructorExpr{isa<CXXConstructExpr>(initializer) or
478  isa<ExprWithCleanups>(initializer)};
479 
480  if(not isConstructorExpr) {
481  auto* lhsMemberExpr = AccessMember(kwInternalThis, fieldDecl, Ptr(GetRecordDeclType(rd)));
482 
483  bodyStmts.AddBodyStmts(Assign(lhsMemberExpr, fieldDecl, initializer));
484 
485  } else {
486  bodyStmts.AddBodyStmts(CallConstructor(fieldDecl->getType(),
487  Ptr(GetRecordDeclType(rd)),
488  fieldDecl,
489  ArgsToExprVector(initializer),
490  DoCast::No,
491  AsReference::Yes));
492  }
493 
494  } else if(const auto* cxxRecordDecl = fieldDecl->getType()->getAsCXXRecordDecl()) {
495  // We don't need any checks like isDefaultConstructible. We would not reach this
496  // point in any other cause.
497 
498  if(auto lhsType = fieldDecl->getType(); HasCtor(lhsType)) {
499  bodyStmts.AddBodyStmts(CallConstructor(GetRecordDeclType(cxxRecordDecl),
500  Ptr(GetRecordDeclType(cxxRecordDecl)),
501  fieldDecl,
502  {},
503  DoCast::No,
504  AsReference::Yes));
505 
506  } else {
507  bodyStmts.Add(
508  Comment(StrCat(GetName(*fieldDecl), " // trivial type, maybe uninitialized"sv)));
509  // Nothing to do here, we are looking at an uninitialized variable
510  }
511  }
512  }
513  };
514 
515  for(const auto& base : parent->bases()) {
516  auto baseType = base.getType();
517  bodyStmts.AddBodyStmts(CallConstructor(baseType, Ptr(baseType), nullptr, {}, DoCast::Yes));
518 
519  insertFields(baseType->getAsRecordDecl());
520  }
521 
522  // insert own fields
523  insertFields(parent);
524 
525  } else {
526  // user ctor
527  for(const auto* init : ctorDecl->inits()) {
528  auto* member = init->getMember();
529 
530  if(not isa<CXXConstructExpr>(init->getInit())) {
531  if(not init->getAnyMember()) {
532  continue;
533  }
534 
535  auto* lhsMemberExpr = AccessMember(kwInternalThis, member, Ptr(GetRecordDeclType(parent)));
536 
537  bodyStmts.AddBodyStmts(Assign(lhsMemberExpr, member, init->getInit()));
538  continue;
539 
540  } else if(init->isBaseInitializer()) {
541  bodyStmts.AddBodyStmts(init->getInit());
542  continue;
543  }
544 
545  auto ctorType = member->getType();
546 
547  auto* lhsMemberExpr = AccessMember(kwInternalThis, member, ctorType);
548 
549  auto callParams{ArgsToExprVector(init->getInit())};
550  callParams.insert(callParams.begin(), Ref(lhsMemberExpr));
551 
552  bodyStmts.AddBodyStmts(
553  Call(GetSpecialMemberName(stmt, GetRecordDeclType(ctorType->getAsRecordDecl())), callParams));
554  }
555  }
556 
557  auto* lhsDeclRef = mkVarDeclRefExpr(kwInternalThis, Ptr(ctorDecl->getType()));
558 
559  bodyStmts.AddBodyStmts(Return(lhsDeclRef));
560 
561  body = mkCompoundStmt({bodyStmts});
562  retType = parentType;
563 
564  // copy and move assignment op
565  } else if(IsCopyOrMoveAssign(stmt)) {
566  if(not stmt->doesThisDeclarationHaveABody() or stmt->isDefaulted()) {
567 
568  // we don't want the default generated body
569  bodyStmts.clear();
570 
571  processBaseClassesAndFields(stmt->getParent());
572  }
573 
574  auto* lhsDeclRef = mkVarDeclRefExpr(kwInternalThis, Ptr(stmt->getType()));
575 
576  bodyStmts.AddBodyStmts(Return(lhsDeclRef));
577 
578  body = mkCompoundStmt({bodyStmts});
579  retType = parentType;
580 
581  } else if(const auto* dtor = dyn_cast_or_null<CXXDestructorDecl>(stmt)) {
582  // Based on: https://www.dre.vanderbilt.edu/~schmidt/PDF/C++-translation.pdf
583 
584  if(not HasDtor(GetRecordDeclType(dtor->getParent()))) {
585  return;
586  }
587 
588  for(const auto& base : llvm::reverse(dtor->getParent()->bases())) {
589  if(not dtor->isVirtual()) {
590  continue;
591  }
592 
593  auto* lhsDeclRef = mkVarDeclRefExpr(kwInternalThis, Ptr(base.getType()));
594  auto* cast = Cast(lhsDeclRef, lhsDeclRef->getType());
595 
596  bodyStmts.Add(
597  Call(GetSpecialMemberName(stmt, GetRecordDeclType(base.getType()->getAsRecordDecl())), {cast}));
598 
599  body = mkCompoundStmt({bodyStmts});
600  }
601  }
602 
603  params_store params{};
604  params.reserve(stmt->getNumParams() + 1);
605 
606  if(not IsStaticStorageClass(stmt) and not stmt->isStatic()) {
607  params.emplace_back(kwInternalThis, parentType);
608  }
609 
610  for(const auto& param : stmt->parameters()) {
611  std::string name{GetName(*param)};
612  auto type = param->getType();
613 
614  // at least in case of a copy constructor modify the parameters
615  if((0 == name.length()) and
616  (IsCopyOrMoveCtor(dyn_cast_or_null<CXXConstructorDecl>(stmt)) or IsCopyOrMoveAssign(stmt))) {
617  name = "__rhs"sv;
618  type = Ptr(type.getNonReferenceType());
619  }
620 
621  params.emplace_back(name, type);
622  }
623 
624  auto* callSpecialMemberFn = Function(GetSpecialMemberName(stmt), retType, to_params_view(params));
625  callSpecialMemberFn->setInlineSpecified(stmt->isInlined());
626  callSpecialMemberFn->setStorageClass((stmt->isStatic() or IsStaticStorageClass(stmt)) ? SC_Static : SC_None);
627  callSpecialMemberFn->setBody(body);
628 
629  InsertArg(callSpecialMemberFn);
630 
631  mOutputFormatHelper.AppendNewLine();
632 }

683 void CfrontCodeGenerator::InsertArg(const CXXRecordDecl* stmt)
684 {
685  auto* recordDecl = Struct(GetName(*stmt));
686 
687  if(stmt->hasDefinition()) {
688  auto AddField = [&](const FieldDecl* field) {
689  auto* fieldDecl = mkFieldDecl(recordDecl, GetName(*field), field->getType());
690 
691  recordDecl->addDecl(fieldDecl);
692  };
693 
694  // Insert field from base classes
695  for(const auto& base : stmt->bases()) {

707  }

750 void CfrontCodeGenerator::InsertArg(const CXXMemberCallExpr* stmt)
751 {
752  if(const auto* me = dyn_cast_or_null<MemberExpr>(stmt->getCallee())) {
753  auto* obj = me->getBase();
754 
755  const bool isReference = IsReferenceType(dyn_cast_or_null<VarDecl>(obj->getReferencedDeclOfCallee()));
756 
757  if(not obj->getType()->isPointerType() and not isReference) {
758  obj = Ref(obj);
759  }
760 
761  if(const auto* matExpr = dyn_cast_or_null<MaterializeTemporaryExpr>(me->getBase())) {
762  if(const auto* tmpExpr = dyn_cast_or_null<CXXBindTemporaryExpr>(matExpr->getSubExpr())) {
763  if(const auto* tmpObjExpr = dyn_cast_or_null<CXXTemporaryObjectExpr>(tmpExpr->getSubExpr())) {
764  obj = const_cast<CXXTemporaryObjectExpr*>(tmpObjExpr);
765  }
766  }
767  }
768 
769  if(const auto* ar = dyn_cast_or_null<ConstantArrayType>(obj->getType())) {
770  if(const auto* dtor = dyn_cast_or_null<CXXDestructorDecl>(me->getMemberDecl())) {
771  // ignore the reference
772  InsertArg(CallVecDtor(dyn_cast_or_null<UnaryOperator>(obj)->getSubExpr(), ar));
773  return;
774  }
775  }
776 
777  SmallVector<Expr*, 16> params{obj};
778  auto* ncStmt = const_cast<CXXMemberCallExpr*>(stmt);
779  params.append(ncStmt->arg_begin(), ncStmt->arg_end());
780 
781  InsertArg(Call(GetSpecialMemberName(me->getMemberDecl()), params));
782 
783  } else {
784  CodeGenerator::InsertArg(stmt);
785  }
786 }

790 {

792  CodeGenerator::InsertArg(stmt);
793  return;
794  }
795 
796  params_store params{};
797  SmallVector<Expr*, 16> args{};
798 
799  for(auto* param : stmt->parameters()) {
800  params.emplace_back(GetName(*param), param->getType());
801  args.push_back(mkDeclRefExpr(param));
802  }
803 
804  auto mainName{"main"sv};

806

808  intMain->setBody(stmt->getBody());
809  intMain->setHasImplicitReturnZero(true);
810 
811  InsertArg(intMain);
812  mOutputFormatHelper.AppendNewLine();
813

818

820

823  modMain->setBody(body);

826 }

829 void CfrontCodeGenerator::InsertArg(const CXXConstructExpr* stmt)
830 {
831  if(P0315Visitor dt{*this}; not dt.TraverseType(stmt->getType())) {
832  if(not mLambdaStack.empty()) {
833  for(const auto& e : mLambdaStack) {
834  RETURN_IF(LambdaCallerType::VarDecl == e.callerType());
835  }
836  }
837  }
838 
839  auto ctor = stmt->getConstructor();
840  auto ctorName = GetSpecialMemberName(ctor);
841  auto* vd = dyn_cast_or_null<VarDecl>(mLastDecl);
842 
843 #if 0
844  // XXX: An expression like C c[4]{4,5,6} with C having a ctor does show up wrong at the moment.
845  if(not ar) {
846  ar = dyn_cast_or_null<ConstantArrayType>(vd->getType());
847 
848  if(ar) {
849  mLastStmt->dump();
850  openScope = true;
851  // mOutputFormatHelper.CloseScopeWithSemi();
852  mOutputFormatHelper.AppendSemiNewLine();
853  }
854  }
855 #endif
856 
857  auto InsertCallCtor = [&](Expr* varNameRef) {
858  SmallVector<Expr*, 16> args{Cast(varNameRef, Ptr(stmt->getType()))};
859 
860  for(int i = 0; auto* arg : stmt->arguments()) {
861  if(IsCopyOrMoveCtor(ctor) or ctor->getParamDecl(i)->getType()->isReferenceType()) {
862  args.push_back(Ref(arg));
863 
864  } else {
865  args.push_back(const_cast<Expr*>(arg));
866  }
867 
868  ++i;
869  }
870 
871  InsertArg(Call(ctorName, args));
872  };
873 
874  // For an array we need to call __vec_new
875  if(const auto* ar = dyn_cast_or_null<ConstantArrayType>(stmt->getType())) {
876  if(not HasCtor(ar->getElementType())) {
877  mInsertSemi = false;
878  return;
879  }
880 
881  InsertArg(CallVecCtor(ctorName, vd, ar->getElementType(), Int32(GetSize(ar))));
882 
883  } else if(const auto* tmpObjectExpr = dyn_cast_or_null<CXXTemporaryObjectExpr>(stmt); vd and not tmpObjectExpr) {
884  if(not HasCtor(vd->getType())) {
885  mInsertSemi = false;
886  } else {
887  auto* varNameRef = Ref(mkDeclRefExpr(vd));
888 
889  InsertCallCtor(varNameRef);
890  }
891 
892  } else if(tmpObjectExpr) {
893  auto* varNameRef = Ref(mkVarDeclRefExpr(GetName(*tmpObjectExpr), stmt->getType()));
894 
895  if(not HasCtor(stmt->getType())) {
896  InsertArg(varNameRef);
897  return;
898  }
899 
900  InsertCallCtor(varNameRef);
901 
902  } else {
903  InsertArg(CallConstructor(
904  stmt->getType(), Ptr(GetRecordDeclType(ctor)), nullptr, ArgsToExprVector(stmt), DoCast::Yes));
905  }
906 }
