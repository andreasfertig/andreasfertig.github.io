259 /*

261  *
262  * This class is a specialization to handle cases where we can have multiple init statements to the same variable and
263  * hence need only one time the \c VarDecl. An example a for-loops:

266 \endcode
267  */
268 class MultiStmtDeclCodeGenerator final : public CodeGenerator
269 {
270 public:

272 
273  // Insert the semi after the last declaration. This implies that this class always requires its own scope.
274  ~MultiStmtDeclCodeGenerator() { mOutputFormatHelper.Append("; "); }
275 
276  using CodeGenerator::InsertArg;
277 
278 protected:
279  OnceTrue mInsertVarDecl; //! Insert the \c VarDecl only once.

281  //! insert another one.
