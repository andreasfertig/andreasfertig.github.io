203 static std::string GetNameInternal(const QualType& t, const Unqualified unqualified)
204 {
205  if(const auto* memberPointerType = t->getAs<MemberPointerType>()) {
206  if(const auto* recordType2 = dyn_cast_or_null<RecordType>(memberPointerType->getClass())) {
207  if(const auto* decl = recordType2->getDecl()) {
208  if(const auto* cxxRecordDecl = dyn_cast_or_null<CXXRecordDecl>(decl)) {
209  if(cxxRecordDecl->isLambda()) {
210  std::string result{GetAsCPPStyleString(t)};
211 
212  static const std::string clangLambdaName{"(lambda)::*"};
213  if(auto pos = result.find(clangLambdaName); std::string::npos != pos) {
214  std::string lambdaName = GetLambdaName(*cxxRecordDecl);
215  lambdaName += "::*";
216 
217  result.replace(pos, clangLambdaName.length(), lambdaName);
218  return result;
219  }
220  }
221  }
222  }
223  }
224  }
225 
226  if(t.getTypePtrOrNull()) {
227  std::string refOrPointer{};
228  const RecordType* recordType = [&]() -> const RecordType* {
229  const auto& ct = t.getCanonicalType();
230  const auto* ctp = ct.getTypePtrOrNull();
231 
232  if(const auto* sta = dyn_cast_or_null<RValueReferenceType>(ctp)) {
233  refOrPointer = "&&";
234  return dyn_cast_or_null<RecordType>(sta->getPointeeTypeAsWritten().getTypePtrOrNull());
235 
236  } else if(const auto* ref = dyn_cast_or_null<ReferenceType>(ctp)) {
237  refOrPointer = "&";
238  return dyn_cast_or_null<RecordType>(ref->getPointeeTypeAsWritten().getTypePtrOrNull());
239 
240  } else if(const auto* ptr = dyn_cast_or_null<PointerType>(ctp)) {
241  refOrPointer = "*";
242  return dyn_cast_or_null<RecordType>(ptr->getPointeeType().getTypePtrOrNull());
243  }
244 
245  return dyn_cast_or_null<RecordType>(ctp);
246  }();
247 
248  if(recordType) {
249  if(const auto* decl = recordType->getDecl()) {
250  const std::string cvqStr{[&]() {
251  if(const auto* refType = dyn_cast_or_null<ReferenceType>(t.getTypePtrOrNull())) {
252  return refType->getPointeeTypeAsWritten().getLocalQualifiers().getAsString();
253  }
254 
255  return t.getCanonicalType().getLocalQualifiers().getAsString();
256  }()};
257 
258  if(const auto* tt = dyn_cast_or_null<ClassTemplateSpecializationDecl>(decl)) {
259  if(const auto* x = t.getBaseTypeIdentifier()) {
260  OutputFormatHelper ofm{};
261 
262  if((Unqualified::No == unqualified) && !cvqStr.empty()) {
263  ofm.Append(cvqStr, " ");
264  }
265 
266  ofm.Append(GetScope(decl->getDeclContext()), x->getName());
267  CodeGenerator codeGenerator{ofm};
268  codeGenerator.InsertTemplateArgs(*tt);
269 
270  if(!refOrPointer.empty()) {
271  ofm.Append(" ", refOrPointer);
272  }
273 
274  return ofm.GetString();
275  }
276  } else if(const auto* cxxRecordDecl = dyn_cast_or_null<CXXRecordDecl>(decl)) {
277  if(cxxRecordDecl->isLambda()) {
278  std::string result{cvqStr};
279  if(!cvqStr.empty()) {
280  result += ' ';
281  }
282 
283  result += GetLambdaName(*cxxRecordDecl);
284  if(!refOrPointer.empty()) {
285  result += ' ';
286  result += refOrPointer;
287  }
288 
289  return result;
290  }
291  }
292  }
293  }
294  }
295 
296  if(Unqualified::Yes == unqualified) {
297  return GetAsCPPStyleString(t.getUnqualifiedType());
298  }
299 
300  return GetAsCPPStyleString(t);
301 }

316 static std::string GetName(const QualType& t, const Unqualified unqualified = Unqualified::No)
317 {
318  const auto t2 = GetDesugarType(t);
319  const auto* at = t2->getContainedAutoType();
320 
321  if(at && at->isSugared()) {
322  const auto dt = at->getDeducedType();
323 
324  // treat LValueReference special at this point. This means we are coming from auto&& and it decayed to an
325  // l-value reference.
326  if(dt->isLValueReferenceType()) {
327  return GetNameInternal(dt, unqualified);
328  }
329 
330  } else if(IsDecltypeType(t)) { // Handle decltype(var)
331  return GetNameInternal(t2, unqualified);
332  }
333 
334  return GetNameInternal(t, unqualified);
335 }

360 std::string GetTypeNameAsParameter(const QualType& t, const std::string& varName, const Unqualified unqualified)
361 {
362  const bool isFunctionPointer = HasTypeWithSubType<ReferenceType, FunctionProtoType>(t);
363  const bool isArrayRef = HasTypeWithSubType<ReferenceType, ArrayType>(t);
364  // Special case for Issue81, auto returns an array-ref and to catch auto deducing an array (Issue106)
365  const bool isAutoType = dyn_cast_or_null<AutoType>(t.getTypePtrOrNull());
366  const auto pointerToArrayBaseType = isAutoType ? t->getContainedAutoType()->getDeducedType() : t;
367  const bool isPointerToArray = HasTypeWithSubType<PointerType, ArrayType>(pointerToArrayBaseType);
368 
369  std::string typeName = details::GetName(t, unqualified);
370 
371  // Sometimes we get char const[2]. If we directly insert the typename we end up with char const__var[2] which is not
372  // a valid type name. Hence check for this condition and, if necessary, insert a space before __var.
373  auto getSpaceOrEmpty = [&](const std::string& needle) -> std::string {
374  if(not Contains(typeName, needle)) {
375  return " ";
376  }
377 
378  return "";
379  };
380 
381  if(t->isArrayType() && !t->isLValueReferenceType()) {
382  std::string space = getSpaceOrEmpty(" [");
383  InsertBefore(typeName, "[", StrCat(space, varName));
384 
385  } else if(isArrayRef) {
386  const bool isRValueRef{HasTypeWithSubType<RValueReferenceType, ArrayType>(t)};
387  const std::string contains{isRValueRef ? "(&&" : "(&"};
388 
389  if(Contains(typeName, contains)) {
390  InsertAfter(typeName, contains, varName);
391  } else {
392  const std::string insertBefore{isRValueRef ? "&&[" : "&["};
393 
394  InsertBefore(typeName, insertBefore, "(");
395  InsertAfter(typeName, contains, StrCat(varName, ")"));
396  }
397 
398  } else if(isFunctionPointer) {
399  const bool isRValueRef{HasTypeWithSubType<RValueReferenceType, FunctionProtoType>(t)};
400  const std::string contains{isRValueRef ? "(&&" : "(&"};
401 
402  if(Contains(typeName, contains)) {
403  InsertAfter(typeName, contains, varName);
404  } else {
405  typeName += StrCat(" ", varName);
406  }
407 
408  } else if(isPointerToArray) {
409  if(Contains(typeName, "(*")) {
410  InsertAfter(typeName, "(*", varName);
411  } else if(Contains(typeName, "*")) {
412  InsertBefore(typeName, "*", "(");
413  InsertAfter(typeName, "*", StrCat(varName, ")"));
414  } else {
415  typeName += StrCat(" ", varName);
416  }
417  } else if(t->isFunctionPointerType()) {
418  InsertAfter(typeName, "(*", varName);
419  } else if(!t->isArrayType() && !varName.empty()) {
420  typeName += StrCat(" ", varName);
421  }
422 
423  return typeName;
424 }

463 /*
464  * \brief Get a usable name from a template parameter pack.
465  *
466  * A template parameter pack, args, as in:
467  * \code
468 template<typename F, typename ...Types>
469 auto forward(F f, Types &&...args) {
470  f(args...);
471 }
472 
473 forward(f,1, 2,3);
474  * \endcode
475  *
476  * gets expanded by clang as
477  * \code
478 f(args, args, args);
479  * \endcode
480  *
481  * which would obviously not compile. For clang AST dump it is the right thing. For C++ Insights where the resulting
482  * code should be compilable it is not. What this function does is, figure out whether it is a pack expansion and if so,
483  * make the parameters unique, such that \c args becomes \c __args1 to \c __args3.
484  *
485  * The expected type for \c T currently is \c ValueDecl or \c VarDecl.
486  */
487 template<typename T>
488 static std::string GetTemplateParameterPackArgumentName(std::string& name, const T* decl)
489 {
490  if(const auto* parmVarDecl = dyn_cast_or_null<ParmVarDecl>(decl)) {
491  if(const auto& originalType = parmVarDecl->getOriginalType(); not originalType.isNull()) {
492  if(const auto* substTemplateTypeParmType = GetSubstTemplateTypeParmType(originalType.getTypePtrOrNull())) {
493  if(substTemplateTypeParmType->getReplacedParameter()->isParameterPack()) {
494  name = StrCat(BuildInternalVarName(name), parmVarDecl->getFunctionScopeIndex());
495  }
496  }
497  }
498  }
499 
500  return name;
501 }

504 std::string GetName(const DeclRefExpr& declRefExpr)
505 {
506  std::string name{};
507  const auto* declRefDecl = declRefExpr.getDecl();
508  const auto* declCtx = declRefDecl->getDeclContext();
509  const bool isFriend{(declRefDecl->getFriendObjectKind() != Decl::FOK_None)};
510  const bool hasNamespace{(declCtx && (declCtx->isNamespace() || declCtx->isInlineNamespace()) &&
511  !declCtx->isTransparentContext() && !isFriend)};
512 
513  // get the namespace as well
514  if(hasNamespace) {
515  name = details::GetScope(declCtx);
516  } else if(declRefExpr.hasQualifier()) {
517  name = details::GetQualifiedName(*declRefDecl);
518  }
519 
520  if(hasNamespace || !declRefExpr.hasQualifier()) {
521  std::string plainName{GetPlainName(declRefExpr)};
522 
523  // try to handle the special case of a function local static with class type and non trivial destructor. In
524  // this case, as we teared that variable apart, we need to adjust the variable named and add a reinterpret
525  // cast
526  if(IsTrivialStaticClassVarDecl(declRefExpr)) {
527  if(const VarDecl* vd = GetVarDeclFromDeclRefExpr(declRefExpr)) {
528  if(const auto* cxxRecordDecl = vd->getType()->getAsCXXRecordDecl()) {
529  plainName = StrCat(
530  "*reinterpret_cast<", GetName(*cxxRecordDecl), "*>(", BuildInternalVarName(plainName), ")");
531  }
532  }
533  }
534 
535  name.append(plainName);
536  }
537 
538  return GetTemplateParameterPackArgumentName(name, declRefDecl);
539 }

561 {

563

565  std::string ret{kwSpaceNoexcept};
566 
567  if(const auto* expr = func->getNoexceptExpr()) {
568  const auto value = [&] {
569  if(const auto* boolExpr = dyn_cast_or_null<CXXBoolLiteralExpr>(expr)) {
570  return boolExpr->getValue();
571 
572  } else if(const auto* noExceptExpr = dyn_cast_or_null<CXXNoexceptExpr>(expr)) {
573  return noExceptExpr->getValue();
574

576  return EvaluateAsBoolenCondition(*bExpr, decl);
577 
578  } else if(const auto* coExpr = dyn_cast_or_null<ConditionalOperator>(expr)) {
579  return EvaluateAsBoolenCondition(*coExpr, decl);

581 #if IS_CLANG_NEWER_THAN(7)

583  return EvaluateAsBoolenCondition(*cExpr, decl);
584 #endif
585  }
586 
587  Error(expr, "INSIGHTS: Unexpected noexcept expr\n");
588 
589  return false;
590  }();
591 
592  ret += "(";
593 
594  if(value) {
595  ret += "true";
596  } else {
597  ret += "false";
598  }
599 
600  ret += ")";
601  }
602 
603  return ret;
604  }
605
