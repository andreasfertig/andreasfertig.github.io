136  /// \brief Append a argument list to the buffer.
137  ///
138  /// This function takes care of the delimiting ',' between the parameters. The lambda \c lambda is called to each
139  /// argument after the comma was inserted.
140  /// Usage:
141  /// \code
142  /// ForEachArg(parameters, [&](const auto& p) {
143  /// // do something with p
144  /// });
145  /// \endcode
146  template<typename T, typename Lambda>
147  static inline void ForEachArg(const T& arguments, OutputFormatHelper& outputFormatHelper, Lambda&& lambda)
148  {
149  bool first{true};
150  for(const auto& arg : arguments) {
151  if(first) {
152  first = false;
153  } else {
154  outputFormatHelper.Append(", ");
155  }
156 
157  lambda(arg);
158  }
159  }
