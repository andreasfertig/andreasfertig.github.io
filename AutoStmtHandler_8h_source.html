22 /// \brief Show the types behind \c auto
23 ///
24 /// The AutoStmtHandler matches all \c auto statements in global space. That means:
25 ///
26 /// \code
27 /// auto i = 0;
28 /// auto [x,y] = Point{1,2};
29 /// auto lambda = [&]() { return errno; };
30 /// \endcode
31 ///
32 /// The transformation itself shows the fully expanded type behind \c auto and a possible associated statmente like
33 /// lambda or structured binding.
34 /// For example: \code auto i = 0; \endcode
35 /// will give you
36 /// \code
37 /// int i = 0;
38 /// \endcode

41 public:
